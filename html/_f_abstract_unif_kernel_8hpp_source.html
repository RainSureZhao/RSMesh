73 FPoint<FReal> getCellCenter(const FTreeCoordinate coordinate, int inLevel)
74 {
75
76 //Set the boxes width needed
77 FReal widthAtCurrentLevel = BoxWidthLeaf*FReal(1 << (TreeHeight-(inLevel+1)));
78 FReal widthAtCurrentLevelDiv2 = widthAtCurrentLevel/FReal(2.);
79
80 //Set the center real coordinates from box corner and widths.
81 FReal X = BoxCorner.getX() + FReal(coordinate.getX())*widthAtCurrentLevel + widthAtCurrentLevelDiv2;
82 FReal Y = BoxCorner.getY() + FReal(coordinate.getY())*widthAtCurrentLevel + widthAtCurrentLevelDiv2;
83 FReal Z = BoxCorner.getZ() + FReal(coordinate.getZ())*widthAtCurrentLevel + widthAtCurrentLevelDiv2;
84
85 return FPoint<FReal>(X,Y,Z);
86 }
