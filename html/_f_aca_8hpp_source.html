33 template <class FReal>
34 void fACA(FReal *const K,
35 const unsigned int nx, const unsigned int ny,
36 const double eps, FReal* &U, FReal* &V, unsigned int &k)
37 {
38 // control vectors (true if not used, false if used)
39 bool *const r = new bool[nx];
40 bool *const c = new bool[ny];
41 for (unsigned int i=0; i<nx; ++i) r[i] = true;
42 for (unsigned int j=0; j<ny; ++j) c[j] = true;
43
44 // compute Frobenius norm of original Matrix K
45 FReal norm2K = 0;
46 for (unsigned int j=0; j<ny; ++j) {
47 const FReal *const colK = K + j*nx;
48 norm2K += FBlas::scpr(nx, colK, colK);
49 }
50
51 // initialize rank k and UV'
52 k = 0;
53 const unsigned int maxk = (nx + ny) / 2;
54 U = new FReal[nx * maxk];
55 V = new FReal[ny * maxk];
56 FBlas::setzero(nx*maxk, U);
57 FBlas::setzero(ny*maxk, V);
58 FReal norm2R;
59
61 // start fully pivoted ACA
62 do {
63
64 // find max(K) and argmax(K)
65 FReal maxK = 0.;
66 unsigned int pi=0, pj=0;
67 for (unsigned int j=0; j<ny; ++j)
68 if (c[j]) {
69 const FReal *const colK = K + j*nx;
70 for (unsigned int i=0; i<nx; ++i)
71 if (r[i] && maxK < FMath::Abs(colK[i])) {
72 maxK = FMath::Abs(colK[i]);
73 pi = i;
74 pj = j;
75 }
76 }
77
78 // copy pivot cross into U and V
79 FReal *const colU = U + k*nx;
80 FReal *const colV = V + k*ny;
81 const FReal pivot = K[pj*nx + pi];
82 for (unsigned int i=0; i<nx; ++i) if (r[i]) colU[i] = K[pj*nx + i];
83 for (unsigned int j=0; j<ny; ++j) if (c[j]) colV[j] = K[j *nx + pi] / pivot;
84
85 // don't use these cols and rows anymore
86 c[pj] = false;
87 r[pi] = false;
88
89 // subtract k-th outer product from K
90 for (unsigned int j=0; j<ny; ++j)
91 if (c[j]) {
92 FReal *const colK = K + j*nx;
93 FBlas::axpy(nx, FReal(-1. * colV[j]), colU, colK);
94 }
95
96 // compute Frobenius norm of updated K
97 norm2R = 0.0;
98 for (unsigned int j=0; j<ny; ++j)
99 if (c[j]) {
100 const FReal *const colK = K + j*nx;
101 norm2R += FBlas::scpr(nx, colK, colK);
102 }
103
104 // increment rank k
105 ++k ;
106
107 } while (norm2R > eps*eps * norm2K);
109
110 delete [] r;
111 delete [] c;
112 }

140 template <class FReal, typename ComputerType>
141 void pACA(const ComputerType& Computer,
142 const unsigned int nx, const unsigned int ny,
143 const FReal eps, FReal* &U, FReal* &V, unsigned int &k)
144 {
145 // control vectors (true if not used, false if used)
146 bool *const r = new bool[nx];
147 bool *const c = new bool[ny];
148 for (unsigned int i=0; i<nx; ++i) r[i] = true;
149 for (unsigned int j=0; j<ny; ++j) c[j] = true;
150
151 // initialize rank k and UV'
152 k = 0;
153 const FReal eps2 = eps * eps;
154 const unsigned int maxk = (nx + ny) / 2;
155 U = new FReal[nx * maxk];
156 V = new FReal[ny * maxk];
157
158 // initialize norm
159 FReal norm2S(0.);
160 FReal norm2uv(0.);
161
163 // start partially pivoted ACA
164 unsigned int J = 0, I = 0;
165
166 do {
167 FReal *const colU = U + nx*k;
168 FReal *const colV = V + ny*k;
169
171 // compute row I and its residual
172 Computer(I, I+1, 0, ny, colV);
173 r[I] = false;
174 for (unsigned int l=0; l<k; ++l) {
175 FReal *const u = U + nx*l;
176 FReal *const v = V + ny*l;
177 FBlas::axpy(ny, FReal(-1. * u[I]), v, colV);
178 }
179
180 // find max of residual and argmax
181 FReal maxval = 0.;
182 for (unsigned int j=0; j<ny; ++j) {
183 const FReal abs_val = FMath::Abs(colV[j]);
184 if (c[j] && maxval < abs_val) {
185 maxval = abs_val;
186 J = j;
187 }
188 }
189 // find pivot and scale column of V
190 const FReal pivot = FReal(1.) / colV[J];
191 FBlas::scal(ny, pivot, colV);
192
194 // compute col J and its residual
195 Computer(0, nx, J, J+1, colU);
196 c[J] = false;
197 for (unsigned int l=0; l<k; ++l) {
198 FReal *const u = U + nx*l;
199 FReal *const v = V + ny*l;
200 FBlas::axpy(nx, FReal(-1. * v[J]), u, colU);
201 }
202
203 // find max of residual and argmax
204 maxval = 0.0;
205 for (unsigned int i=0; i<nx; ++i) {
206 const FReal abs_val = FMath::Abs(colU[i]);
207 if (r[i] && maxval < abs_val) {
208 maxval = abs_val;
209 I = i;
210 }
211 }
212
214 // increment Frobenius norm: |Sk|^2 += |uk|^2 |vk|^2 + 2 sumj ukuj vjvk
215 FReal normuuvv(0.);
216 for (unsigned int l=0; l<k; ++l)
217 normuuvv += FBlas::scpr(nx, colU, U + nx*l) * FBlas::scpr(ny, V + ny*l, colV);
218 norm2uv = FBlas::scpr(nx, colU, colU) * FBlas::scpr(ny, colV, colV);
219 norm2S += norm2uv + 2*normuuvv;
220
222 // increment low-rank
223 ++k;
224
225 } while (norm2uv > eps2 * norm2S);
226
227 delete [] r;
228 delete [] c;
229 }

244 template <class FReal>
245 void recompress(FReal *const UU, FReal *const VV,
246 const unsigned int nx, const unsigned int ny,
247 const double eps, FReal* &U, FReal* &VT, unsigned int &rank)
248 {
249
250 // needed for the SVD
251 int INFO;
252
253 // QR decomposition (UU=QuRu & VV=QvRv, thus UUVV^t=Qu(RuRv^t)Qv^t)
254 FReal* phi = new FReal [rank*rank];
255 {
256 // needed for QR
257 const unsigned int LWORK = 2*4*rank; // for square matrices
258 //const unsigned int LWORK = 2*std::max(3*minMN+maxMN, 5*minMN);
259 FReal *const WORK = new FReal [LWORK];
260
261 // QR of U and V
262 FReal* tauU = new FReal [rank];
263 INFO = FBlas::geqrf(nx, rank, UU, tauU, LWORK, WORK);
264 assert(INFO==0);
265 FReal* tauV = new FReal [rank];
266 INFO = FBlas::geqrf(ny, rank, VV, tauV, LWORK, WORK);
267 assert(INFO==0);
268 // phi = Ru Rv'
269 FReal* rU = new FReal [2 * rank*rank];
270 FReal* rV = rU + rank*rank;
271 FBlas::setzero(2 * rank*rank, rU);
272 for (unsigned int l=0; l<rank; ++l) {
273 FBlas::copy(l+1, UU + l*nx, rU + l*rank);
274 FBlas::copy(l+1, VV + l*ny, rV + l*rank);
275 }
276 FBlas::gemmt(rank, rank, rank, FReal(1.), rU, rank, rV, rank, phi, rank);
277 delete [] rU;
278
279 // get Qu and Qv
280 INFO = FBlas::orgqr(nx, rank, UU, tauU, LWORK, WORK); // Qu -> UU
281 assert(INFO==0);
282 INFO = FBlas::orgqr(ny, rank, VV, tauV, LWORK, WORK); // Qv -> VV
283 assert(INFO==0);
284 delete [] tauU;
285 delete [] tauV;
286 }
287
288 const unsigned int aca_rank = rank;
289
290 // SVD (of phi=RuRv^t, thus phi=UrSVr^t)
291 FReal *const VrT = new FReal [aca_rank*aca_rank];
292 FReal *const Sr = new FReal [aca_rank];
293 {
294 // needed for SVD
295 const unsigned int LWORK = 2*4*aca_rank; // for square matrices
296 //const unsigned int LWORK = 2*std::max(3*minMN+maxMN, 5*minMN);
297 FReal *const WORK = new FReal [LWORK];
298
299 INFO = FBlas::gesvd(aca_rank, aca_rank, phi, Sr, VrT, aca_rank, LWORK, WORK); // Ur -> phi // Sr -> Sr // Vr^t -> VrT
300 if (INFO!=0){
301 std::stringstream stream;
302 stream << INFO;
303 throw std::runtime_error("SVD did not converge with " + stream.str());
304 }
305 rank = FSvd::getRank(Sr, aca_rank, eps);
306 }
307
308 // store Qu(UrS) in U and Vr^t(Qv^t) in VT
309 {
310 // allocate
311 assert(U ==nullptr);
312 assert(VT==nullptr);
313 U = new FReal[rank*nx];
314 VT = new FReal[rank*ny];
315
316 // (Ur Sigma)
317 for (unsigned int r=0; r<rank; ++r)
318 FBlas::scal(aca_rank, Sr[r], phi + r*aca_rank);
319 delete [] Sr;
320
321 // Qu (Ur Sr) in U
322 FBlas::gemm(nx, aca_rank, rank, FReal(1.), UU, nx, phi, aca_rank, U, nx);
323 delete [] phi;
324
325 // VT=Vr^t Qv^t
326 FBlas::gemmt(rank, aca_rank, ny, FReal(1.), VrT, aca_rank, VV, ny, VT, rank);
327 delete [] VrT;
328
329 }
