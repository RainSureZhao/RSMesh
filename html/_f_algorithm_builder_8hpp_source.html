40 static SimulationProperties BuildKernelSimulationProperties(const int realHeight, const FPoint<FReal> realCenterOfBox, const FReal realDimOfBox,
41 const bool isPeriodic = false, const int inUpperLevel = 0){
42 if( isPeriodic == false ){
43 // Simply return the original data
44 SimulationProperties properties = {realHeight, realCenterOfBox, realDimOfBox};
45 return properties;
46 }
47 else{
48 // Get the new data
49 const int offsetRealTree(inUpperLevel + 3);
50 const FReal extendedBoxWidth = realDimOfBox * FReal(1<<offsetRealTree);
51
52 const FReal originalBoxWidth = realDimOfBox;
53 const FReal originalBoxWidthDiv2 = originalBoxWidth/2.0;
54 const FPoint<FReal> originalBoxCenter = realCenterOfBox;
55
56 const FReal offset = extendedBoxWidth/2;
57 const FPoint<FReal> extendedBoxCenter( originalBoxCenter.getX() - originalBoxWidthDiv2 + offset,
58 originalBoxCenter.getY() - originalBoxWidthDiv2 + offset,
59 originalBoxCenter.getZ() - originalBoxWidthDiv2 + offset);
60
61 const int extendedTreeHeight = realHeight + offsetRealTree;
62
63 SimulationProperties properties = {extendedTreeHeight, extendedBoxCenter, extendedBoxWidth};
64 return properties;
65 }
