106 void increaseSizeIfNeeded(FSize sizeInput = 1){
107 if( nbParticles+(sizeInput-1) >= allocatedParticles ){
108 // allocate memory
109 const FSize moduloParticlesNumber = (MemoryAlignement/sizeof(FReal));
110 allocatedParticles = (FMath::Max(DefaultNbParticles,FSize(FReal(nbParticles+sizeInput)*1.5)) + moduloParticlesNumber - 1) & ~(moduloParticlesNumber-1);
111 // init with 0
112 const size_t allocatedBytes = (sizeof(FReal)*3 + sizeof(AttributeClass)*NbAttributesPerParticle)*allocatedParticles;
113 FReal* newData = reinterpret_cast<FReal*>(FAlignedMemory::AllocateBytes<MemoryAlignement>(allocatedBytes));
114 memset( newData, 0, allocatedBytes);
115 // copy memory
116 const char*const toDelete = reinterpret_cast<const char*>(positions[0]);
117 for(int idx = 0 ; idx < 3 ; ++idx){
118 memcpy(newData + (allocatedParticles * idx), positions[idx], sizeof(FReal) * nbParticles);
119 positions[idx] = newData + (allocatedParticles * idx);
120 for(FSize idxEmpty = nbParticles ; idxEmpty < allocatedParticles ; ++idxEmpty){
121 positions[idx][idxEmpty] = std::numeric_limits<FReal>::max()/2;
122 }
123 }
124 // copy attributes
125 AttributeClass* startAddress = reinterpret_cast<AttributeClass*>(positions[2] + allocatedParticles);
126 for(unsigned idx = 0 ; idx < NbAttributesPerParticle ; ++idx){
127 memcpy(startAddress + (allocatedParticles * idx), attributes[idx], sizeof(AttributeClass) * nbParticles);
128 attributes[idx] = startAddress + (idx * allocatedParticles);
129 }
130 // delete old
131 FAlignedMemory::DeallocBytes(toDelete);
132 }
133 }

355 void removeParticles(const FSize indexesToRemove[], const FSize nbParticlesToRemove){
356 FSize offset = 1;
357 FSize idxIndexes = 1;
358 FSize idxIns = indexesToRemove[0] + 1;
359 for( ; idxIns < nbParticles && idxIndexes < nbParticlesToRemove ; ++idxIns){
360 if( idxIns == indexesToRemove[idxIndexes] ){
361 idxIndexes += 1;
362 offset += 1;
363 }
364 else{
365 for(int idx = 0 ; idx < 3 ; ++idx){
366 positions[idx][idxIns-offset] = positions[idx][idxIns];
367 }
368 for(unsigned idx = 0 ; idx < NbAttributesPerParticle ; ++idx){
369 attributes[idx][idxIns-offset] = attributes[idx][idxIns];

371 }
