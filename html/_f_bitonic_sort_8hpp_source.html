111 /*
112 From :
113 http://web.mst.edu/~ercal/387/P3/pr-proj-3.pdf
114
115 Parallel Bitonic Sort Algorithm for processor Pk (for k := 0 . . . P − 1)
116 d:= log P
117 // cube dimension
118 sort(local − datak ) // sequential sort
119 // Bitonic Sort follows
120 for i:=1 to d do
121 window-id = Most Significant (d-i) bits of Pk
122 for j:=(i-1) down to 0 do
123 if((window-id is even AND j th bit of Pk = 0)
124 OR (window-id is odd AND j th bit of Pk = 1))
125 then call CompareLow(j)
126 else
127 call CompareHigh(j)
128 endif
129 endfor
130 endfor
131 */
132 static void Sort(SortType array[], const IndexType size, const FMpi::FComm& comm){
133 const int np = comm.processCount();
134 const int rank = comm.processId();
135
136 {// Work only for power of 2!
137 int flag = 1;
138 while(!(np & flag)) flag <<= 1;
139 FAssert(np == flag, "Bitonic sort work only with power of 2 for now.")
140 }
141
142 FQuickSort<SortType,IndexType>::QsOmp(array, size, [](const SortType& v1, const SortType& v2){
143 return CompareType(v1) <= CompareType(v2);
144 });
145
146 const int logNp = int(log2(np));
147 for(int bitIdx = 1 ; bitIdx <= logNp ; ++bitIdx){
148 // window-id = Most Significant (d-i) bits of Pk
149 const int diBit = (rank >> bitIdx) & 0x1;
150
151 for(int otherBit = bitIdx - 1 ; otherBit >= 0 ; --otherBit){
152 // if((window-id is even AND j th bit of Pk = 0)
153 // OR (window-id is odd AND j th bit of Pk = 1))
154
155 const int myOtherBit = (rank >> otherBit) & 0x1;
156 const int otherRank = rank ^ (1 << otherBit);
157
158 if( diBit != myOtherBit ){
159 SendMinAndGetMax(array, size, otherRank, comm);
160 }
161 else{
162 SendMaxAndGetMin(array, size, otherRank, comm);
163 }
164 // A merge sort is possible since the array is composed
165 // by two part already sorted, but we want to do this in space
166 FQuickSort<SortType,IndexType>::QsOmp(array, size, [](const SortType& v1, const SortType& v2){
167 return CompareType(v1) <= CompareType(v2);
168 });
169 }
170 }
171 }
