176template <class FReal, int ORDER, class MatrixKernelClass>
177unsigned int
178FChebDenseM2LHandler<FReal, ORDER, MatrixKernelClass>::Compute(const MatrixKernelClass *const MatrixKernel, FReal* &C)
179{
180 // allocate memory and store compressed M2L operators
181 if (C) throw std::runtime_error("Full M2L operators are already set");
182
183 // interpolation points of source (Y) and target (X) cell
184 FPoint<FReal> X[nnodes], Y[nnodes];
185 // set roots of target cell (X)
186 FChebTensor<FReal, order>::setRoots(FPoint<FReal>(0.,0.,0.), FReal(2.), X);
187
188 // allocate memory and compute 316 m2l operators
189 FReal *_C;
190 _C = new FReal [nnodes*nnodes * ninteractions];
191 unsigned int counter = 0;
192 for (int i=-3; i<=3; ++i) {
193 for (int j=-3; j<=3; ++j) {
194 for (int k=-3; k<=3; ++k) {
195 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
196 // set roots of source cell (Y)
197 const FPoint<FReal> cy(FReal(2.*i), FReal(2.*j), FReal(2.*k));
198 FChebTensor<FReal, order>::setRoots(cy, FReal(2.), Y);
199 // evaluate m2l operator
200 for (unsigned int n=0; n<nnodes; ++n)
201 for (unsigned int m=0; m<nnodes; ++m)
202 _C[counter*nnodes*nnodes + n*nnodes + m]
203 = MatrixKernel->evaluate(X[m], Y[n]);
204 // increment interaction counter
205 counter++;
206 }
207 }
208 }
209 }
210 if (counter != ninteractions)
211 throw std::runtime_error("Number of interactions must correspond to 316");
212
213 // svd compression of M2L
214 const unsigned int rank = nnodes;
215
216 // store C
217 counter = 0;
218 C = new FReal [343 * rank*rank];
219 for (int i=-3; i<=3; ++i)
220 for (int j=-3; j<=3; ++j)
221 for (int k=-3; k<=3; ++k) {
222 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
223 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
224 FBlas::copy(rank*rank, _C + counter*rank*rank, C + idx*rank*rank);
225 counter++;
226 } else FBlas::setzero(rank*rank, C + idx*rank*rank);
227 }
228 if (counter != ninteractions)
229 throw std::runtime_error("Number of interactions must correspond to 316");
230 delete [] _C;
231
232 // return low rank
233 return rank;
234}

241template <class FReal, int ORDER, class MatrixKernelClass>
242void
243FChebDenseM2LHandler<FReal, ORDER, MatrixKernelClass>::ComputeAndStoreInBinaryFile(const MatrixKernelClass *const MatrixKernel)
244{
245 // measure time
246 FTic time; time.tic();
247 // start computing process
248 FReal *C;
249 C = nullptr;
250 const unsigned int rank = nnodes;
251 // store into binary file
252 const std::string filename(getFileName());
253 std::ofstream stream(filename.c_str(), std::ios::out | std::ios::binary | std::ios::trunc);
254 if (stream.good()) {
255 stream.seekp(0);
256 // 1) write number of interpolation points (int)
257 int _nnodes = nnodes;
258 stream.write(reinterpret_cast<char*>(&_nnodes), sizeof(int));
259 // 2) write low rank (int)
260 int _rank = rank;
261 stream.write(reinterpret_cast<char*>(&_rank), sizeof(int));
262 // 5) write 343 C (343 * rank*rank * FReal)
263 stream.write(reinterpret_cast<char*>(C), sizeof(FReal)*rank*rank*343);
264 } else throw std::runtime_error("File could not be opened to write");
265 stream.close();
266 // free memory
267 if (C != nullptr) delete [] C;
268 // write info
269 std::cout << "Full M2L operators ("<< rank << ") stored in binary file " << filename
270 << " in " << time.tacAndElapsed() << "sec." << std::endl;
271}

274template <class FReal, int ORDER, class MatrixKernelClass>
275void
276FChebDenseM2LHandler<FReal, ORDER, MatrixKernelClass>::ReadFromBinaryFileAndSet()
277{
278 // measure time
279 FTic time; time.tic();
280 // start reading process
281 if (C) throw std::runtime_error("Full M2L operator already set");
282 const std::string filename(getFileName());
283 std::ifstream stream(filename.c_str(),
284 std::ios::in | std::ios::binary | std::ios::ate);
285 const std::ifstream::pos_type size = stream.tellg();
286 if (size<=0) {
287 std::cout << "Info: The requested binary file " << filename
288 << " does not yet exist. Compute it now ... " << std::endl;
289 this->ComputeAndStoreInBinaryFileAndReadFromFileAndSet();
290 return;
291 }
292 if (stream.good()) {
293 stream.seekg(0);
294 // 1) read number of interpolation points (int)
295 int npts;
296 stream.read(reinterpret_cast<char*>(&npts), sizeof(int));
297 if (npts!=nnodes) throw std::runtime_error("nnodes and npts do not correspond");
298 // 2) read low rank (int)
299 stream.read(reinterpret_cast<char*>(&rank), sizeof(int));
300 // 5) write 343 C (343 * rank*rank * FReal)
301 C = new FReal [343 * rank*rank];
302 stream.read(reinterpret_cast<char*>(C), sizeof(FReal)*rank*rank*343);
303 } else throw std::runtime_error("File could not be opened to read");
304 stream.close();
305 // write info
306 std::cout << "Full M2L operators (" << rank << ") read from binary file "
307 << filename << " in " << time.tacAndElapsed() << "sec." << std::endl;
308}
