70 unsigned int countFlopsP2M(const unsigned int N) const {
71 const unsigned first = N * (18 + (ORDER-2) * 6 + (ORDER-1) * (6 + (ORDER-1) * (6 + (ORDER-1) * 2)));
72 const unsigned W2 = 3 * ORDER*(2*(ORDER-1)-1);
73 const unsigned W4 = 3 * (ORDER*(ORDER-1)*(2*(ORDER-1)-1) + ORDER*ORDER*(2*(ORDER-1)-1));
74 const unsigned W8 = 3 * (2*(ORDER-1)-1) * (ORDER*(ORDER-1)*(ORDER-1) + ORDER*ORDER*(ORDER-1) + nnodes);
75 return first + W2 + W4 + W8 + nnodes*11;

77 unsigned int countFlopsL2PTotal(const unsigned int N) const {
78 const unsigned W0 = nnodes;
79 const unsigned W2 = 3 * (ORDER-1)*ORDER*ORDER * 2*ORDER;
80 const unsigned W4 = 3 * ORDER*(ORDER-1)*(ORDER-1) * 2*ORDER;
81 const unsigned W8 = (ORDER-1)*(ORDER-1)*(ORDER-1) * (2*ORDER-1);
82 const unsigned second = N * (38 + (ORDER-2)*15 + (ORDER-1)*((ORDER-1) * (27 + (ORDER-1) * 16))) + 6;

277template <class FReal, class CellClass,
278 class ContainerClass,
279 class MatrixKernelClass,
280 int ORDER>
281struct FChebFlopsSymKernel<FReal, CellClass, ContainerClass, MatrixKernelClass, ORDER>
282::SymmetryHandler
283{
284 // M2L operators
285 FReal* K[343];
286 int LowRank[343];
287
288 // permutation vectors and permutated indices
289 unsigned int pvectors[343][nnodes];
290 unsigned int pindices[343];
291
292
293 // compute rank
294 unsigned int getRank(const FReal singular_values[], const double eps)
295 {
296 FReal nrm2(0.);
297 for (unsigned int k=0; k<nnodes; ++k)
298 nrm2 += singular_values[k] * singular_values[k];
299
300 FReal nrm2k(0.);
301 for (unsigned int k=nnodes; k>0; --k) {
302 nrm2k += singular_values[k-1] * singular_values[k-1];
303 if (nrm2k > eps*eps * nrm2) return k;
304 }
305 throw std::runtime_error("rank cannot be larger than nnodes");
306 return 0;
307 }
308
309
311 SymmetryHandler(const MatrixKernelClass *const MatrixKernel, const double Epsilon)
312 {
313 // init all 343 item to zero, because effectively only 16 exist
314 for (unsigned int t=0; t<343; ++t) {
315 K[t] = nullptr;
316 LowRank[t] = 0;
317 }
318
319 // set permutation vector and indices
320 const FInterpSymmetries<ORDER> Symmetries;
321 for (int i=-3; i<=3; ++i)
322 for (int j=-3; j<=3; ++j)
323 for (int k=-3; k<=3; ++k)
324 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
325 const unsigned int idx = ((i+3) * 7 + (j+3)) * 7 + (k+3);
326 pindices[idx] = Symmetries.getPermutationArrayAndIndex(i,j,k, pvectors[idx]);
327 }
328
329 // precompute 16 M2L operators
330 this->precomputeSVD(MatrixKernel, Epsilon);
331 }

336 ~SymmetryHandler()

338 for (unsigned int t=0; t<343; ++t)
339 if (K[ t]!=nullptr) delete [] K[ t];
340 }

344private:
345 void precomputeSVD(const MatrixKernelClass *const MatrixKernel, const double Epsilon)
346 {
347 // interpolation points of source (Y) and target (X) cell
348 FPoint<FReal> X[nnodes], Y[nnodes];
349 // set roots of target cell (X)
350 FChebTensor<FReal,ORDER>::setRoots(FPoint<FReal>(0.,0.,0.), FReal(2.), X);
351 // temporary matrix
352 FReal* U = new FReal [nnodes*nnodes];
353
354 // needed for the SVD
355 const unsigned int LWORK = 2 * (3*nnodes + nnodes);
356 FReal *const WORK = new FReal [LWORK];
357 FReal *const VT = new FReal [nnodes*nnodes];
358 FReal *const S = new FReal [nnodes];
359
360 unsigned int counter = 0;
361 for (int i=2; i<=3; ++i) {
362 for (int j=0; j<=i; ++j) {
363 for (int k=0; k<=j; ++k) {
364
365 // assemble matrix
366 const FPoint<FReal> cy(FReal(2.*i), FReal(2.*j), FReal(2.*k));
367 FChebTensor<FReal,ORDER>::setRoots(cy, FReal(2.), Y);
368 for (unsigned int n=0; n<nnodes; ++n)
369 for (unsigned int m=0; m<nnodes; ++m)
370 U[n*nnodes + m] = MatrixKernel->evaluate(X[m], Y[n]);
371
372 // applying weights ////////////////////////////////////////
373 FReal weights[nnodes];
374 FChebTensor<FReal,ORDER>::setRootOfWeights(weights);
375 for (unsigned int n=0; n<nnodes; ++n) {
376 FBlas::scal(nnodes, weights[n], U + n, nnodes); // scale rows
377 FBlas::scal(nnodes, weights[n], U + n * nnodes); // scale cols
378 }
380
381 // truncated singular value decomposition of matrix
382 long long info = FBlas::gesvd(nnodes, nnodes, U, S, VT, nnodes, LWORK, WORK);
383 if (info!=0) throw std::runtime_error("SVD did not converge with " + std::to_string(info));
384 const unsigned int rank = this->getRank(S, Epsilon);
385
386 // store
387 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
388 assert(K[idx]==nullptr);
389 K[idx] = new FReal [2*rank*nnodes];
390 LowRank[idx] = rank;
391 for (unsigned int r=0; r<rank; ++r)
392 FBlas::scal(nnodes, S[r], U + r*nnodes);
393 FBlas::copy(rank*nnodes, U, K[idx]);
394 for (unsigned int r=0; r<rank; ++r)
395 FBlas::copy(nnodes, VT + r, nnodes, K[idx] + rank*nnodes + r*nnodes, 1);
396
397 // un-weighting ////////////////////////////////////////////
398 for (unsigned int n=0; n<nnodes; ++n) {
399 FBlas::scal(rank, FReal(1.) / weights[n], K[idx] + n, nnodes); // scale rows
400 FBlas::scal(rank, FReal(1.) / weights[n], K[idx] + rank*nnodes + n, nnodes); // scale rows
401 }
403
404 std::cout << "(" << i << "," << j << "," << k << ") " << idx <<
405 ", low rank = " << rank << std::endl;
406
407 counter++;
408 }
409 }
410 }
411 std::cout << "num interactions = " << counter << std::endl;
412 delete [] U;
413 delete [] WORK;
414 delete [] VT;
415 delete [] S;
416 }
