216template <class FReal, int ORDER, class MatrixKernelClass>
217unsigned int
218FChebM2LHandler<FReal, ORDER, MatrixKernelClass>::ComputeAndCompress(const MatrixKernelClass *const MatrixKernel,
219 const FReal epsilon,
220 FReal* &U,
221 FReal* &C,
222 FReal* &B)
223{
224 // allocate memory and store compressed M2L operators
225 if (U||C||B) throw std::runtime_error("Compressed M2L operators are already set");
226
227 // interpolation points of source (Y) and target (X) cell
228 FPoint<FReal> X[nnodes], Y[nnodes];
229 // set roots of target cell (X)
230 FChebTensor<FReal, order>::setRoots(FPoint<FReal>(0.,0.,0.), FReal(2.), X);
231
232 // allocate memory and compute 316 m2l operators
233 FReal *_U, *_C, *_B;
234 _U = _B = nullptr;
235 _C = new FReal [nnodes*nnodes * ninteractions];
236 unsigned int counter = 0;
237 for (int i=-3; i<=3; ++i) {
238 for (int j=-3; j<=3; ++j) {
239 for (int k=-3; k<=3; ++k) {
240 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
241 // set roots of source cell (Y)
242 const FPoint<FReal> cy(FReal(2.*i), FReal(2.*j), FReal(2.*k));
243 FChebTensor<FReal, order>::setRoots(cy, FReal(2.), Y);
244 // evaluate m2l operator
245 for (unsigned int n=0; n<nnodes; ++n)
246 for (unsigned int m=0; m<nnodes; ++m)
247 _C[counter*nnodes*nnodes + n*nnodes + m]
248 = MatrixKernel->evaluate(X[m], Y[n]);
249 // increment interaction counter
250 counter++;
251 }
252 }
253 }
254 }
255 if (counter != ninteractions)
256 throw std::runtime_error("Number of interactions must correspond to 316");
257
258
260 FReal weights[nnodes];
261 FChebTensor<FReal, order>::setRootOfWeights(weights);
262 for (unsigned int i=0; i<316; ++i)
263 for (unsigned int n=0; n<nnodes; ++n) {
264 FBlas::scal(nnodes, weights[n], _C+i*nnodes*nnodes + n, nnodes); // scale rows
265 FBlas::scal(nnodes, weights[n], _C+i*nnodes*nnodes + n * nnodes); // scale cols
266 }
268
269 // svd compression of M2L
270 const unsigned int rank = Compress<FReal, ORDER>(epsilon, ninteractions, _U, _C, _B);
271 if (!(rank>0)) throw std::runtime_error("Low rank must be larger then 0!");
272
273
274 // store U
275 U = new FReal [nnodes * rank];
276 FBlas::copy(rank*nnodes, _U, U);
277 delete [] _U;
278 // store B
279 B = new FReal [nnodes * rank];
280 FBlas::copy(rank*nnodes, _B, B);
281 delete [] _B;
282 // store C
283 counter = 0;
284 C = new FReal [343 * rank*rank];
285 for (int i=-3; i<=3; ++i)
286 for (int j=-3; j<=3; ++j)
287 for (int k=-3; k<=3; ++k) {
288 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
289 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
290 FBlas::copy(rank*rank, _C + counter*rank*rank, C + idx*rank*rank);
291 counter++;
292 } else FBlas::setzero(rank*rank, C + idx*rank*rank);
293 }
294 if (counter != ninteractions)
295 throw std::runtime_error("Number of interactions must correspond to 316");
296 delete [] _C;
297
298
300 for (unsigned int n=0; n<nnodes; ++n) {
301 FBlas::scal(rank, FReal(1.) / weights[n], U+n, nnodes); // scale rows
302 FBlas::scal(rank, FReal(1.) / weights[n], B+n, nnodes); // scale rows
303 }
305
306
307 // return low rank
308 return rank;
309}

316template <class FReal, int ORDER, class MatrixKernelClass>
317void
318FChebM2LHandler<FReal, ORDER, MatrixKernelClass>::ComputeAndCompressAndStoreInBinaryFile(const MatrixKernelClass *const MatrixKernel, const FReal epsilon)
319{
320 // measure time
321 FTic time; time.tic();
322 // start computing process
323 FReal *U, *C, *B;
324 U = C = B = nullptr;
325 const unsigned int rank = ComputeAndCompress(MatrixKernel, epsilon, U, C, B);
326 // store into binary file
327 const std::string filename(getFileName(epsilon));
328 std::ofstream stream(filename.c_str(),
329 std::ios::out | std::ios::binary | std::ios::trunc);
330 if (stream.good()) {
331 stream.seekp(0);
332 // 1) write number of interpolation points (int)
333 int _nnodes = nnodes;
334 stream.write(reinterpret_cast<char*>(&_nnodes), sizeof(int));
335 // 2) write low rank (int)
336 int _rank = rank;
337 stream.write(reinterpret_cast<char*>(&_rank), sizeof(int));
338 // 3) write U (rank*nnodes * FReal)
339 stream.write(reinterpret_cast<char*>(U), sizeof(FReal)*rank*nnodes);
340 // 4) write B (rank*nnodes * FReal)
341 stream.write(reinterpret_cast<char*>(B), sizeof(FReal)*rank*nnodes);
342 // 5) write 343 C (343 * rank*rank * FReal)
343 stream.write(reinterpret_cast<char*>(C), sizeof(FReal)*rank*rank*343);
344 } else throw std::runtime_error("File could not be opened to write");
345 stream.close();
346 // free memory
347 if (U != nullptr) delete [] U;
348 if (B != nullptr) delete [] B;
349 if (C != nullptr) delete [] C;
350 // write info
351 std::cout << "Compressed M2L operators ("<< rank << ") stored in binary file " << filename
352 << " in " << time.tacAndElapsed() << "sec." << std::endl;
353}

359{
360 // measure time
361 FTic time; time.tic();
362 // start reading process
363 if (U||C||B) throw std::runtime_error("Compressed M2L operator already set");
364 const std::string filename(getFileName(epsilon));
365 std::ifstream stream(filename.c_str(),
366 std::ios::in | std::ios::binary | std::ios::ate);
367 const std::ifstream::pos_type size = stream.tellg();
368 if (size<=0) {
369 std::cout << "Info: The requested binary file " << filename
370 << " does not yet exist. Compute it now ... " << std::endl;
371 this->ComputeAndCompressAndStoreInBinaryFileAndReadFromFileAndSet();
372 return;
373 }
374 if (stream.good()) {
375 stream.seekg(0);
376 // 1) read number of interpolation points (int)
377 int npts;
378 stream.read(reinterpret_cast<char*>(&npts), sizeof(int));
379 if (npts!=nnodes) throw std::runtime_error("nnodes and npts do not correspond");
380 // 2) read low rank (int)
381 stream.read(reinterpret_cast<char*>(&rank), sizeof(int));
382 // 3) write U (rank*nnodes * FReal)
383 U = new FReal [rank*nnodes];
384 stream.read(reinterpret_cast<char*>(U), sizeof(FReal)*rank*nnodes);
385 // 4) write B (rank*nnodes * FReal)
386 B = new FReal [rank*nnodes];
387 stream.read(reinterpret_cast<char*>(B), sizeof(FReal)*rank*nnodes);
388 // 5) write 343 C (343 * rank*rank * FReal)
389 C = new FReal [343 * rank*rank];
390 stream.read(reinterpret_cast<char*>(C), sizeof(FReal)*rank*rank*343);
391 } else throw std::runtime_error("File could not be opened to read");
392 stream.close();
393 // write info
394 std::cout << "Compressed M2L operators (" << rank << ") read from binary file "
395 << filename << " in " << time.tacAndElapsed() << "sec." << std::endl;
396}

437template <class FReal, int ORDER>
438unsigned int Compress(const FReal epsilon, const unsigned int ninteractions,
439 FReal* &U, FReal* &C, FReal* &B)
440{
441 // compile time constants
442 enum {order = ORDER,
443 nnodes = TensorTraits<ORDER>::nnodes};
444
445 // init SVD
446 const unsigned int LWORK = 2 * (3*nnodes + ninteractions*nnodes);
447 FReal *const WORK = new FReal [LWORK];
448
449 // K_col ///////////////////////////////////////////////////////////
450 FReal *const K_col = new FReal [ninteractions * nnodes*nnodes];
451 for (unsigned int i=0; i<ninteractions; ++i)
452 for (unsigned int j=0; j<nnodes; ++j)
453 FBlas::copy(nnodes,
454 C + i*nnodes*nnodes + j*nnodes,
455 K_col + j*ninteractions*nnodes + i*nnodes);
456 // singular value decomposition
457 FReal *const Q = new FReal [nnodes*nnodes];
458 FReal *const S = new FReal [nnodes];
459 const unsigned int info_col
460 = FBlas::gesvd(ninteractions*nnodes, nnodes, K_col, S, Q, nnodes,
461 LWORK, WORK);
462 if (info_col!=0){
463 std::stringstream stream;
464 stream << info_col;
465 throw std::runtime_error("SVD did not converge with " + stream.str());
466 }
467 delete [] K_col;
468 const unsigned int k_col = FSvd::getRank<FReal, ORDER>(S, epsilon);
469
470 // Q' -> B
471 B = new FReal [nnodes*k_col];
472 for (unsigned int i=0; i<k_col; ++i)
473 FBlas::copy(nnodes, Q+i, nnodes, B+i*nnodes, 1);
474
475 // K_row //////////////////////////////////////////////////////////////
476 FReal *const K_row = C;
477
478 const unsigned int info_row
479 = FBlas::gesvdSO(nnodes, ninteractions*nnodes, K_row, S, Q, nnodes,
480 LWORK, WORK);
481 if (info_row!=0){
482 std::stringstream stream;
483 stream << info_row;
484 throw std::runtime_error("SVD did not converge with " + stream.str());
485 }
486 const unsigned int k_row = FSvd::getRank<FReal, ORDER>(S, epsilon);
487 delete [] WORK;
488
489 // Q -> U
490 U = Q;
491
492 // V' -> V
493 FReal *const V = new FReal [nnodes*ninteractions * k_row];
494 for (unsigned int i=0; i<k_row; ++i)
495 FBlas::copy(nnodes*ninteractions, K_row+i, nnodes,
496 V+i*nnodes*ninteractions, 1);
497
498 // rank k(epsilon) /////////////////////////////////////////////////////
499 const unsigned int k = k_row < k_col ? k_row : k_col;
500
501 // C_row ///////////////////////////////////////////////////////////
502 C = new FReal [ninteractions * k*k];
503 for (unsigned int i=0; i<k; ++i) {
504 FBlas::scal(nnodes*ninteractions, S[i], V + i*nnodes*ninteractions);
505 for (unsigned int m=0; m<ninteractions; ++m)
506 for (unsigned int j=0; j<k; ++j)
507 C[m*k*k + j*k + i]
508 = FBlas::scpr(nnodes,
509 V + i*nnodes*ninteractions + m*nnodes,
510 B + j*nnodes);
511 }
512
513 delete [] V;
514 delete [] S;
515 delete [] K_row;
516
517 return k;
518}
