69 void allocateMemoryForPermutedExpansions()
70 {
71 assert(Loc==nullptr && Mul==nullptr && countExp==nullptr);
72 Loc = new FReal* [343];
73 Mul = new FReal* [343];
74 countExp = new unsigned int [343];
75
76 // set all 343 to nullptr
77 for (unsigned int idx=0; idx<343; ++idx) {
78 Mul[idx] = Loc[idx] = nullptr;
79 }
80
81 // init only 16 of 343 possible translations due to symmetries
82 for (int i=2; i<=3; ++i)
83 for (int j=0; j<=i; ++j)
84 for (int k=0; k<=j; ++k) {
85 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
86 assert(Mul[idx]==nullptr || Loc[idx]==nullptr);
87 Mul[idx] = new FReal [24 * nnodes];
88 Loc[idx] = new FReal [24 * nnodes];
89 }
90 }

210 void M2L(CellClass* const FRestrict TargetCell, const CellClass* SourceCells[],
211 const int neighborPositions[], const int inSize, const int TreeLevel) override {
212#ifdef LOG_TIMINGS
213 time.tic();
214#endif
215 for(int idxRhs = 0 ; idxRhs < NVALS ; ++idxRhs){
216 // permute and copy multipole expansion
217 memset(countExp, 0, sizeof(int) * 343);
218 for(int idxExistingNeigh = 0 ; idxExistingNeigh < inSize ; ++idxExistingNeigh){
219 const int idx = neighborPositions[idxExistingNeigh];
220 const unsigned int pidx = SymHandler->pindices[idx];
221 const unsigned int count = (countExp[pidx])++;
222 FReal *const mul = Mul[pidx] + count*nnodes;
223 const unsigned int *const pvec = SymHandler->pvectors[idx];
224 const FReal *const MultiExp = SourceCells[idxExistingNeigh]->getMultipole(idxRhs);
225
226 /*
227 // no loop unrolling
228 for (unsigned int n=0; n<nnodes; ++n)
229 mul[pvec[n]] = MultiExp[n];
230 */
231
232 // explicit loop unrolling
233 for (unsigned int n=0; n<nnodes/4 * 4; n+=4) {
234 mul[pvec[n ]] = MultiExp[n];
235 mul[pvec[n+1]] = MultiExp[n+1];
236 mul[pvec[n+2]] = MultiExp[n+2];
237 mul[pvec[n+3]] = MultiExp[n+3];
238 }
239 for (unsigned int n=nnodes/4 * 4; n<nnodes; ++n){
240 mul[pvec[n]] = MultiExp[n];
241 }
242 }
243
244#ifdef LOG_TIMINGS
245 t_m2l_1 += time.tacAndElapsed();
246#endif
247
248
249#ifdef COUNT_BLOCKED_INTERACTIONS
250 unsigned int count_lidx = 0;
251 unsigned int count_interactions = 0;
252 for (unsigned int idx=0; idx<343; ++idx)
253 count_interactions += countExp[idx];
254 if (count_interactions==189) {
255 for (unsigned int idx=0; idx<343; ++idx) {
256 if (countExp[idx])
257 std::cout << "gidx = " << idx << " gives lidx = " << count_lidx++ << " and has "
258 << countExp[idx] << " interactions" << std::endl;
259 }
260 std::cout << std::endl;
261 }
262#endif
263
264
265#ifdef LOG_TIMINGS
266 time.tic();
267#endif
268
269 // multiply (mat-mat-mul)
270 FReal Compressed [nnodes * 24];
271 const FReal scale = MatrixKernel->getScaleFactor(AbstractBaseClass::BoxWidth, TreeLevel);
272 for (unsigned int pidx=0; pidx<343; ++pidx) {
273 const unsigned int count = countExp[pidx];
274 if (count) {
275 const unsigned int rank = SymHandler->getLowRank(TreeLevel, pidx);
276 // rank * count * (2*nnodes-1) flops
277 FBlas::gemtm(nnodes, rank, count, FReal(1.),
278 const_cast<FReal*>(SymHandler->getK(TreeLevel, pidx))+rank*nnodes,
279 nnodes, Mul[pidx], nnodes, Compressed, rank);
280 // nnodes *count * (2*rank-1) flops
281 FBlas::gemm( nnodes, rank, count, scale,
282 const_cast<FReal*>(SymHandler->getK(TreeLevel, pidx)),
283 nnodes, Compressed, rank, Loc[pidx], nnodes);
284 }
285 }
286
287#ifdef LOG_TIMINGS
288 t_m2l_2 += time.tacAndElapsed();
289#endif
290
291#ifdef LOG_TIMINGS
292 time.tic();
293#endif
294
295 // permute and add contribution to local expansions
296 FReal *const LocalExpansion = TargetCell->getLocal(idxRhs);
297 memset(countExp, 0, sizeof(int) * 343);
298 for(int idxExistingNeigh = 0 ; idxExistingNeigh < inSize ; ++idxExistingNeigh){
299 const int idx = neighborPositions[idxExistingNeigh];
300 const unsigned int pidx = SymHandler->pindices[idx];
301 const unsigned int count = (countExp[pidx])++;
302 const FReal *const loc = Loc[pidx] + count*nnodes;
303 const unsigned int *const pvec = SymHandler->pvectors[idx];
304
305 /*
306 // no loop unrolling
307 for (unsigned int n=0; n<nnodes; ++n)
308 LocalExpansion[n] += loc[pvec[n]];
309 */
310
311 // explicit loop unrolling
312 for (unsigned int n=0; n<nnodes/4 * 4; n+=4) {
313 LocalExpansion[n ] += loc[pvec[n ]];
314 LocalExpansion[n+1] += loc[pvec[n+1]];
315 LocalExpansion[n+2] += loc[pvec[n+2]];
316 LocalExpansion[n+3] += loc[pvec[n+3]];
317 }
318 for (unsigned int n=nnodes/4 * 4; n<nnodes; ++n)
319 LocalExpansion[n] += loc[pvec[n]];
320
321 }
322
323#ifdef LOG_TIMINGS
324 t_m2l_3 += time.tacAndElapsed();
325#endif
326 }
