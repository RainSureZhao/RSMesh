396 // init all 343 item to zero, because effectively only 16 exist

402 K[l] = new FReal* [343];
403 LowRank[l] = new int [343];
404 for (unsigned int t=0; t<343; ++t) {
405 K[l][t] = nullptr;
406 LowRank[l][t] = 0;
407 }
408 }

410
411 // set permutation vector and indices
412 const FInterpSymmetries<ORDER> Symmetries;
413 for (int i=-3; i<=3; ++i)
414 for (int j=-3; j<=3; ++j)
415 for (int k=-3; k<=3; ++k) {
416 const unsigned int idx = ((i+3) * 7 + (j+3)) * 7 + (k+3);
417 pindices[idx] = 0;
418 if (abs(i)>1 || abs(j)>1 || abs(k)>1)
419 pindices[idx] = Symmetries.getPermutationArrayAndIndex(i,j,k, pvectors[idx]);
420 }
421
422 // precompute 16 M2L operators at all levels having far-field interactions
423 FReal CellWidth = RootCellWidth / FReal(2.); // at level 1
424 CellWidth /= FReal(2.); // at level 2
425 for (unsigned int l=2; l<TreeHeight; ++l) {
426 precompute<FReal,ORDER>(MatrixKernel, CellWidth, Epsilon, K[l], LowRank[l]);
427 CellWidth /= FReal(2.); // at level l+1
428 }
429 }

471template <class FReal, int ORDER, typename MatrixKernelClass>
472static void ComputeAndCompressAndStoreInBinaryFile(const MatrixKernelClass *const MatrixKernel, const FReal Epsilon)
473{
474 static const unsigned int nnodes = ORDER*ORDER*ORDER;
475
476 // compute and compress ////////////
477 FReal* K[343];
478 int LowRank[343];
479 for (unsigned int idx=0; idx<343; ++idx) { K[idx] = nullptr; LowRank[idx] = 0; }
480 precompute<FReal,ORDER>(MatrixKernel, FReal(2.), Epsilon, K, LowRank);
481
482 // write to binary file ////////////
483 FTic time; time.tic();
484 // start computing process
485 const char precision = (typeid(FReal)==typeid(double) ? 'd' : 'f');
486 std::stringstream sstream;
487 sstream << "sym2l_" << precision << "_o" << ORDER << "_e" << Epsilon << ".bin";
488 const std::string filename(sstream.str());
489 std::ofstream stream(filename.c_str(),
490 std::ios::out | std::ios::binary | std::ios::trunc);
491 if (stream.good()) {
492 stream.seekp(0);
493 for (unsigned int idx=0; idx<343; ++idx)
494 if (K[idx]!=nullptr) {
495 // 1) write index
496 stream.write(reinterpret_cast<char*>(&idx), sizeof(int));
497 // 2) write low rank (int)
498 int rank = LowRank[idx];
499 stream.write(reinterpret_cast<char*>(&rank), sizeof(int));
500 // 3) write U and V (both: rank*nnodes * FReal)
501 FReal *const U = K[idx];
502 FReal *const V = K[idx] + rank*nnodes;
503 stream.write(reinterpret_cast<char*>(U), sizeof(FReal)*rank*nnodes);
504 stream.write(reinterpret_cast<char*>(V), sizeof(FReal)*rank*nnodes);
505 }
506 } else throw std::runtime_error("File could not be opened to write");
507 stream.close();
508 // write info
509 // std::cout << "Compressed M2L operators stored in binary file " << filename
510 // << " in " << time.tacAndElapsed() << "sec." << std::endl;
511
512 // free memory /////////////////////
513 for (unsigned int t=0; t<343; ++t) if (K[t]!=nullptr) delete [] K[t];
514}

521template <class FReal, int ORDER>
522void ReadFromBinaryFile(const FReal Epsilon, FReal* K[343], int LowRank[343])
523{
524 // compile time constants
525 const unsigned int nnodes = ORDER*ORDER*ORDER;
526
527 // find filename
528 const char precision = (typeid(FReal)==typeid(double) ? 'd' : 'f');
529 std::stringstream sstream;
530 sstream << "sym2l_" << precision << "_o" << ORDER << "_e" << Epsilon << ".bin";
531 const std::string filename(sstream.str());
532
533 // read binary file
534 std::ifstream istream(filename.c_str(),
535 std::ios::in | std::ios::binary | std::ios::ate);
536 const std::ifstream::pos_type size = istream.tellg();
537 if (size<=0) throw std::runtime_error("The requested binary file does not yet exist. Exit.");
538
539 if (istream.good()) {
540 istream.seekg(0);
541 // 1) read index (int)
542 int _idx;
543 istream.read(reinterpret_cast<char*>(&_idx), sizeof(int));
544 // loop to find 16 compressed m2l operators
545 for (int idx=0; idx<343; ++idx) {
546 K[idx] = nullptr;
547 LowRank[idx] = 0;
548 // if it exists
549 if (idx == _idx) {
550 // 2) read low rank (int)
551 int rank;
552 istream.read(reinterpret_cast<char*>(&rank), sizeof(int));
553 LowRank[idx] = rank;
554 // 3) read U and V (both: rank*nnodes * FReal)
555 K[idx] = new FReal [2*rank*nnodes];
556 FReal *const U = K[idx];
557 FReal *const V = K[idx] + rank*nnodes;
558 istream.read(reinterpret_cast<char*>(U), sizeof(FReal)*rank*nnodes);
559 istream.read(reinterpret_cast<char*>(V), sizeof(FReal)*rank*nnodes);
560
561 // 1) read next index
562 istream.read(reinterpret_cast<char*>(&_idx), sizeof(int));
563 }
564 }
565 } else throw std::runtime_error("File could not be opened to read");
566 istream.close();
567}
