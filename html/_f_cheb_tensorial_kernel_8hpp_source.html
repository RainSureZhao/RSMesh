111 const int neighborPositions[], const int inSize, const int TreeLevel) override {
112 // scale factor for homogeneous case
113 const FReal CellWidth(AbstractBaseClass::BoxWidth / FReal(FMath::pow(2, TreeLevel)));
114 const FReal ExtendedCellWidth(CellWidth + AbstractBaseClass::BoxWidthExtension);
115 const FReal scale(MatrixKernel->getScaleFactor(ExtendedCellWidth));
116
117 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
118 for (int idxLhs=0; idxLhs < nLhs; ++idxLhs){
119 // update local index
120 const int idxLoc = idxV*nLhs + idxLhs;
121
122 FReal *const localExpansion = TargetCell->getLocal(idxLoc);
123
124 // update idxRhs
125 const int idxRhs = idxLhs % nPv;
126 // update multipole index
127 const int idxMul = idxV*nRhs + idxRhs;
128
129 // get index in matrix kernel
130 const unsigned int d = MatrixKernel->getPosition(idxLhs);
131
132 for(int idxExistingNeigh = 0 ; idxExistingNeigh < inSize ; ++idxExistingNeigh){
133 const int idx = neighborPositions[idxExistingNeigh];
134 M2LHandler->applyC(idx, TreeLevel, scale, d,
135 SourceCells[idxExistingNeigh]->getMultipole(idxMul),
136 localExpansion);
137 }
138 }// NLHS=NPOT*NPV
