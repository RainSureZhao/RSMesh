174 FChebTensorialM2LHandler(const MatrixKernelClass *const MatrixKernel, const unsigned int inTreeHeight, const FReal inRootCellWidth, const FReal inCellWidthExtension)
175 : TreeHeight(inTreeHeight),
176 RootCellWidth(inRootCellWidth),
177 CellWidthExtension(inCellWidthExtension)
178 {
179 // measure time
180 FTic time; time.tic();
181
182 // allocate rank
183 rank = new unsigned int[TreeHeight];
184
185 // allocate C
186 C = new FReal**[TreeHeight];
187 for (unsigned int l=0; l<TreeHeight; ++l){
188 C[l] = new FReal*[ncmp];
189 for (unsigned int d=0; d<ncmp; ++d)
190 C[l][d]=nullptr;
191 }
192
193 for (unsigned int l=0; l<TreeHeight; ++l) {
194 for (unsigned int d=0; d<ncmp; ++d)
195 if (C[l][d]) throw std::runtime_error("Compressed M2L operator already set");
196 }
197
198 // Compute matrix of interactions at each level !! (since non homog)
199 FReal CellWidth = RootCellWidth / FReal(2.); // at level 1
200 CellWidth /= FReal(2.); // at level 2
201 rank[0]=rank[1]=0;
202 for (unsigned int l=2; l<TreeHeight; ++l) {
203 // compute m2l operator on extended cell
204 rank[l] = Compute<FReal, order>(MatrixKernel, CellWidth, CellWidthExtension, C[l]);
205 // update cell width
206 CellWidth /= FReal(2.); // at level l+1
207 }
208 unsigned long sizeM2L = (TreeHeight-2)*343*ncmp*rank[2]*rank[2]*sizeof(FReal);
209
210 // write info
211 std::cout << "Compute and Set full M2L operators of " << TreeHeight-2 << " levels ("<< long(sizeM2L) <<" Bytes) in "
212 << time.tacAndElapsed() << "sec." << std::endl;
213 }

220 if (C[l][d] != nullptr) delete [] C[l][d];
221 }

262template <class FReal, int ORDER, class MatrixKernelClass>
263unsigned int Compute(const MatrixKernelClass *const MatrixKernel,
264 const FReal CellWidth,
265 const FReal CellWidthExtension,
266 FReal** &C)
267{
268 // PB: need to redefine some constant since not function from m2lhandler class
269 const unsigned int order = ORDER;
270 const unsigned int nnodes = TensorTraits<ORDER>::nnodes;
271 const unsigned int ninteractions = 316;
272 const unsigned int ncmp = MatrixKernelClass::NCMP;
273
274 // allocate memory and store compressed M2L operators
275 for (unsigned int d=0; d<ncmp; ++d)
276 if (C[d]) throw std::runtime_error("Compressed M2L operators are already set");
277
278 // interpolation points of source (Y) and target (X) cell
279 FPoint<FReal> X[nnodes], Y[nnodes];
280 // set roots of target cell (X)
281 const FReal ExtendedCellWidth(CellWidth+CellWidthExtension);
282 FChebTensor<FReal,order>::setRoots(FPoint<FReal>(0.,0.,0.), ExtendedCellWidth, X);
283
284 // allocate memory and compute 316 m2l operators
285 FReal** _C;
286 _C = new FReal* [ncmp];
287 for (unsigned int d=0; d<ncmp; ++d)
288 _C[d] = new FReal [nnodes*nnodes * ninteractions];
289
290 unsigned int counter = 0;
291 for (int i=-3; i<=3; ++i) {
292 for (int j=-3; j<=3; ++j) {
293 for (int k=-3; k<=3; ++k) {
294 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
295 // set roots of source cell (Y)
296 const FPoint<FReal> cy(CellWidth*FReal(i), CellWidth*FReal(j), CellWidth*FReal(k));
297 FChebTensor<FReal,order>::setRoots(cy, ExtendedCellWidth, Y);
298
299 // evaluate m2l operator
300 for (unsigned int n=0; n<nnodes; ++n)
301 for (unsigned int m=0; m<nnodes; ++m){
302
303 // Compute current M2L interaction (block matrix)
304 FReal* block;
305 block = new FReal[ncmp];
306 MatrixKernel->evaluateBlock(X[m], Y[n], block);
307
308 // Copy block in C
309 for (unsigned int d=0; d<ncmp; ++d)
310 _C[d][counter*nnodes*nnodes + n*nnodes + m] = block[d];
311
312 delete [] block;
313
314 }
315
316 // increment interaction counter
317 counter++;
318 }
319 }
320 }
321 }
322 if (counter != ninteractions)
323 throw std::runtime_error("Number of interactions must correspond to 316");
324
325
326 // Copy M2L operators
327 const unsigned int rank = nnodes; //PB: dense Chebyshev
328 if (!(rank>0)) throw std::runtime_error("Size must be larger than 0!");
329
330 // store C
331 counter = 0;
332 for (unsigned int d=0; d<ncmp; ++d)
333 C[d] = new FReal [343 * rank*rank];
334 for (int i=-3; i<=3; ++i)
335 for (int j=-3; j<=3; ++j)
336 for (int k=-3; k<=3; ++k) {
337 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
338 if (abs(i)>1 || abs(j)>1 || abs(k)>1) {
339 for (unsigned int d=0; d<ncmp; ++d)
340 FBlas::copy(rank*rank, _C[d] + counter*rank*rank, C[d] + idx*rank*rank);
341 counter++;
342 } else {
343 for (unsigned int d=0; d<ncmp; ++d)
344 FBlas::setzero(rank*rank, C[d] + idx*rank*rank);
345 }
346 }
347 if (counter != ninteractions)
348 throw std::runtime_error("Number of interactions must correspond to 316");
349 for (unsigned int d=0; d<ncmp; ++d)
350 delete [] _C[d];
351
352 // return low rank
353 return rank;
354}
