43 static std::vector<Package> GetPackToSend(const std::pair<size_t, size_t> myCurrentInterval,
44 const std::vector< std::pair<size_t,size_t> >& allObjectives){
45 std::vector<Package> packToSend;
46
47 unsigned int idxProc = 0;
48
49 // Find the first proc to send to
50 while( idxProc != allObjectives.size()
51 && allObjectives[idxProc].second < myCurrentInterval.first){
52 idxProc += 1;
53 }
54
55 // We will from the first element for sure
56 size_t currentElement = 0;
57
58 // Check each proc to send to
59 while( idxProc != allObjectives.size()
60 && allObjectives[idxProc].first < myCurrentInterval.second){
61 Package pack;
62 pack.idProc = idxProc;
63 // The current element must start where the previous one end
64 FAssertLF(allObjectives[idxProc].first < myCurrentInterval.first
65 || currentElement == allObjectives[idxProc].first - myCurrentInterval.first );
66 pack.elementFrom = currentElement;
67 pack.elementTo = Min(allObjectives[idxProc].second , myCurrentInterval.second) - myCurrentInterval.first;
68 // Next time give from the previous end
69 currentElement = pack.elementTo;
70
71 if(pack.elementTo - pack.elementFrom != 0){
72 packToSend.push_back(pack);
73 }
74 // Progress
75 idxProc += 1;
76 }
77 FAssertLF(currentElement == myCurrentInterval.second - myCurrentInterval.first);
78
79 return packToSend;
80 }
