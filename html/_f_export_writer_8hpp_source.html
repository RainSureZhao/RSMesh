269template <class FReal>
270void driverExportData(std::string & filename, const FReal * particles , const FSize NbPoints, const FSize nbDataPerParticle=4){
271 //
272 std::ofstream file( filename.c_str(), std::ofstream::out);
273 // open particle file
274 std::cout << "Write "<< NbPoints<<" Particles in file " << filename <<std::endl;
275
276 if(filename.find(".vtp") != std::string::npos) {
277 exportVTKxml( file, particles, NbPoints,nbDataPerParticle) ;
278 }
279 else if(filename.find(".vtk")!=std::string::npos ) {
280 exportVTK( file, particles, NbPoints,nbDataPerParticle) ;
281 }
282 else if(filename.find(".cvs")!=std::string::npos ) {
283 exportCVS( file, particles, NbPoints,nbDataPerParticle) ;
284 }
285 else if(filename.find(".cosmo")!=std::string::npos ) {
286 exportCOSMOS( file, particles, NbPoints) ;
287 }
288 else {
289 std::cout << "Output file not allowed only .cvs, .cosmo, .vtk or .vtp extensions" <<std::endl;
290 std::exit ( EXIT_FAILURE) ;
291 }
292}
