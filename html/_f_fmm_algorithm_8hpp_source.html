94 void bottomPass(){
95 FLOG( FLog::Controller.write("\tStart Bottom Pass\n").write(FLog::Flush) );
96 FLOG(FTic counterTime);
97 FLOG(FTic computationCounter);
98
99 typename OctreeClass::Iterator octreeIterator(tree);
100
101 // Iterate on leafs
102 octreeIterator.gotoBottomLeft();
103 do{
104 // We need the current cell that represents the leaf
105 // and the list of particles
106 FLOG(computationCounter.tic());
107 kernels->P2M( octreeIterator.getCurrentCell() , octreeIterator.getCurrentListSrc());
108 FLOG(computationCounter.tac());
109 } while(octreeIterator.moveRight());
110
111 FLOG( FLog::Controller << "\tFinished (@Bottom Pass (P2M) = " << counterTime.tacAndElapsed() << " s)\n" );

135

138 FLOG(FTic counterTimeLevel);

270 octreeIterator.gotoBottomLeft();

273 int neighborPositions[26];

275 do{
276 if(l2pEnabled){
277 FLOG(computationCounterL2P.tic());
278 kernels->L2P(octreeIterator.getCurrentCell(), octreeIterator.getCurrentListTargets());
279 FLOG(computationCounterL2P.tac());
280 }
281 if(p2pEnabled){
282 // need the current particles and neighbors particles
283 const int counter = tree->getLeafsNeighbors(neighbors, neighborPositions, octreeIterator.getCurrentGlobalCoordinate(),heightMinusOne);
284 FLOG(computationCounterP2P.tic());
285 kernels->P2P(octreeIterator.getCurrentGlobalCoordinate(),octreeIterator.getCurrentListTargets(),
286 octreeIterator.getCurrentListSrc(), neighbors, neighborPositions, counter);
287 FLOG(computationCounterP2P.tac());
288 }
289 } while(octreeIterator.moveRight());
290
291
