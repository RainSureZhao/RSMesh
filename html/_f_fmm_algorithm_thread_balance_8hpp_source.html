156 void generateIntervalFromWorkload(std::vector<Workload>* intervals, const FSize totalWork,
157 WorkloadTemp* workPerElement, const FSize nbElements) const {
158 // Now split between thread
159 (*intervals).resize(MaxThreads);
160
161 // Ideally each thread will have this
162 const FSize idealWork = (totalWork/MaxThreads);
164
165 // Assign default value for first thread
166 int idxThread = 0;
167 (*intervals)[idxThread].iterator = workPerElement[0].iterator;
168 (*intervals)[idxThread].nbElements = 1;
169 FSize assignWork = workPerElement[0].amountOfWork;
170
171 for(int idxElement = 1 ; idxElement < nbElements ; ++idxElement){
174 // is it more balance if we add the current element to the current thread
175 if(FMath::Abs((idxThread+1)*idealWork - assignWork) <
176 FMath::Abs((idxThread+1)*idealWork - assignWork - workPerElement[idxElement].amountOfWork)
177 && idxThread != MaxThreads-1){
180 // if not start filling the next thread
181 idxThread += 1;
182 (*intervals)[idxThread].iterator = workPerElement[idxElement].iterator;
183 (*intervals)[idxThread].nbElements = 0;
184 }
185 (*intervals)[idxThread].nbElements += 1;
186 assignWork += workPerElement[idxElement].amountOfWork;
187 }

203 int shapeLeaves[SizeShape] = {0};
204 int leafsNumber = 0;

206 typename OctreeClass::Iterator octreeIterator(tree);
207 octreeIterator.gotoBottomLeft();
208 do{
209 ++leafsNumber;
210 const FTreeCoordinate& coord = octreeIterator.getCurrentCell()->getCoordinate();
211 ++shapeLeaves[P2PExclusionClass::GetShapeIdx(coord)];

219 #pragma omp parallel num_threads(MaxThreads)
220 {
221 #pragma omp single
222 {
223 #pragma omp task
224 { // Prepare P2M
225 if(workloadBufferThread[omp_get_thread_num()] == nullptr){
226 workloadBufferThread[omp_get_thread_num()] = new WorkloadTemp[leafsNumber];

228 WorkloadTemp* workloadBuffer = workloadBufferThread[omp_get_thread_num()];

234 FSize totalWork = 0;
235 do{
236 // Keep track of tree iterator
237 workloadBuffer[idxLeaf].iterator = octreeIterator;
238 // Count the nb of particles as amount of work in the leaf
239 workloadBuffer[idxLeaf].amountOfWork = octreeIterator.getCurrentListSrc()->getNbParticles();
240 // Keep the total amount of work
241 totalWork += workloadBuffer[idxLeaf].amountOfWork;
242 ++idxLeaf;
243 } while(octreeIterator.moveRight());
244
245 generateIntervalFromWorkload(&workloadP2M, totalWork, workloadBuffer, idxLeaf);
246 }
247
248 #pragma omp task
249 { // Prepare L2P
250 if(workloadBufferThread[omp_get_thread_num()] == nullptr){
251 workloadBufferThread[omp_get_thread_num()] = new WorkloadTemp[leafsNumber];
252 }
253 WorkloadTemp* workloadBuffer = workloadBufferThread[omp_get_thread_num()];
255 typename OctreeClass::Iterator octreeIterator(tree);
256 octreeIterator.gotoBottomLeft();
257 FSize idxLeaf = 0;
258 FSize totalWork = 0;
259 do{
260 // Keep track of tree iterator
261 workloadBuffer[idxLeaf].iterator = octreeIterator;
262 // Count the nb of particles as amount of work in the leaf
263 workloadBuffer[idxLeaf].amountOfWork = octreeIterator.getCurrentListTargets()->getNbParticles();
264 // Keep the total amount of work
265 totalWork += workloadBuffer[idxLeaf].amountOfWork;
266 ++idxLeaf;
267 } while(octreeIterator.moveRight());
268
269 generateIntervalFromWorkload(&workloadL2P, totalWork, workloadBuffer, idxLeaf);
270 }

272 #pragma omp task
273 {// Do it for the M2L
274 if(workloadBufferThread[omp_get_thread_num()] == nullptr){
275 workloadBufferThread[omp_get_thread_num()] = new WorkloadTemp[leafsNumber];
276 }
277 WorkloadTemp* workloadBuffer = workloadBufferThread[omp_get_thread_num()];

281 avoidGotoLeftIterator.gotoBottomLeft();
282

284
285 for(int idxLevel = OctreeHeight-1 ; idxLevel >= 2 ; --idxLevel){
287 typename OctreeClass::Iterator octreeIterator(avoidGotoLeftIterator);
288 avoidGotoLeftIterator.moveUp();
289
290 FSize idxCell = 0;
291 FSize totalWork = 0;
292 do{
293 // Keep track of tree iterator
294 workloadBuffer[idxCell].iterator = octreeIterator;
295 // Count the nb of M2L for this cell
296 workloadBuffer[idxCell].amountOfWork = tree->getInteractionNeighbors(neighbors, octreeIterator.getCurrentGlobalCoordinate(), idxLevel, 1);
297 // Keep the total amount of work
298 totalWork += workloadBuffer[idxCell].amountOfWork;
299 ++idxCell;
300 } while(octreeIterator.moveRight());
301
302 // Now split between thread
303 generateIntervalFromWorkload(&workloadM2L[idxLevel], totalWork, workloadBuffer, idxCell);
304 }
305 }
306 #pragma omp task
307 {// Do it for the M2M L2L
308 if(workloadBufferThread[omp_get_thread_num()] == nullptr){

316 avoidGotoLeftIterator.gotoBottomLeft();
317 avoidGotoLeftIterator.moveUp();
318

321 typename OctreeClass::Iterator octreeIterator(avoidGotoLeftIterator);
322 avoidGotoLeftIterator.moveUp();
323
324 FSize idxCell = 0;
325 FSize totalWork = 0;
326 do{
327 // Keep track of tree iterator

333 if(child[idxChild]) workloadBuffer[idxCell].amountOfWork += 1;
334 }

337 ++idxCell;
338 } while(octreeIterator.moveRight());
339
340 // Now split between thread
341 generateIntervalFromWorkload(&workloadM2M[idxLevel], totalWork, workloadBuffer, idxCell);
342 generateIntervalFromWorkload(&workloadL2L[idxLevel], totalWork, workloadBuffer, idxCell);
343 }

346 #pragma omp task
347 {
348 if(workloadBufferThread[omp_get_thread_num()] == nullptr){
349 workloadBufferThread[omp_get_thread_num()] = new WorkloadTemp[leafsNumber];
350 }
351 WorkloadTemp* workloadBuffer = workloadBufferThread[omp_get_thread_num()];
352 memset(workloadBuffer, 0, sizeof(struct WorkloadTemp)*leafsNumber);
353 // Prepare the P2P
354 leafsDataArray.reset(new LeafData[leafsNumber]);
355
356 // We need the offset for each color
357 int startPosAtShape[SizeShape] = {0};
358 for(int idxShape = 1 ; idxShape < SizeShape ; ++idxShape){
359 startPosAtShape[idxShape] = startPosAtShape[idxShape-1] + shapeLeaves[idxShape-1];
360 }
361
362 // Prepare each color
363 typename OctreeClass::Iterator octreeIterator(tree);
364 octreeIterator.gotoBottomLeft();
365
366 FSize workPerShape[SizeShape] = {0};
367
368 // for each leafs
369 for(int idxLeaf = 0 ; idxLeaf < leafsNumber ; ++idxLeaf){
370 const FTreeCoordinate& coord = octreeIterator.getCurrentGlobalCoordinate();
371 const int shapePosition = P2PExclusionClass::GetShapeIdx(coord);
372
373 const int positionToWork = startPosAtShape[shapePosition]++;
374
375 leafsDataArray[positionToWork].index = octreeIterator.getCurrentGlobalIndex();
376 leafsDataArray[positionToWork].coord = coord;
377 leafsDataArray[positionToWork].targets = octreeIterator.getCurrentListTargets();
378 leafsDataArray[positionToWork].sources = octreeIterator.getCurrentListSrc();
379
380 // For now the cost is simply based on the number of particles
381 const FSize nbPartInLeaf = octreeIterator.getCurrentListTargets()->getNbParticles();
382 workloadBuffer[positionToWork].amountOfWork = nbPartInLeaf*nbPartInLeaf;
383 ContainerClass* neighbors[27];
384 tree->getLeafsNeighbors(neighbors, octreeIterator.getCurrentGlobalCoordinate(), OctreeHeight-1);
385 for(int idxNeigh = 0 ; idxNeigh < 27 ; ++idxNeigh){
386 if(neighbors[idxNeigh]){
387 workloadBuffer[positionToWork].amountOfWork +=
388 nbPartInLeaf * neighbors[idxNeigh]->getNbParticles();
389 }
390 }
391
392 workPerShape[shapePosition] += workloadBuffer[positionToWork].amountOfWork;
393
394 octreeIterator.moveRight();
395 }
396
397 workloadP2P.resize(SizeShape);
398 int offsetShape = 0;
399
400 for(int idxShape = 0 ; idxShape < SizeShape ; ++idxShape){
401 std::vector<std::pair<int,int>>* intervals = &workloadP2P[idxShape];
402 const int nbElements = shapeLeaves[idxShape];
403 const FSize totalWork = workPerShape[idxShape];
404
405 // Now split between thread
406 (*intervals).resize(MaxThreads, std::pair<int,int>(0,0));
407 // Ideally each thread will have this
408 const FSize idealWork = (totalWork/MaxThreads);
409 // Assign default value for first thread
410 int idxThread = 0;
411 (*intervals)[idxThread].first = offsetShape;
412 FSize assignWork = workloadBuffer[offsetShape].amountOfWork;
413 for(int idxElement = 1+offsetShape ; idxElement < nbElements+offsetShape ; ++idxElement){
414 if(FMath::Abs((idxThread+1)*idealWork - assignWork) <
415 FMath::Abs((idxThread+1)*idealWork - assignWork - workloadBuffer[idxElement].amountOfWork)
416 && idxThread != MaxThreads-1){
417 (*intervals)[idxThread].second = idxElement;
418 idxThread += 1;
419 (*intervals)[idxThread].first = idxElement;
420 }
421 assignWork += workloadBuffer[idxElement].amountOfWork;
422 }
423 (*intervals)[idxThread].second = nbElements + offsetShape;
424
425 idxThread += 1;
426 while(idxThread != MaxThreads){
427 (*intervals)[idxThread].first = nbElements+offsetShape;
428 (*intervals)[idxThread].second = nbElements+offsetShape;
429 idxThread += 1;
430 }
431
432 offsetShape += nbElements;
433 }
434 }
435 }
436
437 #pragma omp taskwait
438 }
439
440 for(int idxThread = 0 ; idxThread < MaxThreads ; ++idxThread){
441 delete[] workloadBufferThread[idxThread];
442 }
