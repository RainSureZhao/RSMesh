272 octreeIterator.moveUp();

274 }
275 }

391 void upwardPass(){
392 FLOG( FLog::Controller.write("\tStart Upward Pass\n").write(FLog::Flush); );
393 FLOG(FTic counterTime);
394 FLOG(FTic computationCounter);
395 FLOG(FTic singleCounter);
396 FLOG(FTic parallelCounter);
397
398 // Start from leal level (height-1)
399 typename OctreeClass::Iterator octreeIterator(tree);
400 octreeIterator.gotoBottomLeft();
401 octreeIterator.moveUp();
402
403 for(int idxLevel = OctreeHeight - 2 ; idxLevel > FAbstractAlgorithm::lowerWorkingLevel-1 ; --idxLevel){
404 octreeIterator.moveUp();
405 }
406
407 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
408
409 // The proc to send the shared cells to
410 // Starting to the proc on the left this variable will go to 0
411 int currentProcIdToSendTo = (idProcess - 1);
412
413 // There are a maximum of 1 sends and 8-1 receptions
414 MPI_Request requests[8];
415 MPI_Status status[8];
416
417 MPI_Request requestsSize[8];
418 MPI_Status statusSize[8];
419
420 FSize bufferSize;
421 FMpiBufferWriter sendBuffer(1);// Max = 1 + sizeof(cell)*7
422 std::unique_ptr<FMpiBufferReader[]> recvBuffer(new FMpiBufferReader[7]);
423 FSize recvBufferSize[7];
424 CellClass recvBufferCells[7];
425
426 // The first proc that send to me a cell
427 // This variable will go to nbProcess
428 int firstProcThatSend = idProcess + 1;
429 FLOG(computationCounter.tic());
430
431 // for each levels
432 for(int idxLevel = FMath::Min(OctreeHeight - 2, FAbstractAlgorithm::lowerWorkingLevel - 1) ; idxLevel >= FAbstractAlgorithm::upperWorkingLevel ; --idxLevel ){
433 // Does my cells are covered by my neighbors working interval and so I have no more work?
434 const bool noMoreWorkForMe = (idProcess != 0 && !procHasWorkAtLevel(idxLevel+1, idProcess));
435 if(noMoreWorkForMe){
436 FAssertLF(procHasWorkAtLevel(idxLevel, idProcess) == false);
437 break;
438 }
439
440 // Copy and count ALL the cells (even the ones outside the working interval)
441 int totalNbCellsAtLevel = 0;
442 do{
443 iterArray[totalNbCellsAtLevel++] = octreeIterator;
444 } while(octreeIterator.moveRight());
445 avoidGotoLeftIterator.moveUp();
446 octreeIterator = avoidGotoLeftIterator;
447
448 int iterMpiRequests = 0; // The iterator for send/recv requests
449 int iterMpiRequestsSize = 0; // The iterator for send/recv requests
450
451 int nbCellsToSkip = 0; // The number of cells to send
452 // Skip all the cells that are out of my working interval
453 while(nbCellsToSkip < totalNbCellsAtLevel && iterArray[nbCellsToSkip].getCurrentGlobalIndex() < getWorkingInterval(idxLevel, idProcess).leftIndex){
454 ++nbCellsToSkip;
455 }
456
457 // We need to know if we will recv something in order to know if threads skip the last cell
458 int nbCellsForThreads = totalNbCellsAtLevel; // totalNbCellsAtLevel or totalNbCellsAtLevel-1
459 bool hasToReceive = false;
460 if(idProcess != nbProcess-1 && procHasWorkAtLevel(idxLevel , idProcess)){
461 // Find the first proc that may send to me
462 while(firstProcThatSend < nbProcess && !procHasWorkAtLevel(idxLevel+1, firstProcThatSend) ){
463 firstProcThatSend += 1;
464 }
465 // Do we have to receive?
466 if(firstProcThatSend < nbProcess && procHasWorkAtLevel(idxLevel+1, firstProcThatSend) && procCoversMyRightBorderCell(idxLevel, firstProcThatSend) ){
467 hasToReceive = true;
468 // Threads do not compute the last cell, we will do it once data are received
469 nbCellsForThreads -= 1;
470 }
471 }
472
473 FLOG(parallelCounter.tic());
474#pragma omp parallel num_threads(MaxThreads)
475 {
476 KernelClass* myThreadkernels = (kernels[omp_get_thread_num()]);
477 //This single section post and receive the comms, and then do the M2M associated with it.
478#pragma omp single nowait
479 {
480 FLOG(singleCounter.tic());
481 // Master proc never send
482 if(idProcess != 0){
483 // Skip process that have no work at that level
484 while( currentProcIdToSendTo && !procHasWorkAtLevel(idxLevel, currentProcIdToSendTo) ){
485 --currentProcIdToSendTo;
486 }
487 // Does the next proc that has work is sharing the parent of my left cell
488 if(procHasWorkAtLevel(idxLevel, currentProcIdToSendTo) && procCoversMyLeftBorderCell(idxLevel, currentProcIdToSendTo)){
489 FAssertLF(nbCellsToSkip != 0);
490
491 char packageFlags = 0;
492 sendBuffer.write(packageFlags);
493
494 // Only the cell the most on the right out of my working interval should be taken in
495 // consideration (at pos nbCellsToSkip-1) other (x < nbCellsToSkip-1) have already been sent
496 const CellClass* const* const child = iterArray[nbCellsToSkip-1].getCurrentChild();
497 for(int idxChild = 0 ; idxChild < 8 ; ++idxChild){
498 // Check if child exists and it was part of my working interval
499 if( child[idxChild] && getWorkingInterval((idxLevel+1), idProcess).leftIndex <= child[idxChild]->getMortonIndex() ){
500 // Add the cell to the buffer
501 child[idxChild]->serializeUp(sendBuffer);
502 packageFlags = char(packageFlags | (0x1 << idxChild));
503 }
504 }
505 // Add the flag as first value
506 sendBuffer.writeAt(0,packageFlags);
507 // Post the message
508 bufferSize = sendBuffer.getSize();
509 MPI_Isend(&bufferSize, 1, FMpi::GetType(bufferSize), currentProcIdToSendTo,
510 FMpi::TagFmmM2MSize + idxLevel, fcomCompute.getComm(), &requestsSize[iterMpiRequestsSize++]);
511 FAssertLF(sendBuffer.getSize() < std::numeric_limits<int>::max());
512 MPI_Isend(sendBuffer.data(), int(sendBuffer.getSize()), MPI_BYTE, currentProcIdToSendTo,
513 FMpi::TagFmmM2M + idxLevel, fcomCompute.getComm(), &requests[iterMpiRequests++]);
514 }
515 }
516
517 //Post receive, Datas needed in several parts of the section
518 int nbProcThatSendToMe = 0;
519
520 if(hasToReceive){
521 //Test : if the firstProcThatSend father minimal value in interval is lesser than mine
522 int idProcSource = firstProcThatSend;
523 // Find the last proc that should send to me
524 while( idProcSource < nbProcess
525 && ( !procHasWorkAtLevel(idxLevel+1, idProcSource) || procCoversMyRightBorderCell(idxLevel, idProcSource) )){
526 if(procHasWorkAtLevel(idxLevel+1, idProcSource) && procCoversMyRightBorderCell(idxLevel, idProcSource)){
527 MPI_Irecv(&recvBufferSize[nbProcThatSendToMe], 1, FMpi::GetType(recvBufferSize[nbProcThatSendToMe]),
528 idProcSource, FMpi::TagFmmM2MSize + idxLevel, fcomCompute.getComm(), &requestsSize[iterMpiRequestsSize++]);
529 nbProcThatSendToMe += 1;
530 FAssertLF(nbProcThatSendToMe <= 7);
531 }
532 ++idProcSource;
533 }
534 }
535
536 //Wait For the comms, and do the work
537 // Are we sending or waiting anything?
538 if(iterMpiRequestsSize){
539 FAssertLF(iterMpiRequestsSize <= 8);
540 MPI_Waitall( iterMpiRequestsSize, requestsSize, statusSize);
541 }
542
543 if(hasToReceive){
544 nbProcThatSendToMe = 0;
545 //Test : if the firstProcThatSend father minimal value in interval is lesser than mine
546 int idProcSource = firstProcThatSend;
547 // Find the last proc that should send to me
548 while( idProcSource < nbProcess
549 && ( !procHasWorkAtLevel(idxLevel+1, idProcSource) || procCoversMyRightBorderCell(idxLevel, idProcSource) )){
550 if(procHasWorkAtLevel(idxLevel+1, idProcSource) && procCoversMyRightBorderCell(idxLevel, idProcSource)){
551 recvBuffer[nbProcThatSendToMe].cleanAndResize(recvBufferSize[nbProcThatSendToMe]);
552 FAssertLF(recvBufferSize[nbProcThatSendToMe] < std::numeric_limits<int>::max());
553 MPI_Irecv(recvBuffer[nbProcThatSendToMe].data(), int(recvBufferSize[nbProcThatSendToMe]), MPI_BYTE,
554 idProcSource, FMpi::TagFmmM2M + idxLevel, fcomCompute.getComm(), &requests[iterMpiRequests++]);
555 nbProcThatSendToMe += 1;
556 FAssertLF(nbProcThatSendToMe <= 7);
557 }
558 ++idProcSource;
559 }
560 }
561
562 //Wait For the comms, and do the work
563 // Are we sending or waiting anything?
564 if(iterMpiRequests){
565 FAssertLF(iterMpiRequests <= 8);
566 MPI_Waitall( iterMpiRequests, requests, status);
567 }
568
569 // We had received something so we need to proceed the last M2M
570 if( hasToReceive ){
571 FAssertLF(iterMpiRequests != 0);
572 CellClass* currentChild[8];
573 memcpy(currentChild, iterArray[totalNbCellsAtLevel - 1].getCurrentChild(), 8 * sizeof(CellClass*));
574
575 // Retreive data and merge my child and the child from others
576 int positionToInsert = 0;
577 for(int idxProc = 0 ; idxProc < nbProcThatSendToMe ; ++idxProc){
578 unsigned packageFlags = unsigned(recvBuffer[idxProc].getValue<unsigned char>());
579
580 int position = 0;
581 while( packageFlags && position < 8){
582 while(!(packageFlags & 0x1)){
583 packageFlags >>= 1;
584 ++position;
585 }
586 FAssertLF(positionToInsert < 7);
587 FAssertLF(position < 8);
588 FAssertLF(!currentChild[position], "Already has a cell here");
589 recvBufferCells[positionToInsert].deserializeUp(recvBuffer[idxProc]);
590 currentChild[position] = (CellClass*) &recvBufferCells[positionToInsert];
591
592 packageFlags >>= 1;
593 position += 1;
594 positionToInsert += 1;
595 }
596
597 recvBuffer[idxProc].seek(0);
598 }
599 // Finally compute
600 myThreadkernels->M2M( iterArray[totalNbCellsAtLevel - 1].getCurrentCell() , currentChild, idxLevel);
601 firstProcThatSend += nbProcThatSendToMe - 1;
602 }
603 // Reset buffer
604 sendBuffer.reset();
605 FLOG(singleCounter.tac());
606 }//End Of Single section
607
608 // All threads proceed the M2M
609#pragma omp for nowait schedule(dynamic, userChunkSize)
610 for( int idxCell = nbCellsToSkip ; idxCell < nbCellsForThreads ; ++idxCell){
611 myThreadkernels->M2M( iterArray[idxCell].getCurrentCell() , iterArray[idxCell].getCurrentChild(), idxLevel);
612 }
613 }//End of parallel section
614 FLOG(parallelCounter.tac());
615 }
616
617 FLOG(counterTime.tac());
618 FLOG(computationCounter.tac());
619 FLOG( FLog::Controller << "\tFinished (@Upward Pass (M2M) = " << counterTime.elapsed() << " s)\n" );
620 FLOG( FLog::Controller << "\t\t Computation : " << computationCounter.elapsed() << " s\n" );
621 FLOG( FLog::Controller << "\t\t Single : " << singleCounter.cumulated() << " s\n" );
622 FLOG( FLog::Controller << "\t\t Parallel : " << parallelCounter.cumulated() << " s\n" );
623 FLOG( FLog::Controller.flush());
624 }

1044 for(int idxLevel = 2 ; idxLevel < FAbstractAlgorithm::upperWorkingLevel ; ++idxLevel){
1045 octreeIterator.moveDown();
1046 }
1047
1048 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
1049
1050 // Max 1 receive and 7 send (but 7 times the same data)
1051 MPI_Request*const requests = new MPI_Request[8];
1052 MPI_Request*const requestsSize = new MPI_Request[8];
1053
1054 const int heightMinusOne = FAbstractAlgorithm::lowerWorkingLevel - 1;
1055
1056 FMpiBufferWriter sendBuffer;
1057 FMpiBufferReader recvBuffer;
1058
1059 int righestProcToSendTo = nbProcess - 1;
1060
1061 // for each levels exepted leaf level
1062 for(int idxLevel = FAbstractAlgorithm::upperWorkingLevel ; idxLevel < heightMinusOne ; ++idxLevel ){
1063 // If nothing to do in the next level skip the current one
1064 if(idProcess != 0 && !procHasWorkAtLevel(idxLevel+1, idProcess) ){
1065 avoidGotoLeftIterator.moveDown();
1066 octreeIterator = avoidGotoLeftIterator;
1067 continue;
1068 }
1069
1070 // Copy all the cells in an array even the one that are out of my working interval
1071 int totalNbCellsAtLevel = 0;
1072 do{
1073 iterArray[totalNbCellsAtLevel++] = octreeIterator;
1074 } while(octreeIterator.moveRight());
1075 avoidGotoLeftIterator.moveDown();
1076 octreeIterator = avoidGotoLeftIterator;
1077
1078 // Count the number of cells that are out of my working interval
1079 int nbCellsToSkip = 0;
1080 while(nbCellsToSkip < totalNbCellsAtLevel && iterArray[nbCellsToSkip].getCurrentGlobalIndex() < getWorkingInterval(idxLevel , idProcess).leftIndex){
1081 nbCellsToSkip += 1;
1082 }
1083
1084 // Check if someone will send a cell to me
1085 bool hasToReceive = false;
1086 int idxProcToReceive = idProcess - 1;
1087 if(idProcess != 0 && nbCellsToSkip){
1088 // Starting from my left neighbor stop at the first proc that has work to do (not null interval)
1089 while(idxProcToReceive && !procHasWorkAtLevel(idxLevel, idxProcToReceive) ){
1090 idxProcToReceive -= 1;
1091 }
1092 // Check if we find such a proc and that it share a cell with us on the border
1093 if(procHasWorkAtLevel(idxLevel, idxProcToReceive) && procCoversMyLeftBorderCell(idxLevel, idxProcToReceive)){
1094 hasToReceive = true;
1095 }

1098#pragma omp parallel num_threads(MaxThreads)
1099 {
1100 KernelClass* myThreadkernels = (kernels[omp_get_thread_num()]);
1101#pragma omp single nowait
1102 {
1103 FLOG(prepareCounter.tic());

1106 FSize recvBufferSize = 0;
1107 FSize sendBufferSize;

1111 FMpi::TagFmmL2LSize + idxLevel, fcomCompute.getComm(), &requestsSize[iterRequestsSize++]), __LINE__ );
1112 }
1113
1114 // We have to be sure that we are not sending if we have no work in the current level
1115 if(idProcess != nbProcess - 1 && idProcess < righestProcToSendTo && procHasWorkAtLevel(idxLevel, idProcess)){
1116 int idxProcSend = idProcess + 1;
1117 int nbMessageSent = 0;
1118 // From the proc on the right to righestProcToSendTo, check if we have to send something
1119 while(idxProcSend <= righestProcToSendTo && ( !procHasWorkAtLevel(idxLevel+1, idxProcSend) || procCoversMyRightBorderCell(idxLevel, idxProcSend)) ){
1120 // We know that if the proc has work at the next level it share a cell with us due to the while condition
1121 if(procHasWorkAtLevel(idxLevel+1, idxProcSend)){
1122 FAssertLF(procCoversMyRightBorderCell(idxLevel, idxProcSend));
1123 // If first message then serialize the cell to send
1124 if( nbMessageSent == 0 ){
1125 // We send our last cell
1126 iterArray[totalNbCellsAtLevel - 1].getCurrentCell()->serializeDown(sendBuffer);

1128 }

1131 FMpi::TagFmmL2LSize + idxLevel, fcomCompute.getComm(), &requestsSize[iterRequestsSize++]), __LINE__);
1132 FAssertLF(sendBuffer.getSize() < std::numeric_limits<int>::max());

1137 FAssertLF(nbMessageSent <= 7);
1138 }
1139 idxProcSend += 1;
1140 }

1142 if(idxProcSend < righestProcToSendTo){
1143 righestProcToSendTo = idxProcSend;
1144 }
1145 }

1148 if(iterRequestsSize){
1149 FLOG(waitCounter.tic());

1151 FMpi::MpiAssert(MPI_Waitall( iterRequestsSize, requestsSize, MPI_STATUSES_IGNORE), __LINE__);
1152 FLOG(waitCounter.tac());
1153 }

1157 FAssertLF(recvBuffer.getCapacity() < std::numeric_limits<int>::max());
1158 FMpi::MpiAssert( MPI_Irecv( recvBuffer.data(), int(recvBuffer.getCapacity()), MPI_BYTE, idxProcToReceive,
1159 FMpi::TagFmmL2L + idxLevel, fcomCompute.getComm(), &requests[iterRequests++]), __LINE__ );
1160 }
1161
1162 if(iterRequests){
1163 FLOG(waitCounter.tic());
1164 FAssertLF(iterRequests <= 8);
1165 FMpi::MpiAssert(MPI_Waitall( iterRequests, requests, MPI_STATUSES_IGNORE), __LINE__);
1166 FLOG(waitCounter.tac());
1167 }
1168
1169 // If we receive something proceed the L2L
1170 if(hasToReceive){
1171 FAssertLF(iterRequests != 0);
1172 // In this case we know that we have to perform the L2L with the last cell that are
1173 // exclude from our working interval nbCellsToSkip-1
1174 iterArray[nbCellsToSkip-1].getCurrentCell()->deserializeDown(recvBuffer);
1175 myThreadkernels->L2L( iterArray[nbCellsToSkip-1].getCurrentCell() , iterArray[nbCellsToSkip-1].getCurrentChild(), idxLevel);
1176 }
1177 FLOG(prepareCounter.tac());
1178 }
1179
1180#pragma omp single nowait
1181 {
1182 FLOG(computationCounter.tic());
1183 }

1185#pragma omp for nowait schedule(dynamic, userChunkSize)
1186 for(int idxCell = nbCellsToSkip ; idxCell < totalNbCellsAtLevel ; ++idxCell){
1187 myThreadkernels->L2L( iterArray[idxCell].getCurrentCell() , iterArray[idxCell].getCurrentChild(), idxLevel);
1188 }
1189 }
1190 FLOG(computationCounter.tac());
1191
1192 sendBuffer.reset();
1193 recvBuffer.seek(0);
1194 }
1195

1197 delete[] requestsSize;

1200 FLOG( FLog::Controller << "\t\t Computation : " << computationCounter.cumulated() << " s\n" );
1201 FLOG( FLog::Controller << "\t\t Prepare : " << prepareCounter.cumulated() << " s\n" );
1202 FLOG( FLog::Controller << "\t\t Wait : " << waitCounter.cumulated() << " s\n" );
1203 FLOG( FLog::Controller.flush());
