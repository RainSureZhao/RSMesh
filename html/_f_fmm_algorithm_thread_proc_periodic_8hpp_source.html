304 void executeCore(const unsigned operationsToProceed) override {
305 // Count leaf
306 this->numberOfLeafs = 0;
307 {
308 Interval myFullInterval;
309 {//Building the interval with the first and last leaves (and count the number of leaves)
310 typename OctreeClass::Iterator octreeIterator(tree);
311 octreeIterator.gotoBottomLeft();
312 myFullInterval.leftIndex = octreeIterator.getCurrentGlobalIndex();
313 do{
314 ++this->numberOfLeafs;
315 } while(octreeIterator.moveRight());
316 myFullInterval.rightIndex = octreeIterator.getCurrentGlobalIndex();
317 }
318 // Allocate a number to store the pointer of the cells at a level
319 iterArray = new typename OctreeClass::Iterator[numberOfLeafs];
320 iterArrayComm = new typename OctreeClass::Iterator[numberOfLeafs];
321 FAssertLF(iterArray, "iterArray bad alloc");
322 FAssertLF(iterArrayComm, "iterArrayComm bad alloc");
323
324 // We get the leftIndex/rightIndex indexes from each procs
325 FMpi::MpiAssert( MPI_Allgather( &myFullInterval, sizeof(Interval), MPI_BYTE, intervals, sizeof(Interval), MPI_BYTE, comm.getComm()), __LINE__ );
326
327 // Build my intervals for all levels
328 std::unique_ptr<Interval[]> myIntervals(new Interval[OctreeHeight]);
329 // At leaf level we know it is the full interval
330 myIntervals[OctreeHeight - 1] = myFullInterval;
331
332 // We can estimate the interval for each level by using the parent/child relation
333 for(int idxLevel = OctreeHeight - 2 ; idxLevel >= 0 ; --idxLevel){
334 myIntervals[idxLevel].leftIndex = myIntervals[idxLevel+1].leftIndex >> 3;
335 myIntervals[idxLevel].rightIndex = myIntervals[idxLevel+1].rightIndex >> 3;
336 }
337
338 // Process 0 uses the estimates as real intervals, but other processes
339 // should remove cells that belong to others
340 if(idProcess != 0){
341 //We test for each level if process on left (idProcess-1) own cell I thought I owned
342 typename OctreeClass::Iterator octreeIterator(tree);
343 octreeIterator.gotoBottomLeft();
344 octreeIterator.moveUp();
345
346 // At h-1 the working limit is the parent of the right cell of the proc on the left
347 MortonIndex workingLimitAtLevel = intervals[idProcess-1].rightIndex >> 3;
348
349 // We check if we have no more work to do
350 int nullIntervalFromLevel = 0;
351
352 for(int idxLevel = OctreeHeight - 2 ; idxLevel >= 1 && nullIntervalFromLevel == 0 ; --idxLevel){
353 while(octreeIterator.getCurrentGlobalIndex() <= workingLimitAtLevel){
354 if( !octreeIterator.moveRight() ){
355 // We cannot move right we are not owner of any more cell
356 nullIntervalFromLevel = idxLevel;
357 break;
358 }
359 }
360 // If we are responsible for some cells at this level keep the first index
361 if(nullIntervalFromLevel == 0){
362 myIntervals[idxLevel].leftIndex = octreeIterator.getCurrentGlobalIndex();
363 octreeIterator.moveUp();
364 workingLimitAtLevel >>= 3;
365 }
366 }
367 // In case we are not responsible for any cells we put the leftIndex = rightIndex+1
368 for(int idxLevel = nullIntervalFromLevel ; idxLevel >= 1 ; --idxLevel){
369 myIntervals[idxLevel].leftIndex = myIntervals[idxLevel].rightIndex + 1;
370 }
371 }
372
373 // We get the leftIndex/rightIndex indexes from each procs
374 FMpi::MpiAssert( MPI_Allgather( myIntervals.get(), int(sizeof(Interval)) * OctreeHeight, MPI_BYTE,
375 workingIntervalsPerLevel, int(sizeof(Interval)) * OctreeHeight, MPI_BYTE, comm.getComm()), __LINE__ );
376 }
377
378 // run;
379 Timers[P2MTimer].tic();
380 if(operationsToProceed & FFmmP2M) bottomPass();
381 Timers[P2MTimer].tac();
382
383
384 Timers[M2MTimer].tic();
385 if(operationsToProceed & FFmmM2M) upwardPass();
386 Timers[M2MTimer].tac();
387
388 Timers[M2LTimer].tic();
389 if(operationsToProceed & FFmmM2L) transferPass();
390 Timers[M2LTimer].tac();
391
392 Timers[L2LTimer].tic();
393 if(operationsToProceed & FFmmL2L) downardPass();
394 Timers[L2LTimer].tac();
395
396 Timers[NearTimer].tic();
397 if((operationsToProceed & FFmmP2P) || (operationsToProceed & FFmmL2P)) directPass((operationsToProceed & FFmmP2P),(operationsToProceed & FFmmL2P));
398 Timers[NearTimer].tac();
399
400 // delete array
401 delete [] iterArray;
402 delete [] iterArrayComm;
403 iterArray = nullptr;
404 iterArrayComm = nullptr;
405 }

417 FLOG( FLog::Controller.write("\tStart Bottom Pass\n").write(FLog::Flush) );
418 FLOG(FTic counterTime);

421

426 iterArray[leafs++] = octreeIterator;

451 void upwardPass(){
452 FLOG( FLog::Controller.write("\tStart Upward Pass\n").write(FLog::Flush); );
453 FLOG(FTic counterTime);
454 FLOG(FTic computationCounter);
455 FLOG(FTic singleCounter);
456 FLOG(FTic parallelCounter);
457
458 // Start from leal level (height-1)
459 typename OctreeClass::Iterator octreeIterator(tree);
460 octreeIterator.gotoBottomLeft();
461 octreeIterator.moveUp();
462 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
463
464 // The proc to send the shared cells to
465 // Starting to the proc on the left this variable will go to 0
466 int currentProcIdToSendTo = (idProcess - 1);
467
468 // There are a maximum of 1 sends and 8-1 receptions
469 MPI_Request requests[8];
470 MPI_Status status[8];
471
472 MPI_Request requestsSize[8];
473 MPI_Status statusSize[8];
474
475 FSize bufferSize;
476 FMpiBufferWriter sendBuffer(1);// Max = 1 + sizeof(cell)*7
477 std::unique_ptr<FMpiBufferReader[]> recvBuffer(new FMpiBufferReader[7]);
478 FSize recvBufferSize[7];
479 CellClass recvBufferCells[7];
480
481 // The first proc that send to me a cell
482 // This variable will go to nbProcess
483 int firstProcThatSend = idProcess + 1;
484 FLOG(computationCounter.tic());
485
486 // We work from height-1 to 1
487 for(int idxLevel = OctreeHeight - 2 ; idxLevel > 0 ; --idxLevel ){
488 const int fackLevel = idxLevel + offsetRealTree;
489 // Does my cells are covered by my neighbors working interval and so I have no more work?
490 const bool noMoreWorkForMe = (idProcess != 0 && !procHasWorkAtLevel(idxLevel+1, idProcess));
491 if(noMoreWorkForMe){
492 FAssertLF(procHasWorkAtLevel(idxLevel, idProcess) == false);
493 break;
494 }
495
496 // Copy and count ALL the cells (even the ones outside the working interval)
497 int totalNbCellsAtLevel = 0;
498 do{
499 iterArray[totalNbCellsAtLevel++] = octreeIterator;
500 } while(octreeIterator.moveRight());
501 avoidGotoLeftIterator.moveUp();
502 octreeIterator = avoidGotoLeftIterator;
503
504 int iterMpiRequests = 0; // The iterator for send/recv requests
505 int iterMpiRequestsSize = 0; // The iterator for send/recv requests
506
507 int nbCellsToSkip = 0; // The number of cells to send
508 // Skip all the cells that are out of my working interval
509 while(nbCellsToSkip < totalNbCellsAtLevel && iterArray[nbCellsToSkip].getCurrentGlobalIndex() < getWorkingInterval(idxLevel, idProcess).leftIndex){
510 ++nbCellsToSkip;
511 }
512
513 // We need to know if we will recv something in order to know if threads skip the last cell
514 int nbCellsForThreads = totalNbCellsAtLevel; // totalNbCellsAtLevel or totalNbCellsAtLevel-1
515 bool hasToReceive = false;
516 if(idProcess != nbProcess-1 && procHasWorkAtLevel(idxLevel , idProcess)){
517 // Find the first proc that may send to me
518 while(firstProcThatSend < nbProcess && !procHasWorkAtLevel(idxLevel+1, firstProcThatSend) ){
519 firstProcThatSend += 1;
520 }
521 // Do we have to receive?
522 if(firstProcThatSend < nbProcess && procHasWorkAtLevel(idxLevel+1, firstProcThatSend) && procCoversMyRightBorderCell(idxLevel, firstProcThatSend) ){
523 hasToReceive = true;
524 // Threads do not compute the last cell, we will do it once data are received
525 nbCellsForThreads -= 1;
526 }
527 }
528
529 FLOG(parallelCounter.tic());
530 #pragma omp parallel num_threads(MaxThreads)
531 {
532 KernelClass* myThreadkernels = (kernels[omp_get_thread_num()]);
533 //This single section post and receive the comms, and then do the M2M associated with it.
534 #pragma omp single nowait
535 {
536 FLOG(singleCounter.tic());
537 // Master proc never send
538 if(idProcess != 0){
539 // Skip process that have no work at that level
540 while( currentProcIdToSendTo && !procHasWorkAtLevel(idxLevel, currentProcIdToSendTo) ){
541 --currentProcIdToSendTo;
542 }
543 // Does the next proc that has work is sharing the parent of my left cell
544 if(procHasWorkAtLevel(idxLevel, currentProcIdToSendTo) && procCoversMyLeftBorderCell(idxLevel, currentProcIdToSendTo)){
545 FAssertLF(nbCellsToSkip != 0);
546
547 char packageFlags = 0;
548 sendBuffer.write(packageFlags);
549
550 // Only the cell the most on the right out of my working interval should be taken in
551 // consideration (at pos nbCellsToSkip-1) other (x < nbCellsToSkip-1) have already been sent
552 const CellClass* const* const child = iterArray[nbCellsToSkip-1].getCurrentChild();
553 for(int idxChild = 0 ; idxChild < 8 ; ++idxChild){
554 // Check if child exists and it was part of my working interval
555 if( child[idxChild] && getWorkingInterval((idxLevel+1), idProcess).leftIndex <= child[idxChild]->getMortonIndex() ){
556 // Add the cell to the buffer
557 child[idxChild]->serializeUp(sendBuffer);
558 packageFlags = char(packageFlags | (0x1 << idxChild));
559 }
560 }
561 // Add the flag as first value
562 sendBuffer.writeAt(0,packageFlags);
563 // Post the message
564 bufferSize = sendBuffer.getSize();
565 MPI_Isend(&bufferSize, 1, FMpi::GetType(bufferSize), currentProcIdToSendTo,
566 FMpi::TagFmmM2MSize + idxLevel, comm.getComm(), &requestsSize[iterMpiRequestsSize++]);
567 FAssertLF(sendBuffer.getSize() < std::numeric_limits<int>::max());
568 MPI_Isend(sendBuffer.data(), int(sendBuffer.getSize()), MPI_BYTE, currentProcIdToSendTo,
569 FMpi::TagFmmM2M + idxLevel, comm.getComm(), &requests[iterMpiRequests++]);
570 }
571 }
572
573 //Post receive, Datas needed in several parts of the section
574 int nbProcThatSendToMe = 0;
575
576 if(hasToReceive){
577 //Test : if the firstProcThatSend father minimal value in interval is lesser than mine
578 int idProcSource = firstProcThatSend;
579 // Find the last proc that should send to me
580 while( idProcSource < nbProcess
581 && ( !procHasWorkAtLevel(idxLevel+1, idProcSource) || procCoversMyRightBorderCell(idxLevel, idProcSource) )){
582 if(procHasWorkAtLevel(idxLevel+1, idProcSource) && procCoversMyRightBorderCell(idxLevel, idProcSource)){
583 MPI_Irecv(&recvBufferSize[nbProcThatSendToMe], 1, FMpi::GetType(recvBufferSize[nbProcThatSendToMe]),
584 idProcSource, FMpi::TagFmmM2MSize + idxLevel, comm.getComm(), &requestsSize[iterMpiRequestsSize++]);
585 nbProcThatSendToMe += 1;
586 FAssertLF(nbProcThatSendToMe <= 7);
587 }
588 ++idProcSource;
589 }
590 }
591
592 //Wait For the comms, and do the work
593 // Are we sending or waiting anything?
594 if(iterMpiRequestsSize){
595 FAssertLF(iterMpiRequestsSize <= 8);
596 MPI_Waitall( iterMpiRequestsSize, requestsSize, statusSize);
597 }
598
599 if(hasToReceive){
600 nbProcThatSendToMe = 0;
601 //Test : if the firstProcThatSend father minimal value in interval is lesser than mine
602 int idProcSource = firstProcThatSend;
603 // Find the last proc that should send to me
604 while( idProcSource < nbProcess
605 && ( !procHasWorkAtLevel(idxLevel+1, idProcSource) || procCoversMyRightBorderCell(idxLevel, idProcSource) )){
606 if(procHasWorkAtLevel(idxLevel+1, idProcSource) && procCoversMyRightBorderCell(idxLevel, idProcSource)){
607 recvBuffer[nbProcThatSendToMe].cleanAndResize(recvBufferSize[nbProcThatSendToMe]);
608 FAssertLF(recvBufferSize[nbProcThatSendToMe] < std::numeric_limits<int>::max());
609 MPI_Irecv(recvBuffer[nbProcThatSendToMe].data(), int(recvBufferSize[nbProcThatSendToMe]), MPI_BYTE,
610 idProcSource, FMpi::TagFmmM2M + idxLevel, comm.getComm(), &requests[iterMpiRequests++]);
611 nbProcThatSendToMe += 1;
612 FAssertLF(nbProcThatSendToMe <= 7);
613 }
614 ++idProcSource;
615 }
616 }
617
618 //Wait For the comms, and do the work
619 // Are we sending or waiting anything?
620 if(iterMpiRequests){
621 FAssertLF(iterMpiRequests <= 8);
622 MPI_Waitall( iterMpiRequests, requests, status);
623 }
624
625 // We had received something so we need to proceed the last M2M
626 if( hasToReceive ){
627 FAssertLF(iterMpiRequests != 0);
628 CellClass* currentChild[8];
629 memcpy(currentChild, iterArray[totalNbCellsAtLevel - 1].getCurrentChild(), 8 * sizeof(CellClass*));
630
631 // Retreive data and merge my child and the child from others
632 for(int idxProc = 0 ; idxProc < nbProcThatSendToMe ; ++idxProc){
633 unsigned packageFlags = unsigned(recvBuffer[idxProc].getValue<unsigned char>());
634
635 int position = 0;
636 int positionToInsert = 0;
637 while( packageFlags && position < 8){
638 while(!(packageFlags & 0x1)){
639 packageFlags >>= 1;
640 ++position;
641 }
642 FAssertLF(positionToInsert < 7);
643 FAssertLF(position < 8);
644 FAssertLF(!currentChild[position], "Already has a cell here");
645 recvBufferCells[positionToInsert].deserializeUp(recvBuffer[idxProc]);
646 currentChild[position] = (CellClass*) &recvBufferCells[positionToInsert];
647
648 packageFlags >>= 1;
649 position += 1;
650 positionToInsert += 1;
651 }
652
653 recvBuffer[idxProc].seek(0);
654 }
655 // Finally compute
656 myThreadkernels->M2M( iterArray[totalNbCellsAtLevel - 1].getCurrentCell() , currentChild, fackLevel);
657 firstProcThatSend += nbProcThatSendToMe - 1;
658 }
659 // Reset buffer
660 sendBuffer.reset();
661 FLOG(singleCounter.tac());
662 }//End Of Single section
663
664 // All threads proceed the M2M
665 #pragma omp for nowait schedule(dynamic, userChunkSize)
666 for( int idxCell = nbCellsToSkip ; idxCell < nbCellsForThreads ; ++idxCell){
667 myThreadkernels->M2M( iterArray[idxCell].getCurrentCell() , iterArray[idxCell].getCurrentChild(), fackLevel);
668 }
669 }//End of parallel section
670 FLOG(parallelCounter.tac());
671 }

673 FLOG(counterTime.tac());
674 FLOG(computationCounter.tac());

1197 FMpiBufferReader recvBuffer;

1202 if( idProcess == 0){

1243 nbCellsToSkip += 1;
1244 }

1250 // Starting from my left neighbor stop at the first proc that has work to do (not null interval)
1251 while(idxProcToReceive && !procHasWorkAtLevel(idxLevel, idxProcToReceive) ){
1252 idxProcToReceive -= 1;
1253 }

1256 hasToReceive = true;

1261 {
1262 KernelClass* myThreadkernels = (kernels[omp_get_thread_num()]);
1263 #pragma omp single nowait
1264 {
1265 FLOG(prepareCounter.tic());
1266 int iterRequests = 0;

1268 FSize recvBufferSize = 0;
1269 FSize sendBufferSize;

1271 if(hasToReceive){

1277 if(idProcess != nbProcess - 1 && idProcess < righestProcToSendTo && procHasWorkAtLevel(idxLevel, idProcess)){
1278 int idxProcSend = idProcess + 1;
1279 int nbMessageSent = 0;
1280 // From the proc on the right to righestProcToSendTo, check if we have to send something
1281 while(idxProcSend <= righestProcToSendTo && ( !procHasWorkAtLevel(idxLevel+1, idxProcSend) || procCoversMyRightBorderCell(idxLevel, idxProcSend)) ){
1282 // We know that if the proc has work at the next level it share a cell with us due to the while condition
1283 if(procHasWorkAtLevel(idxLevel+1, idxProcSend)){
1284 FAssertLF(procCoversMyRightBorderCell(idxLevel, idxProcSend));
1285 // If first message then serialize the cell to send
1286 if( nbMessageSent == 0 ){
1287 // We send our last cell
1288 iterArray[totalNbCellsAtLevel - 1].getCurrentCell()->serializeDown(sendBuffer);
1289 sendBufferSize = sendBuffer.getSize();
1290 }
1291 // Post the send message
1292 FMpi::MpiAssert( MPI_Isend(&sendBufferSize, 1, FMpi::GetType(sendBufferSize), idxProcSend,
1293 FMpi::TagFmmL2LSize + idxLevel, comm.getComm(), &requestsSize[iterRequestsSize++]), __LINE__);
1294 FAssertLF(sendBuffer.getSize() < std::numeric_limits<int>::max());
1295 FMpi::MpiAssert( MPI_Isend(sendBuffer.data(), int(sendBuffer.getSize()), MPI_BYTE, idxProcSend,
1296 FMpi::TagFmmL2L + idxLevel, comm.getComm(), &requests[iterRequests++]), __LINE__);
1297 // Inc and check the counter
1298 nbMessageSent += 1;
1299 FAssertLF(nbMessageSent <= 7);
1300 }
1301 idxProcSend += 1;
1302 }
1303 // Next time we will not need to go further than idxProcSend
1304 if(idxProcSend < righestProcToSendTo){
1305 righestProcToSendTo = idxProcSend;
1306 }
1307 }
1308
1309 // Finalize the communication
1310 if(iterRequestsSize){
1311 FLOG(waitCounter.tic());
1312 FAssertLF(iterRequestsSize < 8);
1313 FMpi::MpiAssert(MPI_Waitall( iterRequestsSize, requestsSize, MPI_STATUSES_IGNORE), __LINE__);
1314 FLOG(waitCounter.tac());
1315 }
1316
1317 if(hasToReceive){
1318 recvBuffer.cleanAndResize(recvBufferSize);
1319 FAssertLF(recvBuffer.getCapacity() < std::numeric_limits<int>::max());
1320 FMpi::MpiAssert( MPI_Irecv( recvBuffer.data(), int(recvBuffer.getCapacity()), MPI_BYTE, idxProcToReceive,
1321 FMpi::TagFmmL2L + idxLevel, comm.getComm(), &requests[iterRequests++]), __LINE__ );
1322 }
1323
1324 if(iterRequests){
1325 FLOG(waitCounter.tic());
1326 FAssertLF(iterRequests < 8);
1327 FMpi::MpiAssert(MPI_Waitall( iterRequests, requests, MPI_STATUSES_IGNORE), __LINE__);
1328 FLOG(waitCounter.tac());
1329 }
1330
1331 // If we receive something proceed the L2L
1332 if(hasToReceive){
1333 FAssertLF(iterRequests != 0);
1334 // In this case we know that we have to perform the L2L with the last cell that are
1335 // exclude from our working interval nbCellsToSkip-1
1336 iterArray[nbCellsToSkip-1].getCurrentCell()->deserializeDown(recvBuffer);
1337 myThreadkernels->L2L( iterArray[nbCellsToSkip-1].getCurrentCell() , iterArray[nbCellsToSkip-1].getCurrentChild(), fackLevel);
1338 }
1339 FLOG(prepareCounter.tac());
1340 }
1341
1342 #pragma omp single nowait
1343 {
1344 FLOG(computationCounter.tic());
1345 }
1346 // Threads are working on all the cell of our working interval at that level
1347 #pragma omp for nowait schedule(dynamic, userChunkSize)
1348 for(int idxCell = nbCellsToSkip ; idxCell < totalNbCellsAtLevel ; ++idxCell){
1349 myThreadkernels->L2L( iterArray[idxCell].getCurrentCell() , iterArray[idxCell].getCurrentChild(), fackLevel);
1350 }
1351 }
1352 FLOG(computationCounter.tac());
1353
1354 sendBuffer.reset();
1355 recvBuffer.seek(0);
1356 }
1357
1358 delete[] requests;
1359 delete[] requestsSize;

1361 FLOG( FLog::Controller << "\tFinished (@Downward Pass (L2L) = " << counterTime.tacAndElapsed() << " s)\n" );
1362 FLOG( FLog::Controller << "\t\t Computation : " << computationCounter.cumulated() << " s\n" );
1363 FLOG( FLog::Controller << "\t\t Prepare : " << prepareCounter.cumulated() << " s\n" );
1364 FLOG( FLog::Controller << "\t\t Wait : " << waitCounter.cumulated() << " s\n" );
1365 FLOG( FLog::Controller.flush());
