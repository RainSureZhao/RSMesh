81 void bottomPass(){
82 FLOG( FLog::Controller.write("\tStart Bottom Pass\n").write(FLog::Flush) );
83 FLOG( counterTime.tic() );
84 FLOG( double totalComputation = 0 );
85
86 typename OctreeClass::Iterator octreeIterator(tree);
87
88 // Iterate on leafs
89 octreeIterator.gotoBottomLeft();
90 do{
91 // We need the current cell that represent the leaf
92 // and the list of particles
93 FLOG(computationCounter.tic());
94 ContainerClass* const sources = octreeIterator.getCurrentListSrc();
95 if(sources->getNbParticles()){
96 octreeIterator.getCurrentCell()->setSrcChildTrue();
97 kernels->P2M( octreeIterator.getCurrentCell() , sources);
98 }
99 if(octreeIterator.getCurrentListTargets()->getNbParticles()){
100 octreeIterator.getCurrentCell()->setTargetsChildTrue();
101 }
102 FLOG(computationCounter.tac());
103 FLOG(totalComputation += computationCounter.elapsed());
104 } while(octreeIterator.moveRight());
105
106 FLOG( counterTime.tac() );
107 FLOG( FLog::Controller << "\tFinished (@Bottom Pass (P2M) = " << counterTime.elapsed() << " s)\n" );
108 FLOG( FLog::Controller << "\t\t Computation : " << totalComputation << " s\n" );

113 void upwardPass(){
114 FLOG( FLog::Controller.write("\tStart Upward Pass\n").write(FLog::Flush); );
115 FLOG( counterTime.tic() );
116 FLOG( double totalComputation = 0 );
117
118 // Start from leal level - 1
119 typename OctreeClass::Iterator octreeIterator(tree);
120 octreeIterator.gotoBottomLeft();
121 octreeIterator.moveUp();
122
123 for(int idxLevel = OctreeHeight - 2 ; idxLevel > FAbstractAlgorithm::lowerWorkingLevel-1 ; --idxLevel){
124 octreeIterator.moveUp();
125 }
126
127 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
128
129 // for each levels
130 for(int idxLevel = FMath::Min(OctreeHeight - 2, FAbstractAlgorithm::lowerWorkingLevel - 1) ; idxLevel >= FAbstractAlgorithm::upperWorkingLevel ; --idxLevel ){
131 FLOG(FTic counterTimeLevel);
132 // for each cells
133 do{
134 // We need the current cell and the child
135 // child is an array (of 8 child) that may be null
136 FLOG(computationCounter.tic());
137
138 CellClass* potentialChild[8];
139 CellClass** const realChild = octreeIterator.getCurrentChild();
140 CellClass* const currentCell = octreeIterator.getCurrentCell();
141 for(int idxChild = 0 ; idxChild < 8 ; ++idxChild){
142 potentialChild[idxChild] = nullptr;
143 if(realChild[idxChild]){
144 if(realChild[idxChild]->hasSrcChild()){
145 currentCell->setSrcChildTrue();
146 potentialChild[idxChild] = realChild[idxChild];
147 }
148 if(realChild[idxChild]->hasTargetsChild()){
149 currentCell->setTargetsChildTrue();
150 }
151 }
152 }
153 kernels->M2M( currentCell , potentialChild, idxLevel);
154
155 FLOG(computationCounter.tac());
156 FLOG(totalComputation += computationCounter.elapsed());
157 } while(octreeIterator.moveRight());
158
159 avoidGotoLeftIterator.moveUp();
160 octreeIterator = avoidGotoLeftIterator;// equal octreeIterator.moveUp(); octreeIterator.gotoLeft();
161 FLOG( FLog::Controller << "\t\t>> Level " << idxLevel << " = " << counterTimeLevel.tacAndElapsed() << " s\n" );
162 }
163
164 FLOG( counterTime.tac() );
165 FLOG( FLog::Controller << "\tFinished (@Upward Pass (M2M) = " << counterTime.elapsed() << " s)\n" );
166 FLOG( FLog::Controller << "\t\t Computation : " << totalComputation << " s\n" );
167
168 }

171 void transferPass(){
172 FLOG( FLog::Controller.write("\tStart Downward Pass (M2L)\n").write(FLog::Flush); );
173 FLOG( counterTime.tic() );
174 FLOG( double totalComputation = 0 );
175
176 typename OctreeClass::Iterator octreeIterator(tree);
177 octreeIterator.moveDown();
178
179 for(int idxLevel = 2 ; idxLevel < FAbstractAlgorithm::upperWorkingLevel ; ++idxLevel){
180 octreeIterator.moveDown();
181 }
182
183 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
184
185 const CellClass* neighbors[342];
186 int neighborPositions[342];
187
188 // for each levels
189 for(int idxLevel = FAbstractAlgorithm::upperWorkingLevel ; idxLevel < FAbstractAlgorithm::lowerWorkingLevel ; ++idxLevel ){
190 FLOG(FTic counterTimeLevel);
191 const int separationCriteria = (idxLevel != FAbstractAlgorithm::lowerWorkingLevel-1 ? 1 : leafLevelSeparationCriteria);
192 // for each cells
193 do{
194 FLOG(computationCounter.tic());
195 CellClass* const currentCell = octreeIterator.getCurrentCell();
196
197 if(currentCell->hasTargetsChild()){
198 const int counter = tree->getInteractionNeighbors(neighbors, neighborPositions, octreeIterator.getCurrentGlobalCoordinate(),idxLevel, separationCriteria);
199 if( counter ){
200 int counterWithSrc = 0;
201 for(int idxRealNeighbors = 0 ; idxRealNeighbors < counter ; ++idxRealNeighbors ){
202 if(neighbors[idxRealNeighbors]->hasSrcChild()){
203 neighbors[counterWithSrc] = neighbors[idxRealNeighbors];
204 neighborPositions[counterWithSrc] = neighborPositions[idxRealNeighbors];
205 ++counterWithSrc;
206 }
207 }
208 if(counterWithSrc){
209 kernels->M2L( currentCell , neighbors, neighborPositions, counterWithSrc, idxLevel);
210 }
211 }
212 }
213 FLOG(computationCounter.tac());
214 FLOG(totalComputation += computationCounter.elapsed());
215 } while(octreeIterator.moveRight());
216
217 FLOG(computationCounter.tic());
218 kernels->finishedLevelM2L(idxLevel);
219 FLOG(computationCounter.tac());
220
221 avoidGotoLeftIterator.moveDown();
222 octreeIterator = avoidGotoLeftIterator;
223 FLOG( FLog::Controller << "\t\t>> Level " << idxLevel << " = " << counterTimeLevel.tacAndElapsed() << " s\n" );
224 }
225
226 FLOG( counterTime.tac() );
227 FLOG( FLog::Controller << "\tFinished (@Downward Pass (M2L) = " << counterTime.elapsed() << " s)\n" );
228 FLOG( FLog::Controller << "\t\t Computation : " << totalComputation << " s\n" );

247 // for each levels exepted leaf level
248 for(int idxLevel = FAbstractAlgorithm::upperWorkingLevel ; idxLevel < heightMinusOne ; ++idxLevel ){
249 FLOG(FTic counterTimeLevel);
250 // for each cells
251 do{
252 if( octreeIterator.getCurrentCell()->hasTargetsChild() ){
253 FLOG(computationCounter.tic());

259 potentialChild[idxChild] = realChild[idxChild];

262 potentialChild[idxChild] = nullptr;

284 void directPass(const bool p2pEnabled, const bool l2pEnabled){
285 FLOG( FLog::Controller.write("\tStart Direct Pass\n").write(FLog::Flush); );
286 FLOG( counterTime.tic() );
287 FLOG( double totalComputation = 0 );
288
289 const int heightMinusOne = OctreeHeight - 1;
290
291 typename OctreeClass::Iterator octreeIterator(tree);
292 octreeIterator.gotoBottomLeft();
293 // There is a maximum of 26 neighbors
294 ContainerClass* neighbors[26];
295 int neighborPositions[26];
296 // for each leafs
297 do{
298 if( octreeIterator.getCurrentCell()->hasTargetsChild() ){
299 FLOG(computationCounter.tic());
300 if(l2pEnabled){
301 kernels->L2P(octreeIterator.getCurrentCell(), octreeIterator.getCurrentListTargets());
302 }
303 if(p2pEnabled){
304 if(octreeIterator.getCurrentCell()->hasSrcChild()){
305 kernels->P2P( octreeIterator.getCurrentGlobalCoordinate(), octreeIterator.getCurrentListTargets(),
306 octreeIterator.getCurrentListSrc() , neighbors, neighborPositions, 0);
307 }
308 // need the current particles and neighbors particles
309 const int counter = tree->getLeafsNeighbors(neighbors, neighborPositions, octreeIterator.getCurrentGlobalCoordinate(), heightMinusOne);
310 kernels->P2PRemote( octreeIterator.getCurrentGlobalCoordinate(), octreeIterator.getCurrentListTargets(),
311 octreeIterator.getCurrentListSrc() , neighbors, neighborPositions, counter);
312 }
313 FLOG(computationCounter.tac());
314 FLOG(totalComputation += computationCounter.elapsed());
315 }
316 } while(octreeIterator.moveRight());
317
318 FLOG( counterTime.tac() );
319 FLOG( FLog::Controller << "\tFinished (@Direct Pass (L2P + P2P) = " << counterTime.elapsed() << " s)\n" );
320 FLOG( FLog::Controller << "\t\t Computation L2P + P2P : " << totalComputation << " s\n" );
321
322 }
