318template <class FReal>
319FReal plummerDist(FSize& cpt, const FReal &R) {
320 FReal radius, u;
321 while(true) {
322 u = FMath::pow(getRandom<FReal>(), 2.0/3.0);
323 radius = FMath::Sqrt(u/(1.0-u));
324 ++cpt;
325 if(radius <= R) {
326 return static_cast<FReal>(radius);
327 }
328 }
329}

347 //Pb of large box
348 constexpr const FReal rand_max = 0.8;
349 const FReal r_max = std::sqrt(1.0/(std::pow(rand_max, -2.0/3.0) - 1.0));

358 points[j+1] *= r;
359 points[j+2] *= r;
360 points[j+3] = mc; // the mass
