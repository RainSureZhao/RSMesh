94 FGroupOfParticles(unsigned char* inBuffer, const size_t inAllocatedMemoryInByte,
95 unsigned char* inAttributes)
96 : allocatedMemoryInByte(inAllocatedMemoryInByte), memoryBuffer(inBuffer),
97 blockHeader(nullptr), leafHeader(nullptr),
98 attributesBuffer(nullptr), deleteBuffer(false){
99 // Move the pointers to the correct position
100 blockHeader = reinterpret_cast<BlockHeader*>(inBuffer);
101 inBuffer += sizeof(BlockHeader);
102 leafHeader = reinterpret_cast<LeafHeader*>(inBuffer);
103
104 // Init particle pointers
105 FAssertLF( blockHeader->positionsLeadingDim == (sizeof(FReal) * blockHeader->nbParticlesAllocatedInGroup) );
106 particlePosition[0] = reinterpret_cast<FReal*>(memoryBuffer + blockHeader->offsetPosition);
107 particlePosition[1] = (particlePosition[0] + blockHeader->nbParticlesAllocatedInGroup);
108 particlePosition[2] = (particlePosition[1] + blockHeader->nbParticlesAllocatedInGroup);
109
110 // Redirect pointer to data
111 FAssertLF(blockHeader->attributeLeadingDim == (sizeof(AttributeClass) * blockHeader->nbParticlesAllocatedInGroup));
112 AttributeClass* symAttributes = (AttributeClass*)(&particlePosition[2][blockHeader->nbParticlesAllocatedInGroup]);
113 for(unsigned idxAttribute = 0 ; idxAttribute < NbSymbAttributes ; ++idxAttribute){
114 particleAttributes[idxAttribute] = symAttributes;
115 symAttributes += blockHeader->nbParticlesAllocatedInGroup;
116 }
117#ifndef SCALFMM_SIMGRID_NODATA
118 if(inAttributes){
119 attributesBuffer = (AttributeClass*)inAttributes;
120 for(unsigned idxAttribute = 0 ; idxAttribute < NbAttributesPerParticle ; ++idxAttribute){
121 particleAttributes[idxAttribute+NbSymbAttributes] = &attributesBuffer[idxAttribute*blockHeader->nbParticlesAllocatedInGroup];
122 }
123 }
124#endif
125 }

133 FGroupOfParticles(const MortonIndex inStartingIndex, const MortonIndex inEndingIndex, const int inNumberOfLeaves, const FSize inNbParticles)
134 : allocatedMemoryInByte(0), memoryBuffer(nullptr), blockHeader(nullptr), leafHeader(nullptr),
135 deleteBuffer(true){
136 memset(particlePosition, 0, sizeof(particlePosition));
137 memset(particleAttributes, 0, sizeof(particleAttributes));
138
139 const FSize nbParticlesAllocatedInGroup = RoundToUpperParticles(inNbParticles+(MemoryAlignementParticles-1)*inNumberOfLeaves);
140
141 // Find the number of leaf to allocate in the blocks
142 FAssertLF((inEndingIndex-inStartingIndex) >= MortonIndex(inNumberOfLeaves));
143 // Total number of bytes in the block
144 const size_t sizeOfOneParticle = (3*sizeof(FReal) + NbSymbAttributes*sizeof(AttributeClass));
145 const size_t memoryToAlloc = sizeof(BlockHeader)
146 + (inNumberOfLeaves*sizeof(MortonIndex))
147 + (inNumberOfLeaves*sizeof(LeafHeader))
148 + nbParticlesAllocatedInGroup*sizeOfOneParticle;
149
150 // Allocate
151 FAssertLF(0 <= int(memoryToAlloc) && int(memoryToAlloc) < std::numeric_limits<int>::max());
152 allocatedMemoryInByte = memoryToAlloc;
153 memoryBuffer = (unsigned char*)FAlignedMemory::AllocateBytes<MemoryAlignementBytes>(memoryToAlloc);
154 FAssertLF(memoryBuffer);
155 memset(memoryBuffer, 0, memoryToAlloc);
156
157 // Move the pointers to the correct position
158 unsigned char* bufferPtr = memoryBuffer;
159 blockHeader = reinterpret_cast<BlockHeader*>(bufferPtr);
160 bufferPtr += sizeof(BlockHeader);
161 leafHeader = reinterpret_cast<LeafHeader*>(bufferPtr);
162
163 // Init header
164 blockHeader->startingIndex = inStartingIndex;
165 blockHeader->endingIndex = inEndingIndex;
166 blockHeader->numberOfLeavesInBlock = inNumberOfLeaves;
167 blockHeader->nbParticlesAllocatedInGroup = nbParticlesAllocatedInGroup;
168 blockHeader->nbParticlesInGroup = inNbParticles;
169
170 // Init particle pointers
171 blockHeader->positionsLeadingDim = (sizeof(FReal) * nbParticlesAllocatedInGroup);
172 particlePosition[0] = reinterpret_cast<FReal*>((reinterpret_cast<size_t>(leafHeader + inNumberOfLeaves)
173 +MemoryAlignementBytes-1) & ~(MemoryAlignementBytes-1));
174 particlePosition[1] = (particlePosition[0] + nbParticlesAllocatedInGroup);
175 particlePosition[2] = (particlePosition[1] + nbParticlesAllocatedInGroup);
176
177 blockHeader->offsetPosition = size_t(particlePosition[0]) - size_t(memoryBuffer);
178
179 // Redirect pointer to data
180 blockHeader->attributeLeadingDim = (sizeof(AttributeClass) * nbParticlesAllocatedInGroup);
181
182 AttributeClass* symAttributes = (AttributeClass*)(&particlePosition[2][blockHeader->nbParticlesAllocatedInGroup]);
183 for(unsigned idxAttribute = 0 ; idxAttribute < NbSymbAttributes ; ++idxAttribute){
184 particleAttributes[idxAttribute] = symAttributes;
185 symAttributes += blockHeader->nbParticlesAllocatedInGroup;
186 }
187#ifndef SCALFMM_SIMGRID_NODATA
188 attributesBuffer = (AttributeClass*)FAlignedMemory::AllocateBytes<MemoryAlignementBytes>(blockHeader->attributeLeadingDim*NbAttributesPerParticle);
189 memset(attributesBuffer, 0, blockHeader->attributeLeadingDim*NbAttributesPerParticle);
190 for(unsigned idxAttribute = 0 ; idxAttribute < NbAttributesPerParticle ; ++idxAttribute){
191 particleAttributes[idxAttribute+NbSymbAttributes] = &attributesBuffer[idxAttribute*nbParticlesAllocatedInGroup];
192 }
193#else
194 attributesBuffer = nullptr;
195#endif
196
197 // Set all index to not used
198 for(int idxLeafPtr = 0 ; idxLeafPtr < inNumberOfLeaves ; ++idxLeafPtr){
199 leafHeader[idxLeafPtr].mindex = -1;
200 }
201 }

278 int getLeafIndex(const MortonIndex leafIdx) const{
279 int idxLeft = 0;
280 int idxRight = blockHeader->numberOfLeavesInBlock-1;
281 while(idxLeft <= idxRight){
282 const int idxMiddle = (idxLeft+idxRight)/2;
283 if(leafHeader[idxMiddle].mindex == leafIdx){
284 return idxMiddle;
285 }
286 if(leafIdx < leafHeader[idxMiddle].mindex){
287 idxRight = idxMiddle-1;
288 }
289 else{
290 idxLeft = idxMiddle+1;
291 }
292 }
293 return -1;
294 }

302 size_t newLeaf(const MortonIndex inIndex, const int id, const FSize nbParticles, const size_t offsetInGroup){
303 FAssertLF(isInside(inIndex));
304 FAssertLF(!exists(inIndex));
305 FAssertLF(id < blockHeader->numberOfLeavesInBlock);
306 FAssertLF(offsetInGroup < size_t(blockHeader->nbParticlesAllocatedInGroup));
307 leafHeader[id].nbParticles = nbParticles;
308 leafHeader[id].offSet = offsetInGroup;
309 leafHeader[id].mindex = inIndex;
310
311 const size_t nextLeafOffsetInGroup = RoundToUpperParticles(offsetInGroup+nbParticles);
312 FAssertLF(nextLeafOffsetInGroup <= size_t(blockHeader->nbParticlesAllocatedInGroup + MemoryAlignementParticles));
313 return nextLeafOffsetInGroup;
314 }

361 void extractData(const std::vector<int>& inLeavesIdxToExtract,
362 unsigned char* outputBuffer, const size_t outputBufferSize) const {
363 size_t idxValue = 0;
364 for(size_t idxEx = 0 ; idxEx < inLeavesIdxToExtract.size() ; ++idxEx){
365 const int idLeaf = inLeavesIdxToExtract[idxEx];
366 for(unsigned idxAttr = 0 ; idxAttr < NbSymbAttributes ; ++idxAttr){
367 const FReal* ptrData = particlePosition[0] + leafHeader[idLeaf].offSet
368 + blockHeader->positionsLeadingDim*idxAttr;
369 for(int idxPart = 0 ; idxPart < leafHeader[idLeaf].nbParticles ; ++idxPart){
370 ((FReal*)outputBuffer)[idxValue++] = ptrData[idxPart];
371 }
372 }
373 }
374 FAssertLF(idxValue*sizeof(FReal) == outputBufferSize);
