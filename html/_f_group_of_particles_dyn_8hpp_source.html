93 FGroupOfParticlesDyn(const MortonIndex inStartingIndex, const MortonIndex inEndingIndex, const int inNumberOfLeaves,
94 const size_t sizePerLeafSymb[], const size_t sizePerLeafDown[])
95 : allocatedMemoryInByteSymb(0), memoryBuffer(nullptr), blockHeader(nullptr), leafHeader(nullptr),
96 symbPart(nullptr), downPart(nullptr), allocatedMemoryInByteDown(0), deleteBuffer(true){
97
98 size_t totalSymb = 0;
99 size_t totalDown = 0;
100 for(int idxLeaf = 0 ; idxLeaf < inNumberOfLeaves ; ++idxLeaf){
101 totalSymb += sizePerLeafSymb[idxLeaf];
102 totalDown += sizePerLeafDown[idxLeaf];
103 }
104
105 FAssertLF((inEndingIndex-inStartingIndex) >= MortonIndex(inNumberOfLeaves));
106 // Total number of bytes in the block
107 const size_t memoryToAllocSymb = sizeof(BlockHeader)
108 + (inNumberOfLeaves*sizeof(LeafHeader))
109 + (totalSymb+MemoryAlignementBytes-1);
110
111 // Allocate
112 allocatedMemoryInByteSymb = memoryToAllocSymb;
113 memoryBuffer = (unsigned char*)FAlignedMemory::AllocateBytes<MemoryAlignementBytes>(memoryToAllocSymb);
114 FAssertLF(memoryBuffer || !totalSymb);
115 memset(memoryBuffer, 0, memoryToAllocSymb);
116
117 allocatedMemoryInByteDown = totalDown;
118 downPart = (unsigned char*)FAlignedMemory::AllocateBytes<MemoryAlignementBytes>(allocatedMemoryInByteDown);
119 FAssertLF(downPart || !totalDown);
120 memset(downPart, 0, allocatedMemoryInByteDown);
121
122 // Move the pointers to the correct position
123 unsigned char* bufferPtr = memoryBuffer;
124 blockHeader = reinterpret_cast<BlockHeader*>(bufferPtr);
125 bufferPtr += sizeof(BlockHeader);
126 leafHeader = reinterpret_cast<LeafHeader*>(bufferPtr);
127 bufferPtr += (inNumberOfLeaves*sizeof(LeafHeader));
128 symbPart = reinterpret_cast<unsigned char*>(size_t(bufferPtr+MemoryAlignementBytes-1) & ~size_t(MemoryAlignementBytes-1));
129
130 FAssertLF(size_t(bufferPtr-memoryBuffer) <= allocatedMemoryInByteSymb);
131
132 // Init header
133 blockHeader->startingIndex = inStartingIndex;
134 blockHeader->endingIndex = inEndingIndex;
135 blockHeader->numberOfLeavesInBlock = inNumberOfLeaves;
136 blockHeader->offsetSymbPart = size_t(symbPart)-size_t(blockHeader);
137
138 if(inNumberOfLeaves){
139 leafHeader[0].mindex = -1;
140 leafHeader[0].sizeSymb = sizePerLeafSymb[0];
141 leafHeader[0].sizeDown = sizePerLeafDown[0];
142 leafHeader[0].offSetSymb = 0;
143 leafHeader[0].offSetDown = 0;
144
145 for(int idxLeaf = 1 ; idxLeaf < inNumberOfLeaves ; ++idxLeaf){
146 leafHeader[idxLeaf].mindex = -1;
147 leafHeader[idxLeaf].sizeSymb = sizePerLeafSymb[idxLeaf];
148 leafHeader[idxLeaf].sizeDown = sizePerLeafDown[idxLeaf];
149 leafHeader[idxLeaf].offSetSymb = leafHeader[idxLeaf-1].sizeSymb + leafHeader[idxLeaf-1].offSetSymb;
150 leafHeader[idxLeaf].offSetDown = leafHeader[idxLeaf-1].sizeDown + leafHeader[idxLeaf-1].offSetDown;
151 }
152 FAssertLF(leafHeader[inNumberOfLeaves-1].offSetSymb+leafHeader[inNumberOfLeaves-1].sizeSymb == totalSymb);
153 FAssertLF(leafHeader[inNumberOfLeaves-1].offSetDown+leafHeader[inNumberOfLeaves-1].sizeDown == totalDown);
154 }
155 }

227 int getLeafIndex(const MortonIndex leafIdx) const{
228 int idxLeft = 0;
229 int idxRight = blockHeader->numberOfLeavesInBlock-1;
230 while(idxLeft <= idxRight){
231 const int idxMiddle = (idxLeft+idxRight)/2;
232 if(leafHeader[idxMiddle].mindex == leafIdx){
233 return idxMiddle;
234 }
235 if(leafIdx < leafHeader[idxMiddle].mindex){
236 idxRight = idxMiddle-1;
237 }
238 else{
239 idxLeft = idxMiddle+1;
240 }
241 }
242 return -1;
243 }
