86 void computeLegendre(const FReal inCosTheta, const FReal inSinTheta){
87 const FReal invSinTheta = -inSinTheta;
88
89 legendre[0] = 1.0; // P_0,0(cosTheta) = 1
90 legendre[1] = inCosTheta; // P_1,0(cosTheta) = cosTheta
91 legendre[2] = invSinTheta;// P_1,1(cosTheta) = -sinTheta
92
93 int idxCurrentLM = 3; //current pointer on P_l,m
94 int idxCurrentL1M = 1; //pointer on P_{l-1},m => P_1,0
95 int idxCurrentL2M = 0; //pointer on P_{l-2},m => P_0,0
96 FReal fact = 3.0;
97
98 for(int l = 2; l <= devP ; ++l ){
99 // m from 0 to l - 2
100 for( int m = 0; m <= l - 2 ; ++m ){
101 // cosTheta x (2 x l - 1) x P_l-1_m - (l + m - 1) x P_l-2_m
102 // P_l_m = --------------------------------------------------------
103 // l - m
104 legendre[idxCurrentLM] = (inCosTheta * FReal( 2 * l - 1 ) * legendre[idxCurrentL1M]
105 - FReal( l + m - 1 ) * legendre[idxCurrentL2M] )
106 / FReal( l - m );
107
108
109 // progress
110 ++idxCurrentLM;
111 ++idxCurrentL1M;
112 ++idxCurrentL2M;
113 }
114
115 // Compute P_l,{l-1}
116 legendre[idxCurrentLM++] = inCosTheta * FReal( 2 * l - 1 ) * legendre[idxCurrentL1M];
117
118 // Compute P_l,l
119 legendre[idxCurrentLM++] = fact * invSinTheta * legendre[idxCurrentL1M];
120
121 fact += FReal(2.0);
122 ++idxCurrentL1M;
123 }

207 // For i^|m|

237 // Not used : const FReal PiArrayInner[4] = {0, FMath::FPiDiv2<FReal>(), FMath::FPi<FReal>(), -FMath::FPiDiv2<FReal>()};

303 void computeInnerTheta(const FSpherical<FReal>& inSphere){
304 const FReal PiArrayInner[4] = {0, FMath::FPiDiv2<FReal>(), FMath::FPi<FReal>(), -FMath::FPiDiv2<FReal>()};
305 computeCosSin(inSphere.getPhi(),PiArrayInner);
306
307 // fill legendre array
308 computeLegendre(inSphere.getCosTheta(), inSphere.getSinTheta());
309
310 int index_l_m = 0;
311 // r^l
312 FReal r_pow_l = 1.0;
313
314 for (int l = 0 ; l <= devP ; ++l){
315 FReal magnitude;
316 // m<l:
317 int m = 0;
318 for(; m < l ; ++m){
319 magnitude = sphereHarmoInnerCoef[index_l_m] * r_pow_l * legendre[index_l_m];
320
321 // Computation of Inner_l^m(r, theta, phi):
322 harmonic[index_l_m].setReal( magnitude * cosSin[m].getReal());
323 harmonic[index_l_m].setImag( magnitude * cosSin[m].getImag());
324
325 // Computation of {\partial Inner_l^m(r, theta, phi)}/{\partial theta}:
326 magnitude = sphereHarmoInnerCoef[index_l_m] * r_pow_l * ((FReal(l)*inSphere.getCosTheta()*legendre[index_l_m]
327 - FReal(l+m)*(legendre[getPreExpRedirJ(l-1) + m])) / inSphere.getSinTheta());
328 thetaDerivatedResult[index_l_m].setReal(magnitude * cosSin[m].getReal());
329 thetaDerivatedResult[index_l_m].setImag(magnitude * cosSin[m].getImag());
330
331 ++index_l_m;
332 }
333
334 // m=l:
335 // Computation of Inner_m^m(r, theta, phi):
336 magnitude = sphereHarmoInnerCoef[index_l_m] * r_pow_l * legendre[index_l_m];
337 harmonic[index_l_m].setReal(magnitude * cosSin[m].getReal());
338 harmonic[index_l_m].setImag(magnitude * cosSin[m].getImag());
339
340 // Computation of {\partial Inner_m^m(r, theta, phi)}/{\partial theta}:
341 magnitude = sphereHarmoInnerCoef[index_l_m] * r_pow_l * (FReal(m) * inSphere.getCosTheta() * legendre[index_l_m] / inSphere.getSinTheta());
342 thetaDerivatedResult[index_l_m].setReal(magnitude * cosSin[m].getReal());
343 thetaDerivatedResult[index_l_m].setImag(magnitude * cosSin[m].getImag());
344
345 r_pow_l *= inSphere.getR();
346 ++index_l_m;
347 }
