101 * r(x)=exp(-(|x|/l)^2)

103 FReal evaluate(const FReal* x, const FReal* y) const
104 {
105 FReal dist2 = FReal(0.0);
106 for(int d=0; d<3; ++d){
107 FReal distX = FMath::Abs(x[d]-y[d]) / lengthScale_;
108 dist2 += distX*distX;
109 }
110
111 FReal res = FMath::Exp(FReal(-0.5)*dist2);
112
113 return res;
114 }

116 // evaluate interaction
117 template <class ValueClass>
118 ValueClass evaluate(const ValueClass& x1, const ValueClass& y1, const ValueClass& z1,
119 const ValueClass& x2, const ValueClass& y2, const ValueClass& z2) const
120 {
121 const ValueClass diff[3] = {(x1-x2),(y1-y2),(z1-z2)};
122
123 ValueClass dist2 = FMath::Zero<ValueClass>();
124 for(int d=0; d<3; ++d){
125 const ValueClass distX = diff[d] / FMath::ConvertTo<ValueClass,FReal>(lengthScale_);
126 dist2 += distX*distX;
127 }
128
129 return FMath::Exp(FMath::ConvertTo<ValueClass,FReal>(-0.5)*dist2);
