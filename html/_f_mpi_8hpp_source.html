238 void groupReduce(const int from , const int to){
239 int * procsIdArray = new int [to - from + 1];
240 for(int idxProc = from ;idxProc <= to ; ++idxProc){
241 procsIdArray[idxProc - from] = idxProc;
242 }
243
244 MPI_Group previousGroup = group;
245 FMpi::Assert( MPI_Group_incl(previousGroup, to - from + 1 , procsIdArray, &group), __LINE__ );
246
247 MPI_Comm previousComm = communicator;
248 FMpi::Assert( MPI_Comm_create(previousComm, group, &communicator), __LINE__ );
249
250 MPI_Comm_free(&previousComm);
251 MPI_Group_free(&previousGroup);
252
253 reset();
254 delete[] procsIdArray ;
255 }

260 void groupReduce(const int processInGroup[]){
261 int * procsIdArray = new int [nbProc];
262 int counterNewGroup = 0;
263 for(int idxProc = 0 ;idxProc < nbProc ; ++idxProc){
264 if(processInGroup[rank] && processInGroup[idxProc]){
265 procsIdArray[counterNewGroup++] = idxProc;
266 }
267 else if(!processInGroup[rank] && !processInGroup[idxProc]){
268 procsIdArray[counterNewGroup++] = idxProc;
269 }
270 }
271
272 MPI_Group previousGroup = group;
273 FMpi::Assert( MPI_Group_incl(previousGroup, counterNewGroup , procsIdArray, &group), __LINE__ );
274
275 MPI_Comm previousComm = communicator;
276 FMpi::Assert( MPI_Comm_create(previousComm, group, &communicator), __LINE__ );
277
278 MPI_Comm_free(&previousComm);
279 MPI_Group_free(&previousGroup);
280
281 reset();
282 FAssertLF(nbProc == counterNewGroup);
283 delete[] procsIdArray ;
284 }

475 template <class ObjectType, class VectorType>
476 static int ISendSplit(const ObjectType toSend[], const size_t nbItems,
477 const int dest, const int tagBase, const FMpi::FComm& communicator,
478 VectorType* requestVector){
479 const size_t totalByteToSend = (nbItems*sizeof(ObjectType));
480 unsigned char*const ptrDataToSend = (unsigned char*)const_cast<ObjectType*>(toSend);
481 for(size_t idxSize = 0 ; idxSize < totalByteToSend ; idxSize += MaxBytesPerDivMess){
482 MPI_Request currentRequest;
483 const size_t nbBytesInMessage = FMath::Min(MaxBytesPerDivMess, totalByteToSend-idxSize);
484 FAssertLF(nbBytesInMessage < std::numeric_limits<int>::max());
485 FMpi::Assert( MPI_Isend(&ptrDataToSend[idxSize], int(nbBytesInMessage), MPI_BYTE , dest,
486 tagBase + int(idxSize/MaxBytesPerDivMess), communicator.getComm(), &currentRequest) , __LINE__);
487
488 requestVector->push_back(currentRequest);
489 }
490 return int((totalByteToSend+MaxBytesPerDivMess-1)/MaxBytesPerDivMess);
491 }
