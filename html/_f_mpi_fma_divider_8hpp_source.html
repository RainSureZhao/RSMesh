308 _nbRecordsPerLine = loader.getNbRecordPerline();

322 FLOG(

349 void writeFiles() {
350
351 std::vector<std::vector<FmaParticle> > filesParticles(_splitCount);
352
353 // Temporary vector to store all particles with same Morton index
354 std::vector<FmaParticle> container;
355
356 using size_type = typename std::vector<FmaParticle>::size_type;
357
358 size_type totalCount = _particleCount;
359 size_type currentCount = 0;
360 MortonIndex currentMortonIdx = 0;
361 long long int currentFileIdx = 0;
362
363 if( BOXES == _dispatchPolicy ) {
364 totalCount = 1;
365 for ( auto p : _particles ) {
366 if( p.getMortonIndex() != currentMortonIdx) {
367 totalCount++;
368 currentMortonIdx = p.getMortonIndex();
369 }
370 }
371 currentMortonIdx = 0;
372 }
373
374 // FIND OUT PARTICLES' FILE ////////////////////////////////////////////
375 FLOG( // Choose file progress log
376 FRepeatAction progress(
377 [&]() -> bool {
378 std::cerr
379 << " \rCreating particle-files association: "
380 << currentCount * 100 / totalCount
381 << "%" ;
382 return true; }, 200 ); );
384
385 while(_particles.begin() != _particles.end()) {
386 /* Particles are sorted, we pull out all those with the same Morton
387 * index and put them in the temporary container. */
388 while( _particles.begin() != _particles.end()
389 && (*_particles.begin()).getMortonIndex() == currentMortonIdx) {
390 container.push_back(*(_particles.begin()));
391 _particles.erase(_particles.begin());
392 }
393
394
395 // File number
396 currentFileIdx = currentCount * _splitCount
397 / totalCount;
398
399 // Set particles to file.
400 for(FmaParticle p : container) {
401 filesParticles[currentFileIdx].push_back(p);
402 }
403
404 if( BOXES == _dispatchPolicy ) {
405 currentCount++;
406 } else if ( PARTICLES == _dispatchPolicy ) {
407 currentCount += container.size();
408 }
409 container.clear();
410 currentMortonIdx = (*_particles.begin()).getMortonIndex();
411 }
412
414 FLOG({
415 progress.stop();
416 std::cerr << std::endl;
417 std::cout << "@@ distribute-particle-time:"
418 << progress.cumulated()
419 << std::endl;
420 });
422
423 currentFileIdx = 0;
424 currentCount = 0;
425
426
427
428 // WRITE FILES /////////////////////////////////////////////////////////
429 FLOG(
430 std::cerr << "\rWriting particles to files... ";
431 progress.setFunction(
432 [&]() -> bool {
433 std::cerr
434 << " \rWriting particles to file "
435 << currentFileIdx
436 << ": "
437 << currentCount * 100 / _particleCount
438 << "%" ;
439 return true; } );
440 progress.start();
441 );
443
444 // Write common file header
445 {
446 FFmaGenericWriter<FReal>
447 commonWriter(_outputBasename + ".main" + _outputExt);
448 commonWriter.writeHeader(FmaParticle::_boxCenter,
449 FmaParticle::_boxWidth,
450 _particleCount,
451 _dataType,
452 _nbRecordsPerLine);
453 }
454
455 // Write files
456 for ( std::vector<FmaParticle>& filecontent : filesParticles ) {
457 // Open the current file writer
458 FFmaGenericWriter<FReal>
459 writer(_outputBasename + "." + std::to_string(currentFileIdx)
460 + _outputExt);
461
462 writer.writeHeader(FmaParticle::_boxCenter,
463 FmaParticle::_boxWidth,
464 filecontent.size(),
465 _dataType,
466 _nbRecordsPerLine);
467
468 for( FmaParticle p : filecontent ) {
469 writer.writeArrayOfReal(p.getPtrFirstData(),
470 p.getWriteDataNumber(), 1);
471
472 FLOG(currentCount++);
473 }
474 currentFileIdx++;
475 }
476
478 FLOG({
479 progress.stop();
480 std::cerr << std::endl;
481 std::cout << "@@ write-particle-time:"
482 << progress.cumulated()
483 << std::endl;
484 });
486
487 }
