92

102 }

151 static void MergeSplitedLeaves(const FMpi::FComm& communicator, IndexedParticle** workingArray, FSize* workingSize,
152 FSize ** leavesOffsetInParticles, ParticleClass** particlesArrayInLeafOrder, FSize* const leavesSize){
153 const int myRank = communicator.processId();
154 const int nbProcs = communicator.processCount();
155
156 FVector<LeafInfo> leavesInfo;
157 { // Get the information of the leaves
158 leavesInfo.clear();
159 if((*workingSize)){
160 leavesInfo.push({(*workingArray)[0].index, 1, 0});
161 for(FSize idxPart = 1 ; idxPart < (*workingSize) ; ++idxPart){
162 if(leavesInfo.data()[leavesInfo.getSize()-1].mindex == (*workingArray)[idxPart].index){
163 leavesInfo.data()[leavesInfo.getSize()-1].nbParts += 1;
164 }
165 else{
166 leavesInfo.push({(*workingArray)[idxPart].index, 1, idxPart});
167 }
168 }
169 }
170 }
171
172 if(nbProcs != 1){
173 // Some leaf might be divived on several processes, we should move them to the first process
174 const MortonIndex noDataFlag = std::numeric_limits<MortonIndex>::max();
175
176 LeafInfo borderLeavesState[2] = { {noDataFlag, 0, 0}, {noDataFlag, 0, 0} };
177 if( (*workingSize) != 0 ){
178 borderLeavesState[0] = leavesInfo[0];
179 borderLeavesState[1] = leavesInfo[leavesInfo.getSize()-1];
180 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] First " << borderLeavesState[0].mindex << "\n"; FLog::Controller.flush(); );
181 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Last " << borderLeavesState[1].mindex << "\n"; FLog::Controller.flush(); );
182 }
183
184 std::unique_ptr<LeafInfo[]> allProcFirstLeafStates(new LeafInfo[nbProcs*2]);
185 FMpi::MpiAssert(MPI_Allgather(&borderLeavesState, sizeof(LeafInfo)*2, MPI_BYTE,
186 allProcFirstLeafStates.get(), sizeof(LeafInfo)*2, MPI_BYTE, communicator.getComm()),__LINE__);
187
188 std::vector<MPI_Request> requests;
189
190 // Find what to send/recv from who
191 bool hasSentFirstLeaf = false;
192 if( (*workingSize) != 0 ){
193 // Find the owner of the leaf
194 int idProcToSendTo = myRank;
195 while(0 < idProcToSendTo &&
196 (allProcFirstLeafStates[(idProcToSendTo-1)*2 + 1].mindex == borderLeavesState[0].mindex
197 || allProcFirstLeafStates[(idProcToSendTo-1)*2 + 1].mindex == noDataFlag)){
198 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] idProcToSendTo "
199 << idProcToSendTo << " allProcFirstLeafStates[(idProcToSendTo-1)*2 + 1].mindex " <<
200 allProcFirstLeafStates[(idProcToSendTo-1)*2 + 1].mindex << " borderLeavesState[0].mindex " <<
201 borderLeavesState[0].mindex << "\n"; FLog::Controller.flush(); );
202 idProcToSendTo -= 1;
203 }
204 // We found someone
205 if(idProcToSendTo != myRank && allProcFirstLeafStates[(idProcToSendTo)*2 + 1].mindex == borderLeavesState[0].mindex){
206 // Post and send message for the first leaf
207 FMpi::ISendSplit(&(*workingArray)[0], borderLeavesState[0].nbParts, idProcToSendTo,
208 FMpi::TagExchangeIndexs, communicator, &requests);
209 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] send " << borderLeavesState[0].nbParts << " to " << idProcToSendTo << "\n"; FLog::Controller.flush(); );
210 hasSentFirstLeaf = true;
211 }
212 }
213
214 bool hasExtendLastLeaf = false;
215 std::vector<IndexedParticle> receivedParticles;
216
217 {
218 // Count all the particle of our first leaf on other procs
219 FSize totalNbParticlesToRecv = 0;
220 int idProcToRecvFrom = myRank;
221 if(!hasSentFirstLeaf || borderLeavesState[0].mindex != borderLeavesState[1].mindex){
222 while(idProcToRecvFrom+1 < nbProcs &&
223 (borderLeavesState[1].mindex == allProcFirstLeafStates[(idProcToRecvFrom+1)*2].mindex
224 || allProcFirstLeafStates[(idProcToRecvFrom+1)*2].mindex == noDataFlag)){
225 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] idProcToRecvFrom "
226 << idProcToRecvFrom << " allProcFirstLeafStates[(idProcToRecvFrom+1)*2].mindex " <<
227 allProcFirstLeafStates[(idProcToRecvFrom+1)*2].mindex << " borderLeavesState[1].mindex " <<
228 borderLeavesState[1].mindex << "\n"; FLog::Controller.flush(); );
229
230 idProcToRecvFrom += 1;
231 totalNbParticlesToRecv += allProcFirstLeafStates[(idProcToRecvFrom)*2].nbParts;
232 }
233 }
234 // If there are some
235 if(totalNbParticlesToRecv){
236 // Alloc a received buffer
237 receivedParticles.resize(totalNbParticlesToRecv);
238 // Post the recv
239 FSize postPositionRecv = 0;
240 for(int postRecvIdx = (myRank+1); postRecvIdx <= idProcToRecvFrom ; ++postRecvIdx){
241 // If there are some on this proc
242 if(allProcFirstLeafStates[(postRecvIdx)*2].mindex != noDataFlag){
243 FMpi::IRecvSplit(&receivedParticles[postPositionRecv], allProcFirstLeafStates[(postRecvIdx)*2].nbParts, postRecvIdx,
244 FMpi::TagExchangeIndexs, communicator, &requests);
245 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] recv " << allProcFirstLeafStates[(postRecvIdx)*2].nbParts << " from " << postRecvIdx << "\n"; FLog::Controller.flush(); );
246 // Inc the write position
247 postPositionRecv += allProcFirstLeafStates[(postRecvIdx)*2].nbParts;
248 }
249 }
250 hasExtendLastLeaf = true;
251 }
252 }
253
254 // Finalize communication
255 FMpi::MpiAssert(MPI_Waitall(int(requests.size()), requests.data(), MPI_STATUSES_IGNORE),__LINE__);
256
257 // IF we sent we need to remove the first leaf
258 if(hasSentFirstLeaf){
259 const FSize offsetParticles = borderLeavesState[0].nbParts;
260 // Move all the particles
261 for(FSize idxPart = offsetParticles ; idxPart < (*workingSize) ; ++idxPart){
262 (*workingArray)[idxPart - offsetParticles] = (*workingArray)[idxPart];
263 }
264 // Move all the leaf
265 for(int idxLeaf = 1 ; idxLeaf < leavesInfo.getSize() ; ++idxLeaf){
266 leavesInfo[idxLeaf].startingPoint -= offsetParticles;
267 leavesInfo[idxLeaf - 1] = leavesInfo[idxLeaf];
268 }
269 (*workingSize) -= offsetParticles;
270 }
271
272 // If we received we need to merge both arrays
273 if(hasExtendLastLeaf){
274 // Allocate array
275 const FSize finalParticlesNumber = (*workingSize) + receivedParticles.size();
276 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Create array "
277 << finalParticlesNumber << " particles\n"; FLog::Controller.flush(); );
278 IndexedParticle* particlesWithExtension = new IndexedParticle[finalParticlesNumber];
279 // Copy old data
280 memcpy(particlesWithExtension, (*workingArray), (*workingSize)*sizeof(IndexedParticle));
281 // Copy received data
282 memcpy(particlesWithExtension + (*workingSize), receivedParticles.data(), receivedParticles.size()*sizeof(IndexedParticle));
283 // Move ptr
284 delete[] (*workingArray);
285 (*workingArray) = particlesWithExtension;
286 (*workingSize) = finalParticlesNumber;
287 leavesInfo[leavesInfo.getSize()-1].nbParts += receivedParticles.size();
288 }
289 }
290 {//Filling the Array with leaves and parts //// COULD BE MOVED IN AN OTHER FUCTION
291
292 (*leavesSize) = 0; //init ptr
293 (*particlesArrayInLeafOrder) = nullptr; //init ptr
294 (*leavesOffsetInParticles) = nullptr; //init ptr
295
296 if((*workingSize)){
297 //Copy all the particles
298 (*particlesArrayInLeafOrder) = new ParticleClass[(*workingSize)];
299 for(FSize idxPart = 0 ; idxPart < (*workingSize) ; ++idxPart){
300 memcpy(&(*particlesArrayInLeafOrder)[idxPart],&(*workingArray)[idxPart].particle,sizeof(ParticleClass));
301 }
302 // Assign the number of leaf
303 (*leavesSize) = leavesInfo.getSize();
304 // Store the offset position for each leaf
305 (*leavesOffsetInParticles) = new FSize[leavesInfo.getSize() + 1];
306 for(int idxLeaf = 0 ; idxLeaf < leavesInfo.getSize() ; ++idxLeaf){
307 (*leavesOffsetInParticles)[idxLeaf] = leavesInfo[idxLeaf].startingPoint;
308 }
309 (*leavesOffsetInParticles)[leavesInfo.getSize()] = (*workingSize);
310 }
311 }
312 }

320 template <class ContainerClass>
321 static void EqualizeAndFillContainer(const FMpi::FComm& communicator, ContainerClass* particlesSaver,
322 const FSize leavesOffsetInParticles[], const ParticleClass particlesArrayInLeafOrder[],
323 const FSize currentNbLeaves,
324 const FSize currentNbParts, FAbstractBalanceAlgorithm * balancer){
325 const int myRank = communicator.processId();
326 const int nbProcs = communicator.processCount();
327
328 if(nbProcs == 1){
329 //Just copy each part into the Particle Saver
330 for(FSize idxPart =0 ; idxPart < currentNbParts ; ++idxPart){
331 particlesSaver->push(particlesArrayInLeafOrder[idxPart]);
332 }
333 }
334 else{
335 // We need to know the number of leaves per procs
336 std::unique_ptr<FSize[]> numberOfLeavesPerProc(new FSize[nbProcs]);
337 FMpi::MpiAssert(MPI_Allgather(const_cast<FSize*>(&currentNbLeaves), 1, MPI_LONG_LONG_INT, numberOfLeavesPerProc.get(),
338 1, MPI_LONG_LONG_INT, communicator.getComm()), __LINE__);
339
340
341 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Exchange number of leaves\n"; FLog::Controller.flush(); );
342
343 // prefix sum
344 std::unique_ptr<FSize[]> diffNumberOfLeavesPerProc(new FSize[nbProcs+1]);
345 diffNumberOfLeavesPerProc[0] = 0;
346 for(int idxProc = 0 ; idxProc < nbProcs ; ++idxProc ){
347 diffNumberOfLeavesPerProc[idxProc+1] = diffNumberOfLeavesPerProc[idxProc] + numberOfLeavesPerProc[idxProc];
348 }
349
350 const FSize totalNumberOfLeavesInSimulation = diffNumberOfLeavesPerProc[nbProcs];
351 // Compute the objective interval
352 std::vector< std::pair<size_t,size_t> > allObjectives;
353 allObjectives.resize(nbProcs);
354 for(int idxProc = 0 ; idxProc < nbProcs ; ++idxProc){
355 allObjectives[idxProc].first = balancer->getLeft(totalNumberOfLeavesInSimulation,nbProcs,idxProc);
356 allObjectives[idxProc].second = balancer->getRight(totalNumberOfLeavesInSimulation,nbProcs,idxProc);
357 if(idxProc != 0) FAssertLF(allObjectives[idxProc].first == allObjectives[idxProc-1].second);
358 }
359
360 // Ask for the pack to send
361 std::pair<size_t, size_t> myCurrentInter = {diffNumberOfLeavesPerProc[myRank], diffNumberOfLeavesPerProc[myRank+1]};
362 const std::vector<FEqualize::Package> packsToSend = FEqualize::GetPackToSend(myCurrentInter, allObjectives);
363
364 FAssertLF((currentNbLeaves == 0 && packsToSend.size() == 0) ||
365 (currentNbLeaves != 0 && packsToSend.size() && FSize(packsToSend[packsToSend.size()-1].elementTo) == currentNbLeaves));
366
367 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Previous currentNbLeaves (" << currentNbLeaves << ")\n"; FLog::Controller.flush(); );
368 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Get my interval (" << packsToSend.size() << ")\n"; FLog::Controller.flush(); );
369 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Send data\n"; FLog::Controller.flush(); );
370
371 // Store the requests
372 std::vector<MPI_Request> requestsNbParts;
373 requestsNbParts.reserve(packsToSend.size());
374
375 // Send every thing except for me or if size == 0
376 FSize totalSend = 0;
377 FSize sendToMe = 0;
378 for(unsigned int idxPack = 0; idxPack< packsToSend.size() ; ++idxPack){
379 const FEqualize::Package& pack = packsToSend[idxPack];
380
381 if(idxPack != 0) FAssertLF(packsToSend[idxPack].elementFrom == packsToSend[idxPack-1].elementTo);
382 FAssertLF(FSize(pack.elementTo) <= FSize(currentNbParts));
383 FAssertLF(pack.elementFrom <= pack.elementTo);
384 const long long int nbPartsPerPackToSend = leavesOffsetInParticles[pack.elementTo]-leavesOffsetInParticles[pack.elementFrom];
385 totalSend += nbPartsPerPackToSend;
386
387 if(pack.idProc != myRank && 0 < (pack.elementTo-pack.elementFrom)){
388 // If not to me and if there is something to send
389 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] pre-send to " << pack.idProc << " nb " << nbPartsPerPackToSend
390 << " from " << pack.elementFrom << " to " << pack.elementTo << " \n"; FLog::Controller.flush(); );
391 // Send the size of the data
392 requestsNbParts.emplace_back();
393 FMpi::MpiAssert(MPI_Isend((void*)&nbPartsPerPackToSend,1,MPI_LONG_LONG_INT,pack.idProc,
394 FMpi::TagExchangeIndexs, communicator.getComm(), &requestsNbParts.back()),__LINE__);
395
396 }
397 else {
398 sendToMe = nbPartsPerPackToSend;
399 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] skip " << idxPack
400 << " from " << pack.elementFrom << " to " << pack.elementTo << " \n"; FLog::Controller.flush(); );
401 }
402 }
403 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Send done \n"; FLog::Controller.flush(); );
404 // Ensure everything has been proceed
405 FAssertLF(totalSend == currentNbParts);
406
407 // Compute the current intervals
408 std::vector< std::pair<size_t,size_t> > allCurrentIntervals;
409 allCurrentIntervals.resize(nbProcs);
410 for(int idxProc = 0 ; idxProc < nbProcs ; ++idxProc){
411 allCurrentIntervals[idxProc].first = diffNumberOfLeavesPerProc[idxProc];
412 allCurrentIntervals[idxProc].second = diffNumberOfLeavesPerProc[idxProc+1];
413 }
414 // Ask the packs to receive to fill my objective
415 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Get my receive interval \n"; FLog::Controller.flush(); );
416 std::pair<size_t, size_t> myObjective = allObjectives[myRank];
417 const std::vector<FEqualize::Package> packsToRecv = FEqualize::GetPackToRecv(myObjective, allCurrentIntervals);
418
419 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] recv nb particles \n"; FLog::Controller.flush(); );
420 // Count the number of parts to receive
421 std::unique_ptr<FSize[]> nbPartsPerPackToRecv(new FSize[packsToRecv.size()]);
422 for(unsigned int idxPack = 0; idxPack < packsToRecv.size(); ++idxPack){
423 const FEqualize::Package& pack = packsToRecv[idxPack];
424
425 if(idxPack != 0) FAssertLF(packsToRecv[idxPack].elementFrom == packsToRecv[idxPack-1].elementTo);
426
427 if(pack.idProc != myRank && 0 < (pack.elementTo-pack.elementFrom)){
428 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] pre-recv from " << pack.idProc << " \n"; FLog::Controller.flush(); );
429 // We need to know how much particles to receive
430 requestsNbParts.emplace_back();
431 FMpi::MpiAssert(MPI_Irecv(&nbPartsPerPackToRecv[idxPack], 1, MPI_LONG_LONG_INT, pack.idProc,
432 FMpi::TagExchangeIndexs, communicator.getComm(), &requestsNbParts.back()), __LINE__);
433 }
434 else{
435 if(pack.idProc == myRank){
436 // Take my own data
437 const FSize sourcePosition = FMath::Max(myObjective.first, myCurrentInter.first) - myCurrentInter.first;
438 const FSize nbLeavesToCopy = pack.elementTo-pack.elementFrom;
439 nbPartsPerPackToRecv[idxPack] = leavesOffsetInParticles[sourcePosition+nbLeavesToCopy] - leavesOffsetInParticles[sourcePosition];
440 FAssertLF(nbPartsPerPackToRecv[idxPack] == sendToMe);
441 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] skip recv " <<
442 idxPack << " nb " << nbPartsPerPackToRecv[idxPack] << " \n"; FLog::Controller.flush(); );
443 }
444 else{
445 // Nothing to receive from this so avoid communication
446 nbPartsPerPackToRecv[idxPack] = 0;
447 }
448 }
449 }
450
451 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Wait \n"; FLog::Controller.flush(); );
452
453 FMpi::MpiAssert(MPI_Waitall(int(requestsNbParts.size()), requestsNbParts.data(), MPI_STATUSES_IGNORE), __LINE__);
454
455 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] Wait Done \n"; FLog::Controller.flush(); );
456
457 std::vector<MPI_Request> requestsParts;
458
459 for(unsigned int idxPack = 0; idxPack< packsToSend.size() ; ++idxPack){
460 const FEqualize::Package& pack = packsToSend[idxPack];
461 if(pack.idProc != myRank && 0 < (pack.elementTo-pack.elementFrom)){
462 const long long int nbPartsPerPackToSend = leavesOffsetInParticles[pack.elementTo]-leavesOffsetInParticles[pack.elementFrom];
463
464 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] send to "
465 << pack.idProc << " nb " << nbPartsPerPackToSend << " \n"; FLog::Controller.flush(); );
466
467 FMpi::ISendSplit(&particlesArrayInLeafOrder[leavesOffsetInParticles[pack.elementFrom]],
468 nbPartsPerPackToSend,
469 pack.idProc,
470 FMpi::TagExchangeIndexs + 1,
471 communicator,
472 &requestsParts);
473 }
474 }
475
476
477 FLOG(if(VerboseLog) FLog::Controller << "SCALFMM-DEBUG [" << communicator.processId() << "] barrier after all send \n"; FLog::Controller.flush(); );
478
480 // Count the number of leaf to receive
482 FSize totalPartsToReceive = 0;
483 for(unsigned int idxPack = 0; idxPack < packsToRecv.size(); ++idxPack){
484 totalPartsToReceive += nbPartsPerPackToRecv[idxPack];
485 }
486
487 std::unique_ptr<ParticleClass[]> particlesRecvBuffer(new ParticleClass[totalPartsToReceive]);
488
490 // Post all the receive and copy mine
491 // it is based on the nbPartsPerPackToRecv array
493 if(totalPartsToReceive){
494 FSize offsetToRecv = 0;
495 for(unsigned int idxPack = 0; idxPack < packsToRecv.size(); ++idxPack){
496 const FEqualize::Package& pack = packsToRecv[idxPack];
497 // If it is not from me

535 particlesSaver->push(particlesRecvBuffer[idPartsToStore]);
536 }
