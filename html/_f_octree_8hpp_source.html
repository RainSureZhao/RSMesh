474 bool moveRight(){
475 // Function variables
476 SubOctreeTypes workingTree = this->current; // To cover parent other sub octre
477 int workingLevel = this->currentLocalLevel; // To know where we are
478 int workingIndex = this->currentLocalIndex; // To know where we are
479
480 // -- First we have to go in a tree where we can move on the right --
481 // Number of time we go into parent subtree
482 int countUpward = 0;
483 // We stop when we can right move or if there is no more parent (root)
484 while( workingIndex == TransposeIndex(workingTree.tree->getRightLeafIndex(), (workingTree.tree->getSubOctreeHeight() - workingLevel - 1) )
485 && workingTree.tree->hasParent() ){
486 // Goto the leaf level into parent at current_tree.position_into_parent_array
487 workingIndex = workingTree.tree->getIndexInParent();
488 workingTree.tree = workingTree.tree->getParent();
489 workingLevel = workingTree.tree->getSubOctreeHeight() - 1;
490 // inc counter
491 ++countUpward;
492 }
493
494 // Do we stop because we are on the root (means we cannot move right?)
495 if( workingIndex < TransposeIndex(workingTree.tree->getRightLeafIndex(), (workingTree.tree->getSubOctreeHeight() - workingLevel - 1) ) ){
496 // Move to the first right cell pointer(!)
497 ++workingIndex;
498
499 // Maybe it is null, but we know there is almost one cell on the right
500 // we need to find it
501 if( !workingTree.tree->cellsAt(workingLevel)[workingIndex] ){
502 // While we are not on a allocated cell
503 while( true ){
504 // Test element on the right (test brothers)
505 const int rightLimite = (workingIndex | 7) + 1;
506 while( workingIndex < rightLimite && !workingTree.tree->cellsAt(workingLevel)[workingIndex]){
507 ++workingIndex;
508 }
509 // Stop if we are on an allocated cell
510 if( workingTree.tree->cellsAt(workingLevel)[workingIndex] ){
511 break;
512 }
513 // Else go to the upper level
514 --workingLevel;
515 workingIndex >>= 3;
516 }
517 }
518
519 // if wokring tree != current tree => working tree leafs level ; else current level
520 const int objectiveLevel = (countUpward ? workingTree.tree->getSubOctreeHeight() - 1 : this->currentLocalLevel );
521
522 // We need to go down as left as possible
523 while( workingLevel != objectiveLevel ){
524 ++workingLevel;
525 workingIndex <<= 3;
526 const int rightLimite = (workingIndex | 7); // not + 1 because if the 7th first are null it must be the 8th!
527 while( workingIndex < rightLimite && !workingTree.tree->cellsAt(workingLevel)[workingIndex]){
528 ++workingIndex;
529 }
530 }
531
532 // Do we change from the current sub octree?
533 if( countUpward ){
534 // Then we simply need to go down the same number of time
535 workingTree.tree = workingTree.middleTree->leafs(workingIndex);
536 while( --countUpward ){
537 workingTree.tree = workingTree.middleTree->leafs(workingTree.tree->getLeftLeafIndex());
538 }
539 // level did not change, simpli set octree and get left limit of this octree at the current level
540 this->current = workingTree;
541 this->currentLocalIndex = TransposeIndex(workingTree.tree->getLeftLeafIndex(), (workingTree.tree->getSubOctreeHeight() - this->currentLocalLevel - 1) );
542 }
543 else{
544 // We are on the right tree
545 this->currentLocalIndex = workingIndex;
546 }
547
548 return true;
549 }
550 return false;
551 }

587 // We are on the leaf of the current suboctree?

589 // Yes change sub octree

591 this->currentLocalIndex = 0;

595 else{
596 ++this->currentLocalLevel;

598 }
599 // Find the first allocated cell from left to right
600 while(!this->current.tree->cellsAt(this->currentLocalLevel)[this->currentLocalIndex]){
601 ++this->currentLocalIndex;
602 }
603 return true;
604 }
605 return false;

739 SubOctreeTypesConst workingTree;
740 workingTree.tree = this->root;
741 const MortonIndex treeSubLeafMask = ~(~0ULL << (3 * workingTree.tree->getSubOctreeHeight() ));
742
743 // Find the suboctree a the correct level
744 while(inLevel >= workingTree.tree->getSubOctreeHeight() + workingTree.tree->getSubOctreePosition()) {
745 // compute the leaf index
746 const MortonIndex fullIndex = inIndex >> (3 * (inLevel + 1 - (workingTree.tree->getSubOctreeHeight() + workingTree.tree->getSubOctreePosition()) ) );
747 // point to next suboctree
748 workingTree.tree = workingTree.middleTree->leafs(treeSubLeafMask & fullIndex);

754 return workingTree.tree->cellsAt(inLevel - workingTree.tree->getSubOctreePosition())[treeLeafMask & inIndex];

851 const FTreeCoordinate& workingCell,
852 const int inLevel, const int neighSeparation = 1) const{
853 // reset
854 memset(inNeighbors, 0, sizeof(CellClass*) * 343);
855
856 // Then take each child of the parent's neighbors if not in directNeighbors
857 // Father coordinate
858 const FTreeCoordinate parentCell(workingCell.getX()>>1,workingCell.getY()>>1,workingCell.getZ()>>1);
859
860 // Limite at parent level number of box (split by 2 by level)
861 const int boxLimite = FMath::pow2(inLevel-1);
862
863 int idxNeighbors = 0;
864 // We test all cells around
865 for(int idxX = -1 ; idxX <= 1 ; ++idxX){
866 if(!FMath::Between(parentCell.getX() + idxX,0,boxLimite)) continue;
867
868 for(int idxY = -1 ; idxY <= 1 ; ++idxY){
869 if(!FMath::Between(parentCell.getY() + idxY,0,boxLimite)) continue;
870
871 for(int idxZ = -1 ; idxZ <= 1 ; ++idxZ){
872 if(!FMath::Between(parentCell.getZ() + idxZ,0,boxLimite)) continue;
873
874 // if we are not on the current cell
875 if( neighSeparation<1 || idxX || idxY || idxZ ){
876 const FTreeCoordinate otherParent(parentCell.getX() + idxX,parentCell.getY() + idxY,parentCell.getZ() + idxZ);
877 const MortonIndex mortonOtherParent = otherParent.getMortonIndex() << 3;
878 // Get child
879 CellClass** const cells = getCellPt(mortonOtherParent, inLevel);
880
881 // If there is one or more child
882 if(cells){
883 // For each child
884 for(int idxCousin = 0 ; idxCousin < 8 ; ++idxCousin){
885 if(cells[idxCousin]){
886 const int xdiff = ((otherParent.getX()<<1) | ( (idxCousin>>2) & 1)) - workingCell.getX();
887 const int ydiff = ((otherParent.getY()<<1) | ( (idxCousin>>1) & 1)) - workingCell.getY();
888 const int zdiff = ((otherParent.getZ()<<1) | (idxCousin&1)) - workingCell.getZ();
889
890 // Test if it is a direct neighbor
891 if(FMath::Abs(xdiff) > neighSeparation || FMath::Abs(ydiff) > neighSeparation || FMath::Abs(zdiff) > neighSeparation){
892 // add to neighbors
893 inNeighbors[ (((xdiff+3) * 7) + (ydiff+3)) * 7 + zdiff + 3] = cells[idxCousin];
894 ++idxNeighbors;
895 }
896 }
897 }
898 }
899 }
900 }
901 }
902 }
