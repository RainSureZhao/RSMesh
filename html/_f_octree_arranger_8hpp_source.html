59 void rearrange(){
60 {
61 typename OctreeClass::Iterator octreeIterator(tree);
62 octreeIterator.gotoBottomLeft();
63 do{
64 const MortonIndex currentMortonIndex = octreeIterator.getCurrentGlobalIndex();
65 //First we test sources
66 ContainerClass * particles = octreeIterator.getCurrentLeaf()->getSrc();
67 for(FSize idxPart = 0 ; idxPart < particles->getNbParticles(); /*++idxPart*/){
68 FPoint<FReal> currentPart;
69 interface->getParticlePosition(particles,idxPart,&currentPart);
70 checkPosition(currentPart);
71 const MortonIndex particuleIndex = tree->getMortonFromPosition(currentPart);
72 if(particuleIndex != currentMortonIndex){
73 //Need to move this one
74 interface->removeFromLeafAndKeep(particles,currentPart,idxPart,FParticleType::FParticleTypeSource);
75 }
76 else{
77 //Need to increment idx;
78 ++idxPart;
79 }
80 }
81 //Then we test targets
82 if(octreeIterator.getCurrentLeaf()->getTargets() != particles){ //Leaf is TypedLeaf
83 ContainerClass * particleTargets = octreeIterator.getCurrentLeaf()->getTargets();
84 for(FSize idxPart = 0 ; idxPart < particleTargets->getNbParticles(); /*++idxPart*/){
85 FPoint<FReal> currentPart;
86 interface->getParticlePosition(particleTargets,idxPart,&currentPart);
87 checkPosition(currentPart);
88 const MortonIndex particuleIndex = tree->getMortonFromPosition(currentPart);
89 if(particuleIndex != currentMortonIndex){
90 //Need to move this one
91 interface->removeFromLeafAndKeep(particleTargets,currentPart,idxPart, FParticleType::FParticleTypeTarget);
92 }
93 else{
94 //Need to increment idx;
95 ++idxPart;
96 }
97 }
98 }
99 }while(octreeIterator.moveRight());
100 }
101 printf("Insert back particles\n");
102 //Insert back the parts that have been removed
103 interface->insertAllParticles(tree);
104
105 //Then, remove the empty leaves
106 { // Remove empty leaves
107 typename OctreeClass::Iterator octreeIterator(tree);
108 octreeIterator.gotoBottomLeft();
109 bool workOnNext = true;
110 do{
111 // Empty leaf
112 if( octreeIterator.getCurrentListTargets()->getNbParticles() == 0 &&
113 octreeIterator.getCurrentListSrc()->getNbParticles() == 0 ){
114 const MortonIndex currentIndex = octreeIterator.getCurrentGlobalIndex();
115 workOnNext = octreeIterator.moveRight();
116 tree->removeLeaf( currentIndex );
117 }
118 // Not empty, just continue
119 else {
120 workOnNext = octreeIterator.moveRight();
121 }
122 } while( workOnNext );
123 }
