36 int getInterval(const MortonIndex mindex, const int size, const Interval intervals[]) const{
37 for(int idxProc = 0 ; idxProc < size ; ++idxProc){
38 // does it contains the index?
39 if( intervals[idxProc].min <= mindex && mindex < intervals[idxProc].max){
40 return idxProc;
41 }
42 }
43 // if no interval found return the lastest one
44 return size - 1;

57 bool rearrange(const FMpi::FComm& comm, const int isPeriodic = DirNone){
58 // interval of each procs
59 Interval*const intervals = new Interval[comm.processCount()];
60 memset(intervals, 0, sizeof(Interval) * comm.processCount());
61
62 { // We need to exchange interval of each process, this interval
63 // will be based on the current morton min max
64 Interval myLastInterval;
65
66 // take fist index
67 typename OctreeClass::Iterator octreeIterator(tree);
68 octreeIterator.gotoBottomLeft();
69 myLastInterval.min = octreeIterator.getCurrentGlobalIndex();
70 // take last index
71 octreeIterator.gotoRight();
72 myLastInterval.max = octreeIterator.getCurrentGlobalIndex();
73
74 // We get the min/max indexes from each procs
75 FMpi::MpiAssert( MPI_Allgather( &myLastInterval, sizeof(Interval), MPI_BYTE, intervals, sizeof(Interval), MPI_BYTE, comm.getComm()), __LINE__ );
76
77 // increase interval in the empty morton index
78 intervals[0].min = 0;
79 for(int idxProc = 1 ; idxProc < comm.processCount() ; ++idxProc){
80 intervals[idxProc].min = ((intervals[idxProc].min - intervals[idxProc-1].max)/2) + intervals[idxProc-1].max;
81 intervals[idxProc-1].max = intervals[idxProc].min;
82 }
83
84 intervals[comm.processCount() - 1].max = ((1 << (3*(tree->getHeight()-1))) - 1);
85 }
86
87 // Particles that move
88 FVector<ParticleClass>*const toMove = new FVector<ParticleClass>[comm.processCount()];
89
90 { // iterate on the leafs and found particle to remove or to send
91 // For periodic
92 const FReal boxWidth = tree->getBoxWidth();
93 const FPoint<FReal> min(tree->getBoxCenter(),-boxWidth/2);
94 const FPoint<FReal> max(tree->getBoxCenter(),boxWidth/2);
95
96 FVector<FSize> indexesToExtract;
97
98 typename OctreeClass::Iterator octreeIterator(tree);
99 octreeIterator.gotoBottomLeft();
100 do{
101 const MortonIndex currentIndex = octreeIterator.getCurrentGlobalIndex();
102 ContainerClass* particles = octreeIterator.getCurrentLeaf()->getSrc();
103 //IdxPart is incremented at the end of the loop
104 for(FSize idxPart = 0 ; idxPart < particles->getNbParticles(); /*++idxPart*/){
105 FPoint<FReal> partPos( particles->getPositions()[0][idxPart],
106 particles->getPositions()[1][idxPart],
107 particles->getPositions()[2][idxPart] );
108 // XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX
109 if( TestPeriodicCondition(isPeriodic, DirPlusX) ){
110 while(partPos.getX() >= max.getX()){
111 partPos.incX(-boxWidth);
112 }
113 }
114 else if(partPos.getX() >= max.getX()){
115 printf("Error, particle out of Box in +X, index %lld\n", currentIndex);
116 printf("Application is exiting...\n");
117 }
118 if( TestPeriodicCondition(isPeriodic, DirMinusX) ){
119 while(partPos.getX() < min.getX()){
120 partPos.incX(boxWidth);
121 }
122 }
123 else if(partPos.getX() < min.getX()){
124 printf("Error, particle out of Box in -X, index %lld\n", currentIndex);
125 printf("Application is exiting...\n");
126 }
127 // YYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYYY
128 if( TestPeriodicCondition(isPeriodic, DirPlusY) ){
129 while(partPos.getY() >= max.getY()){
130 partPos.incY(-boxWidth);
131 }
132 }
133 else if(partPos.getY() >= max.getY()){
134 printf("Error, particle out of Box in +Y, index %lld\n", currentIndex);
135 printf("Application is exiting...\n");
136 }
137 if( TestPeriodicCondition(isPeriodic, DirMinusY) ){
138 while(partPos.getY() < min.getY()){
139 partPos.incY(boxWidth);
140 }
141 }
142 else if(partPos.getY() < min.getY()){
143 printf("Error, particle out of Box in -Y, index %lld\n", currentIndex);
144 printf("Application is exiting...\n");
145 }
146 // ZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZZ
147 if( TestPeriodicCondition(isPeriodic, DirPlusX) ){
148 while(partPos.getZ() >= max.getZ()){
149 partPos.incZ(-boxWidth);
150 }
151 }
152 else if(partPos.getZ() >= max.getZ()){
153 printf("Error, particle out of Box in +Z, index %lld\n", currentIndex);
154 printf("Application is exiting...\n");
155 }
156 if( TestPeriodicCondition(isPeriodic, DirMinusX) ){
157 while(partPos.getZ() < min.getZ()){
158 partPos.incZ(boxWidth);
159 }
160 }
161 else if(partPos.getZ() < min.getZ()){
162 printf("Error, particle out of Box in -Z, index %lld\n", currentIndex);
163 printf("Application is exiting...\n");
164 }
165 // set pos
166 particles->getWPositions()[0][idxPart] = partPos.getX();
167 particles->getWPositions()[1][idxPart] = partPos.getY();
168 particles->getWPositions()[2][idxPart] = partPos.getZ();
169
170 const MortonIndex particuleIndex = tree->getMortonFromPosition(partPos);
171 // is this particle need to be changed from its leaf
172 if(particuleIndex != currentIndex){
173 // find the right interval
174 const int procConcerned = getInterval( particuleIndex, comm.processCount(), intervals);
175 toMove[procConcerned].push(ConverterClass::GetParticleAndRemove(particles,idxPart));
176 indexesToExtract.push(idxPart);
177 //No need to increment idxPart, since the array has been staggered
178 }
179 else{
180 idxPart++;
181 }
182 }
183
184 particles->removeParticles(indexesToExtract.data(), indexesToExtract.getSize());
185 indexesToExtract.clear();
186
187 } while(octreeIterator.moveRight());
188 }

190 // To send and recv
191 ParticleClass* toReceive = nullptr;
192 MPI_Request*const requests = new MPI_Request[comm.processCount()*2];
193 memset(requests, 0, sizeof(MPI_Request) * comm.processCount() * 2);
194 FSize*const indexToReceive = new FSize[comm.processCount() + 1];
195 memset(indexToReceive, 0, sizeof(FSize) * comm.processCount() + 1);

197 int iterRequests = 0;
198 int limitRecvSend = 0;
199 int hasToRecvFrom = 0;
200
201 { // gather what to send to who + isend data
202 FSize*const counter = new FSize[comm.processCount()];
203 memset(counter, 0, sizeof(FSize) * comm.processCount());
204
205 for(int idxProc = 0 ; idxProc < comm.processCount() ; ++idxProc){
206 counter[idxProc] = toMove[idxProc].getSize();
207 }
208 // say who send to who
209 FSize*const allcounter = new FSize[comm.processCount()*comm.processCount()];
210 FMpi::MpiAssert( MPI_Allgather( counter, comm.processCount(), FMpi::GetType(*counter), allcounter, comm.processCount(),
211 FMpi::GetType(*counter), comm.getComm()), __LINE__ );
212
213 // prepare buffer to receive
214 FSize sumToRecv = 0;
215 indexToReceive[0] = 0;
216 for(int idxProc = 0 ; idxProc < comm.processCount() ; ++idxProc){
217 if( idxProc != comm.processId()){
218 sumToRecv += allcounter[idxProc * comm.processCount() + comm.processId()];
219 }
220 indexToReceive[idxProc + 1] = sumToRecv;
221 }
222 toReceive = new ParticleClass[sumToRecv];
223
224 // send
225 for(int idxProc = 0 ; idxProc < comm.processCount() ; ++idxProc){
226 if(idxProc != comm.processId() && allcounter[idxProc * comm.processCount() + comm.processId()]){
227 FAssertLF( allcounter[idxProc * comm.processCount() + comm.processId()] * sizeof(ParticleClass) < std::numeric_limits<int>::max());
228 FMpi::MpiAssert( MPI_Irecv(&toReceive[indexToReceive[idxProc]], int(allcounter[idxProc * comm.processCount() + comm.processId()] * sizeof(ParticleClass)), MPI_BYTE,
229 idxProc, 0, comm.getComm(), &requests[iterRequests++]), __LINE__ );
230 hasToRecvFrom += 1;
231 }
232 }
233
234 limitRecvSend = iterRequests;
235
236 // recv
237 for(int idxProc = 0 ; idxProc < comm.processCount() ; ++idxProc){
238 if(idxProc != comm.processId() && toMove[idxProc].getSize()){
239 FAssertLF( toMove[idxProc].getSize() * sizeof(ParticleClass) < std::numeric_limits<int>::max());
240 FMpi::MpiAssert( MPI_Isend(toMove[idxProc].data(), int(toMove[idxProc].getSize() * sizeof(ParticleClass)), MPI_BYTE,
241 idxProc, 0, comm.getComm(), &requests[iterRequests++]), __LINE__ );
242 }
243 }
244
245 delete[] allcounter;
246 delete[] counter;
247 }
248
249 { // insert particles that moved
250 for(FSize idxPart = 0 ; idxPart < toMove[comm.processId()].getSize() ; ++idxPart){
251 ConverterClass::Insert( tree , toMove[comm.processId()][idxPart]);
252 }
253 }
254
255 { // wait any recv or send
256 // if it is a recv then insert particles
257 MPI_Status status;
258 while( hasToRecvFrom ){
259 int done = 0;
260 FMpi::MpiAssert( MPI_Waitany( iterRequests, requests, &done, &status ), __LINE__ );
261 if( done < limitRecvSend ){
262 const int source = status.MPI_SOURCE;
263 for(FSize idxPart = indexToReceive[source] ; idxPart < indexToReceive[source+1] ; ++idxPart){
264 ConverterClass::Insert( tree , toReceive[idxPart]);
265 }
266 hasToRecvFrom -= 1;
267 }
268 }
269 }
270
271 int counterLeavesAlive = 0;
272 { // Remove empty leaves
273 typename OctreeClass::Iterator octreeIterator(tree);
274 octreeIterator.gotoBottomLeft();
275 bool workOnNext = true;
276
277 do{
278 // Empty leaf
279 if( octreeIterator.getCurrentListTargets()->getNbParticles() == 0 ){
280 const MortonIndex currentIndex = octreeIterator.getCurrentGlobalIndex();
281 workOnNext = octreeIterator.moveRight();
282 tree->removeLeaf( currentIndex );
283 }
284 // Not empty, just continue
285 else {
286 workOnNext = octreeIterator.moveRight();
287 counterLeavesAlive += 1;
288 }
289 } while( workOnNext );
290 }
291
292 // wait all send
293 FMpi::MpiAssert( MPI_Waitall( iterRequests, requests, MPI_STATUSES_IGNORE), __LINE__ );
294
295 delete[] intervals;
296 delete[] toMove;
297 delete[] requests;
298 delete[] toReceive;
299 delete[] indexToReceive;
300
301 // return false if tree is empty
302 return counterLeavesAlive != 0;
303 }
