37inline void MutualParticlesKIJ(const FReal targetX,const FReal targetY,const FReal targetZ, const FReal* targetPhysicalValue,
38 FReal* targetForceX, FReal* targetForceY, FReal* targetForceZ, FReal* targetPotential,
39 const FReal sourceX,const FReal sourceY,const FReal sourceZ, const FReal* sourcePhysicalValue,
40 FReal* sourceForceX, FReal* sourceForceY, FReal* sourceForceZ, FReal* sourcePotential,
41 const MatrixKernelClass *const MatrixKernel){
42
43 // get information on tensorial aspect of matrix kernel
44 const int ncmp = MatrixKernelClass::NCMP;
45 const int npv = MatrixKernelClass::NPV;
46 const int npot = MatrixKernelClass::NPOT;
47
48 // evaluate kernel and its partial derivatives
49 const FPoint<FReal> sourcePoint(sourceX,sourceY,sourceZ);
50 const FPoint<FReal> targetPoint(targetX,targetY,targetZ);
51 FReal Kxy[ncmp];
52 FReal dKxy[ncmp][3];
53 MatrixKernel->evaluateBlockAndDerivative(targetPoint,sourcePoint,Kxy,dKxy);
54 const FReal mutual_coeff = MatrixKernel->getMutualCoefficient(); // 1 if symmetric; -1 if antisymmetric
55
56 for(unsigned int i = 0 ; i < npot ; ++i){
57 for(unsigned int j = 0 ; j < npv ; ++j){
58
59 // update component to be applied
60 const int d = MatrixKernel->getPosition(i*npv+j);
61
62 // forces prefactor
63 const FReal coef = (targetPhysicalValue[j] * sourcePhysicalValue[j]);
64
65 targetForceX[i] += dKxy[d][0] * coef;
66 targetForceY[i] += dKxy[d][1] * coef;
67 targetForceZ[i] += dKxy[d][2] * coef;
68 targetPotential[i] += ( Kxy[d] * sourcePhysicalValue[j] );
69
70 sourceForceX[i] -= dKxy[d][0] * coef;
71 sourceForceY[i] -= dKxy[d][1] * coef;
72 sourceForceZ[i] -= dKxy[d][2] * coef;
73 sourcePotential[i] += ( mutual_coeff * Kxy[d] * targetPhysicalValue[j] );
74
75 }// j
76 }// i
