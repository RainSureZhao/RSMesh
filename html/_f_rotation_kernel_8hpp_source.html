107 void precomputeTranslationCoef(){
108 {// M2M & L2L
109 // Allocate
110 M2MTranslationCoef = new FReal[treeHeight-1][P+1];
111 L2LTranslationCoef = new FReal[treeHeight-1][P+1];
112 // widthAtLevel represents half of the size of a box
113 FReal widthAtLevel = boxWidth/4;
114 // we go from the root to the leaf-1
115 for( int idxLevel = 0 ; idxLevel < treeHeight - 1 ; ++idxLevel){
116 // b is the parent-child distance = norm( vec(widthAtLevel,widthAtLevel,widthAtLevel))
117 const FReal b = FMath::Sqrt(widthAtLevel*widthAtLevel*3);
118 // we compute b^idx iteratively
119 FReal bPowIdx = 1.0;
120 // we compute -1^idx iteratively
121 FReal minus_1_pow_idx = 1.0;
122 for(int idx = 0 ; idx <= P ; ++idx){
123 // coef m2m = (-b)^j/j!
124 M2MTranslationCoef[idxLevel][idx] = minus_1_pow_idx * bPowIdx / factorials[idx];
125 // coef l2l = b^j/j!
126 L2LTranslationCoef[idxLevel][idx] = bPowIdx / factorials[idx];
127 // increase
128 bPowIdx *= b;
129 minus_1_pow_idx = -minus_1_pow_idx;
130 }
131 // divide by two per level
132 widthAtLevel /= 2;
133 }
134 }
135 {// M2L
136 // Allocate
137 M2LTranslationCoef = new FReal[treeHeight][343][P+1];
138 // This is the width of a box at each level
139 FReal boxWidthAtLevel = widthAtLeafLevel;
140 // from leaf level to the root
141 for(int idxLevel = treeHeight-1 ; idxLevel > 0 ; --idxLevel){
142 // we compute all possibilities
143 for(int idxX = -3 ; idxX <= 3 ; ++idxX ){
144 for(int idxY = -3 ; idxY <= 3 ; ++idxY ){
145 for(int idxZ = -3 ; idxZ <= 3 ; ++idxZ ){

614 dlmk[l][P+m][P+k] = -dlmkZPositif[l][P+m][P+k];

618 dlmk[l][P+m][P+l] = -dlmkZPositif[l][P+m][P+l];
619 }

621 // else we start per (+1)
622 for(int k = -l ; k < l ; k+=2){

624 dlmk[l][P+m][P+k+1] = -dlmkZPositif[l][P+m][P+k+1];
625 }
626 // l is always odd
627 dlmk[l][P+m][P+l] = dlmkZPositif[l][P+m][P+l];

641 // theta should be between [0;pi] as the inclinaison angle
642 void DlmkBuild(FReal dlmk[P+1][P2+1][P2+1], const FReal inTheta) const {
643 FAssertLF(0 <= inTheta && inTheta < FMath::FTwoPi<FReal>());
644 // To have constants for very used values
645 const FReal F0 = FReal(0.0);
646 const FReal F1 = FReal(1.0);
647 const FReal F2 = FReal(2.0);
648
649 const FReal cosTheta = FMath::Cos(inTheta);
650 const FReal sinTheta = FMath::Sin(inTheta);
651
652 // First compute g
653 FReal g[SizeArray];
654 {// Equ 29
655 // g{0,0} = 1
656 g[0] = F1;
657
658 // g{l,0} = sqrt( (2l - 1) / 2l) g{l-1,0} for l > 0
659 {
660 int index_l0 = 1;
661 FReal fl = F1;
662 for(int l = 1; l <= P ; ++l, ++fl ){
663 g[index_l0] = FMath::Sqrt((fl*F2-F1)/(fl*F2)) * g[index_l0-l];
664 index_l0 += l + 1;
665 }
666 }
667 // g{l,m} = sqrt( (l - m + 1) / (l+m)) g{l,m-1} for l > 0, 0 < m <= l
668 {
669 int index_lm = 2;
670 FReal fl = F1;
671 for(int l = 1; l <= P ; ++l, ++fl ){
672 FReal fm = F1;
673 for(int m = 1; m <= l ; ++m, ++index_lm, ++fm ){
674 g[index_lm] = FMath::Sqrt((fl-fm+F1)/(fl+fm)) * g[index_lm-1];
675 }
676 ++index_lm;
677 }
678 }
679 }
680 { // initial condition
681 // Equ 28
682 // d{l,m,l} = -1^(l+m) g{l,m} (1+cos(theta))^m sin(theta)^(l-m) , For l > 0, 0 <= m <= l
683 int index_lm = 0;
684 FReal sinTheta_pow_l = F1;
685 for(int l = 0 ; l <= P ; ++l){
686 // build variable iteratively
687 FReal minus_1_pow_lm = l&0x1 ? FReal(-1) : FReal(1);
688 FReal cosTheta_1_pow_m = F1;
689 FReal sinTheta_pow_l_minus_m = sinTheta_pow_l;
690 for(int m = 0 ; m <= l ; ++m, ++index_lm){
691 dlmk[l][P+m][P+l] = minus_1_pow_lm * g[index_lm] * cosTheta_1_pow_m * sinTheta_pow_l_minus_m;
692 // update
693 minus_1_pow_lm = -minus_1_pow_lm;
694 cosTheta_1_pow_m *= F1 + cosTheta;
695 sinTheta_pow_l_minus_m /= sinTheta;
696 }
697 // update
698 sinTheta_pow_l *= sinTheta;
699 }
700 }
701 { // build the rest of the matrix
702 FReal fl = F1;
703 for(int l = 1 ; l <= P ; ++l, ++fl){
704 FReal fk = fl;
705 for(int k = l ; k > -l ; --k, --fk){
706 // Equ 25
707 // For l > 0, 0 <= m < l, -l < k <= l, cos(theta) >= 0
708 // d{l,m,k-1} = sqrt( l(l+1) - m(m+1) / l(l+1) - k(k-1)) d{l,m+1,k}
709 // + (m+k) sin(theta) d{l,m,k} / sqrt(l(l+1) - k(k-1)) (1+cos(theta))
710 FReal fm = F0;
711 for(int m = 0 ; m < l ; ++m, ++fm){
712 dlmk[l][P+m][P+k-1] =
713 (FMath::Sqrt((fl*(fl+F1)-fm*(fm+F1))/(fl*(fl+F1)-fk*(fk-F1))) * dlmk[l][P+m+1][P+k])
714 + ((fm+fk)*sinTheta*dlmk[l][P+m][P+k]/(FMath::Sqrt(fl*(fl+F1)-fk*(fk-F1))*(F1+cosTheta)));
715 }
716 // Equ 26
717 // For l > 0, -l < k <= l, cos(theta) >= 0
718 // d{l,l,k-1} = (l+k) sin(theta) d{l,l,k}
719 // / sqrt(l(l+1)-k(k-1)) (1+cos(theta))
720 dlmk[l][P+l][P+k-1] = (fl+fk)*sinTheta*dlmk[l][P+l][P+k]/(FMath::Sqrt(fl*(fl+F1)-fk*(fk-F1))*(F1+cosTheta));
721 }
722 // Equ 27
723 // d{l,m,k} = -1^(m+k) d{l,-m,-k} , For l > 0, -l <= m < 0, -l <= k <= l
724 for(int m = -l ; m < 0 ; ++m){
725 FReal minus_1_pow_mk = (m-l)&0x1 ? FReal(-1) : FReal(1);
726 for(int k = -l ; k <= l ; ++k){
727 dlmk[l][P+m][P+k] = minus_1_pow_mk * dlmk[l][P-m][P-k];
728 minus_1_pow_mk = -minus_1_pow_mk;
729 }
730 }
731 }
732 }

748 void computeLegendre(FReal legendre[], const FReal inCosTheta, const FReal inSinTheta) const {
749 const FReal invSinTheta = -inSinTheta;
750
751 legendre[0] = 1.0; // P_0,0(1) = 1
752
753 legendre[1] = inCosTheta; // P_1,0 = cos(theta)
754 legendre[2] = invSinTheta; // P_1,1 = -sin(theta)
755
756 // work with pointers
757 FReal* FRestrict legendre_l1_m1 = legendre; // P{l-2,m} starts with P_{0,0}
758 FReal* FRestrict legendre_l1_m = legendre + 1; // P{l-1,m} starts with P_{1,0}
759 FReal* FRestrict legendre_lm = legendre + 3; // P{l,m} starts with P_{2,0}
760
761 // Compute using recurrence
762 FReal l2_minus_1 = 3; // 2 * l - 1
763 FReal fl = FReal(2.0);// To get 'l' as a float
764 for(int l = 2; l <= P ; ++l, ++fl ){
765 FReal lm_minus_1 = fl - FReal(1.0); // l + m - 1
766 FReal l_minus_m = fl; // l - m
767 for( int m = 0; m < l - 1 ; ++m ){
768 // P_{l,m} = \frac{(2l-1) cos( \theta ) P_{l-1,m} - (l+m-1) P_{l-2,m}x}{(l-m)}
769 *(legendre_lm++) = (l2_minus_1 * inCosTheta * (*legendre_l1_m++) - (lm_minus_1++) * (*legendre_l1_m1++) )
770 / (l_minus_m--);
771 }
772 // P_{l,l-1} = (2l-1) cos( \theta ) P_{l-1,l-1}
773 *(legendre_lm++) = l2_minus_1 * inCosTheta * (*legendre_l1_m);
774 // P_{l,l} = (2l-1) sin( \theta ) P_{l-1,l-1}
775 *(legendre_lm++) = l2_minus_1 * invSinTheta * (*legendre_l1_m);
776 // goto P_{l-1,0}
777 ++legendre_l1_m;
778 l2_minus_1 += FReal(2.0); // 2 * l - 1 => progress by two
779 }

899 const FReal i_pow_m[4] = {0, FMath::FPiDiv2<FReal>(), FMath::FPi<FReal>(), -FMath::FPiDiv2<FReal>()};
900 // w is the multipole moment
901 FComplex<FReal>* FRestrict const w = inPole->getMultipole();
902
903 // Copying the position is faster than using cell position
904 const FPoint<FReal> cellPosition = getLeafCenter(inPole->getCoordinate());
905
906 // We need a legendre array
907 FReal legendre[SizeArray];
908 FReal angles[P+1][2];
909
910 // For all particles in the leaf box
911 const FReal*const physicalValues = inParticles->getPhysicalValues();
912 const FReal*const positionsX = inParticles->getPositions()[0];
913 const FReal*const positionsY = inParticles->getPositions()[1];
914 const FReal*const positionsZ = inParticles->getPositions()[2];
915
916 for(FSize idxPart = 0 ; idxPart < inParticles->getNbParticles() ; ++ idxPart){
917 // P2M
918 const FPoint<FReal> position(positionsX[idxPart],positionsY[idxPart],positionsZ[idxPart]);
919 const FSpherical<FReal> sph(position - cellPosition);
920
921 // The physical value (charge, mass)
922 const FReal q = physicalValues[idxPart];
923 // The distance between the SH and the particle
924 const FReal a = sph.getR();
925
926 // Compute the legendre polynomial
927 computeLegendre(legendre, sph.getCosTheta(), sph.getSinTheta());
928
929 // w{l,m}(q,a) = q a^l/(l+|m|)! P{l,m}(cos(alpha)) exp(-i m Beta)
930 FReal q_aPowL = q; // To consutrct q*a^l continously
931 int index_l_m = 0; // To construct the index of (l,m) continously
932 FReal fl = 0.0;
933 for(int l = 0 ; l <= P ; ++l, ++fl ){
934 { // We need to compute the angles to use in the "m" loop
935 // So we can compute only the one needed after "l" inc
936 const FReal angle = fl * sph.getPhi() + i_pow_m[l & 0x3];
937 angles[l][0] = FMath::Cos(angle);
938 angles[l][1] = FMath::Sin(angle);
939 }
940 for(int m = 0 ; m <= l ; ++m, ++index_l_m){
941 const FReal magnitude = q_aPowL * legendre[index_l_m] / factorials[l+m];
942 w[index_l_m].incReal(magnitude * angles[m][0]);
943 w[index_l_m].incImag(magnitude * angles[m][1]);
944 }
945 q_aPowL *= a;
946 }

1216 FReal minus_r_pow_l_legendre_div_fact_lm[SizeArray];

1271 }

1280 }

1286 Fp /= sph.getR() * sph.getSinTheta();

1290 const FReal sinPhi = FMath::Sin(sph.getPhi());

1296 cosPhi * sph.getCosTheta() * FO +
1297 (-sinPhi) * Fp) * physicalValue;
1298

1300 sinPhi * sph.getSinTheta() * Fr +
1301 sinPhi * sph.getCosTheta() * FO +
1302 cosPhi * Fp) * physicalValue;
1303
1304 const FReal forceZ = (
1305 sph.getCosTheta() * Fr +
1306 (-sph.getSinTheta()) * FO) * physicalValue;
1307
1308 // inc particles forces
1309 forcesX[idxPart] += forceX;
1310 forcesY[idxPart] += forceY;
1311 forcesZ[idxPart] += forceZ;
1312 }

1317 int index_lm = 0;

1326 const FReal I_real = coef * cos_m_phi_i_pow_m[m];

1329 }
1330 }
1331 // inc potential
1332 potentials[idxPart] += magnitude;
1333 }
1334 }
