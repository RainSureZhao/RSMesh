104 FReal d_lmk_analytical(const FReal cosTheta, const FReal sinTheta, const int l , const int m, const int k){
105 if( l >= 0 && -l <= k && k <= l && FMath::Abs(k) <= m && m <= l ){
106 FReal sum = 0;
107 for(int n = FMath::Max(-(m+k),0) ; n <= FMath::Min(l-m,l-k) ; ++n){
108 sum += FMath::pow(FReal(-1.0), FMath::Abs(l-m-n)) * combin(l-k, n) * combin(l+k, l-m-n) * FMath::pow(FReal(1.0)+cosTheta,FMath::Abs(n)) * FMath::pow(FReal(1.0)-cosTheta, FMath::Abs(l-m-n));
109 }
110 return (FReal(1.0)/FMath::pow(FReal(2.0),FMath::Abs(l))) * FMath::Sqrt((fact(l-m)*fact(l+m))/(fact(l-k)*fact(l+k)))
111 * FMath::pow(FReal(1.0) + FReal(Sgn(k))*cosTheta, FMath::Abs(k)) * FMath::pow(sinTheta, FMath::Abs(m - FMath::Abs(k))) * sum;
112 }
113 // P^{l}_{m,k} = (-1)^{m+k} P^{l}_{k,m} , l \ge 0, -l \leq m \le 0, |m| \leq k \leq l
114 else if( (l > 0 && -l <= m && m < 0 && FMath::Abs(m) <= k && k <= l)
115 || (l > 0 && 0 <= m && m < l && m < k && k <= l )) {
116 return FMath::pow(FReal(-1.0), FMath::Abs(m+k)) * d_lmk_analytical(cosTheta,sinTheta, l, k ,m);
117 }
118 // P^{l}_{m,k} = (-1)^{m+k} P^{l}_{k,m} , l \ge 0, 0 \leq m \le l, m \le k \leq l
119 else if( l > 0 && -l <= m && m < l && -l <= k && k < -m ){
120 return FMath::pow(FReal(-1.0), FMath::Abs(m+k)) * d_lmk_analytical(cosTheta,sinTheta, l, -m, -k);
121 }
122 else{
123 printf("Error that should not be possible!\n");
124 return FReal(0.0);
125 }

146 void computeLegendre(FReal legendre[], const FReal inCosTheta, const FReal inSinTheta){
147 const FReal invSinTheta = -inSinTheta;
148 legendre[atLm(0,0)] = 1.0; // P_0,0(1) = 1
149
150 legendre[atLm(1,0)] = inCosTheta; // P_1,0 = cos(theta)
151 legendre[atLm(1,1)] = invSinTheta; // P_1,1 = -sin(theta)
152
153 // Compute using recurrence
154 for(int l = 2; l <= P ; ++l ){
155 for( int m = 0; m < l - 1 ; ++m ){
156 // P_{l,m} = \frac{(2l-1) cos( \theta ) P_{l-1,m} - (l+m-1) P_{l-2,m}x}{(l-k)}
157 legendre[atLm(l,m)] = (FReal(2*l-1) * inCosTheta * legendre[atLm(l-1,m)] - FReal( l + m - 1 ) * legendre[atLm(l-2,m)] )
158 / FReal( l - m );
159 }
160 // P_{l,l-1} = (2l-1) cos( \theta ) P_{l-1,l-1}
161 legendre[atLm(l,l-1)] = FReal(2*l-1) * inCosTheta * legendre[atLm(l-1,l-1)];
162 // P_{l,l} = (2l-1) sin( \theta ) P_{l-1,l-1}
163 legendre[atLm(l,l)] = FReal(2*l-1) * invSinTheta * legendre[atLm(l-1,l-1)];
164 }

227 void rotateMultipoleAroundY(FComplex<FReal> vec[], const FReal theta){
228 FComplex<FReal> cell_rotate[SizeArray];
229 for(int l = 0 ; l <= P ; ++l){
230 for(int m = 0 ; m <= l ; ++m ){
231 FReal w_lkm_real = 0.0;
232 FReal w_lkm_imag = 0.0;
233
234 for(int k = -l ; k < 0 ; ++k){
235 // O_{l,-m} = \bar{ O_{l,m} } (-1)^m
236 const FReal d_lmk = d_lmk_analytical(FMath::Cos(theta),FMath::Sin(theta),l,m,k);
237 // \sqrt{ \frac{(l-k)!(l+k)!}{(l-|m|)!(l+|m|)!} }
238 const FReal factor = FMath::Sqrt((fact(l-k)*fact(l+k))/(fact(l-abs(m))*fact(l+abs(m))));
239 w_lkm_real += FMath::pow(FReal(-1.0),-k) * factor * d_lmk * vec[atLm(l,-k)].getReal(); // k<0 => Conjugate * -1^k
240 w_lkm_imag -= FMath::pow(FReal(-1.0),-k) * factor * d_lmk * vec[atLm(l,-k)].getImag(); // k<0 => Conjugate * -1^k
241 }
242 for(int k = 0 ; k <= l ; ++k){
243 const FReal d_lmk = d_lmk_analytical(FMath::Cos(theta),FMath::Sin(theta),l,m,k);
244 // \sqrt{ \frac{(l-k)!(l+k)!}{(l-|m|)!(l+|m|)!} }
245 const FReal factor = FMath::Sqrt((fact(l-k)*fact(l+k))/(fact(l-abs(m))*fact(l+abs(m))));
246 w_lkm_real += factor * d_lmk * vec[atLm(l,k)].getReal();
247 w_lkm_imag += factor * d_lmk * vec[atLm(l,k)].getImag();
248 }
249 cell_rotate[atLm(l,m)].setRealImag(w_lkm_real, w_lkm_imag);
250 }

346 void preComputePosition(){
347 // Compute the parent/children relation for M2M L2L
348 childrenPosition = new FSpherical<FReal>[8 * (treeHeight-1)];
349 {
350 FReal subBoxWidth = widthAtLeafLevelDiv2;
351 // For each
352 for(int idxLevel = treeHeight-2 ; idxLevel > 0 ; --idxLevel){
353 for(int idxChild = 0 ; idxChild < 8 ; ++idxChild ){
354 // coord from child to parent
355 const FReal x = FReal((idxChild&4)? -subBoxWidth : subBoxWidth);
356 const FReal y = FReal((idxChild&2)? -subBoxWidth : subBoxWidth);
357 const FReal z = FReal((idxChild&1)? -subBoxWidth : subBoxWidth);
358 const FPoint<FReal> relativePosition( x , y , z );
359
360 childrenPosition[(idxLevel-1)*8 + idxChild] = FSpherical<FReal>(relativePosition);
361 }
362 subBoxWidth *= FReal(2.0);
363 }
364 }
365 // Compute the interaction relations for M2L
366 interactionsPosition = new FSpherical<FReal>[343 * (treeHeight-1)];
367 {
368 FReal boxWidthAtLevel = widthAtLeafLevel;
369 for(int idxLevel = treeHeight-1 ; idxLevel > 0 ; --idxLevel){
370 for(int idxX = -3 ; idxX <= 3 ; ++idxX ){
371 for(int idxY = -3 ; idxY <= 3 ; ++idxY ){
372 for(int idxZ = -3 ; idxZ <= 3 ; ++idxZ ){
373 if( idxX != 0 || idxY != 0 || idxZ != 0 ){
374 const FPoint<FReal> relativePosition( -FReal(idxX)*boxWidthAtLevel,
375 -FReal(idxY)*boxWidthAtLevel,
376 -FReal(idxZ)*boxWidthAtLevel);
377 const int position = ((( (idxX+3) * 7) + (idxY+3))) * 7 + idxZ + 3;
378 interactionsPosition[(idxLevel-1)*343 + position] = FSpherical<FReal>(relativePosition);
379 }
380 }
381 }
382 }
383 boxWidthAtLevel *= FReal(2.0);
384 }

432 void P2M(CellClass* const inPole, const ContainerClass* const inParticles )override {
433 const FReal i_pow_m[4] = {0, FMath::FPiDiv2<FReal>(), FMath::FPi<FReal>(), -FMath::FPiDiv2<FReal>()};
434 // w is the multipole moment
435 FComplex<FReal>* FRestrict const w = inPole->getMultipole();
436
437 // Copying the position is faster than using cell position
438 const FPoint<FReal> cellPosition = getLeafCenter(inPole->getCoordinate());
439
440 // We need a legendre array
441 FReal legendre[SizeArray];
442
443 // For all particles in the leaf box
444 const FReal*const physicalValues = inParticles->getPhysicalValues();
445 const FReal*const positionsX = inParticles->getPositions()[0];
446 const FReal*const positionsY = inParticles->getPositions()[1];
447 const FReal*const positionsZ = inParticles->getPositions()[2];
448
449 for(FSize idxPart = 0 ; idxPart < inParticles->getNbParticles() ; ++ idxPart){
450 // P2M
451 const FPoint<FReal> position(positionsX[idxPart],positionsY[idxPart],positionsZ[idxPart]);
452 const FSpherical<FReal> sph(position - cellPosition);
453
454 // The physical value (charge, mass)
455 const FReal q = physicalValues[idxPart];
456 // The distance between the SH and the particle
457 const FReal a = sph.getR();
458
459 // Compute the legendre polynomial
460 computeLegendre(legendre, sph.getCosTheta(), sph.getSinTheta());
461
462 // w{l,m}(q,a) = q a^l/(l+|m|)! P{l,m}(cos(alpha)) exp(-i m Beta)
463 for(int l = 0 ; l <= P ; ++l ){
464 for(int m = 0 ; m <= l ; ++m ){
465 const FReal magnitude = q * (FMath::pow( a , l )/fact(l+m))
466 * legendre[atLm(l,m)];
467 w[atLm(l,m)].incReal(magnitude * FMath::Cos(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]));
468 w[atLm(l,m)].incImag(magnitude * FMath::Sin(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]));
469 }
470 }
471 }

485 void M2M(CellClass* const FRestrict inPole, const CellClass*const FRestrict *const FRestrict inChildren, const int inLevel)override {
486 // A buffer to copy the source w allocated once
487 FComplex<FReal> source_w[SizeArray];
488 // For all children
489 for(int idxChild = 0 ; idxChild < 8 ; ++idxChild){
490 // if child exists
491 if(inChildren[idxChild]){
492 // Copy the source
493 FMemUtils::copyall(source_w, inChildren[idxChild]->getMultipole(), SizeArray);
494
495 // rotate it forward
496 const FSpherical<FReal> sph = getSphericalChild(inLevel, idxChild);
497 rotateMultipole(source_w, sph.getPhiZero2Pi(), sph.getInclination());
498
499 const FReal b = -sph.getR();
500 // Translate it
501 FComplex<FReal> target_w[SizeArray];
502 for(int l = 0 ; l <= P ; ++l ){
503 for(int m = 0 ; m <= l ; ++m ){
504 // w{l,m}(a+b) = sum(j=m:l, b^(l-j)/(l-j)! w{j,m}(a)
505 FReal w_lm_real = 0.0;
506 FReal w_lm_imag = 0.0;
507 for(int j = m ; j <= l ; ++j ){
508 const FReal coef = FMath::pow(b,l-j) / fact(l-j);
509 w_lm_real += coef * source_w[atLm(j,m)].getReal();
510 w_lm_imag += coef * source_w[atLm(j,m)].getImag();
511 }
512 target_w[atLm(l,m)].setRealImag(w_lm_real,w_lm_imag);
513 }
514 }
515
516 // Rotate it back
517 deRotateMultipole(target_w, sph.getPhiZero2Pi(), sph.getInclination());
518
519 // Sum the result
520 FMemUtils::addall( inPole->getMultipole(), target_w, SizeArray);
521 }
522 }

537 const int neighborPositions[], const int inSize, const int inLevel) override {
538 // To copy the multipole data allocated once
539 FComplex<FReal> source_w[SizeArray];
540 // For all children
541 for(int idxExistingNeigh = 0 ; idxExistingNeigh < inSize ; ++idxExistingNeigh){
542 const int idxNeigh = neighborPositions[idxExistingNeigh];
543 // Copy multipole data into buffer
544 FMemUtils::copyall(source_w, inInteractions[idxExistingNeigh]->getMultipole(), SizeArray);
545
546 // Rotate
547 const FSpherical<FReal> sph = getSphericalInteraction(inLevel, idxNeigh);
548 rotateMultipole(source_w, sph.getPhiZero2Pi(), sph.getInclination());
549
550 const FReal b = sph.getR();
551 // Transfer to u
552 FComplex<FReal> target_u[SizeArray];
553 for(int l = 0 ; l <= P ; ++l ){
554 for(int m = 0 ; m <= l ; ++m ){
555 // u{l,m}(a-b) = sum(j=|m|:P-l, (j+l)!/b^(j+l+1) w{j,-m}(a)
556 FReal u_lm_real = 0.0;
557 FReal u_lm_imag = 0.0;
558 for(int j = m ; j <= P-l ; ++j ){
559 const FReal coef = FMath::pow(FReal(-1.0),m) * ( fact(j+l) / FMath::pow(b,j+l+1));
560 // because {l,-m} => {l,m} conjugate -1^m with -i
561 u_lm_real += coef * source_w[atLm(j,m)].getReal();
562 u_lm_imag -= coef * source_w[atLm(j,m)].getImag();
563 }
564 target_u[atLm(l,m)].setRealImag(u_lm_real,u_lm_imag);
565 }
566 }
567
568 // Rotate it back
569 deRotateTaylor(target_u, sph.getPhiZero2Pi(), sph.getInclination());
570
571 // Sum
572 FMemUtils::addall(inLocal->getLocal(), target_u, SizeArray);
573 }

674 {
675 FReal Fr = 0;
676 FReal FO = 0;
677 FReal Fp = 0;
678
679 for(int l = 1 ; l <= P ; ++l){
680 {
681 const FReal coef = FMath::pow(FReal(-1.0),l) * FMath::pow(r,l) * legendre[atLm(l,0)] / fact(l+0);
682 const FReal I_real = coef * FMath::Cos(FReal(0) * sph.getPhi() + i_pow_m[0]);
683 const FReal I_imag = coef * FMath::Sin(FReal(0) * sph.getPhi() + i_pow_m[0]);
684
685 Fr += FReal(l) * (u[atLm(l,0)].getReal() * I_real - u[atLm(l,0)].getImag() * I_imag);
686 }
687 {
688 const FReal coef = (FMath::pow(FReal(-1.0),l)/fact(l+0)) * FMath::pow(r,l) * ((FReal(l) * sph.getCosTheta()*legendre[atLm(l,0)]
689 - FReal(l) * legendre[atLm(l-1,0)]) / sph.getSinTheta());
690 const FReal dI_real = coef * FMath::Cos(FReal(0) * sph.getPhi() + i_pow_m[0 & 0x3]);
691 const FReal dI_imag = coef * FMath::Sin(FReal(0) * sph.getPhi() + i_pow_m[0 & 0x3]);
692 // F(O) += 2 * Real(L dI/dO)
693 FO += u[atLm(l,0)].getReal() * dI_real - u[atLm(l,0)].getImag() * dI_imag;
694 }
695
696 for(int m = 1 ; m <= l ; ++m){
697 {
698 const FReal coef = FMath::pow(FReal(-1.0),l) * FMath::pow(r,l) * legendre[atLm(l,m)] / fact(l+m);
699 const FReal I_real = coef * FMath::Cos(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
700 const FReal I_imag = coef * FMath::Sin(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
701 // F(r) += 2 x l x Real(LI)
702 Fr += 2 * FReal(l) * (u[atLm(l,m)].getReal() * I_real - u[atLm(l,m)].getImag() * I_imag);
703 // F(p) += -2 x m x Imag(LI)
704 Fp -= 2 * FReal(m) * (u[atLm(l,m)].getReal() * I_imag + u[atLm(l,m)].getImag() * I_real);
705 }
706 {
707 const FReal legendre_l_minus_1 = (m == l) ? 0 : FReal(l+m)*legendre[atLm(l-1,m)];
708 const FReal coef = (FMath::pow(FReal(-1.0),l)/fact(l+m)) * FMath::pow(r,l) * ((FReal(l) * sph.getCosTheta()*legendre[atLm(l,m)]
709 - legendre_l_minus_1) / sph.getSinTheta());
710 const FReal dI_real = coef * FMath::Cos(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
711 const FReal dI_imag = coef * FMath::Sin(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
712 // F(O) += 2 * Real(L dI/dO)
713 FO += FReal(2.0) * (u[atLm(l,m)].getReal() * dI_real - u[atLm(l,m)].getImag() * dI_imag);
714 }
715 }
716 }
717 // div by r
718 Fr /= sph.getR();
719 FO /= sph.getR();
720 Fp /= sph.getR() * sph.getSinTheta();
721
722 // copy variable from spherical position
723 const FReal cosPhi = FMath::Cos(sph.getPhi());
724 const FReal sinPhi = FMath::Sin(sph.getPhi());
725 const FReal physicalValue = physicalValues[idxPart];
726
727 // compute forces
728 const FReal forceX = (
729 cosPhi * sph.getSinTheta() * Fr +
730 cosPhi * sph.getCosTheta() * FO +
731 (-sinPhi) * Fp) * physicalValue;
732
733 const FReal forceY = (
734 sinPhi * sph.getSinTheta() * Fr +
735 sinPhi * sph.getCosTheta() * FO +
736 cosPhi * Fp) * physicalValue;
737
738 const FReal forceZ = (
739 sph.getCosTheta() * Fr +
740 (-sph.getSinTheta()) * FO) * physicalValue;
741
742 // inc particles forces
743 forcesX[idxPart] += forceX;
744 forcesY[idxPart] += forceY;
745 forcesZ[idxPart] += forceZ;
746 }
747
748 { // Result for potential
749 FReal magnitude = 0;
750 // E = sum( l = 0:P, sum(m = -l:l, u{l,m} ))
751 for(int l = 0 ; l <= P ; ++l ){
752 {
753 // (l-|m|)! * P{l,0} / r^(l+1)
754 const FReal coef = FMath::pow(FReal(-1.0),l) * FMath::pow(r,l) * legendre[atLm(l,0)] / fact(l+0);
755 const FReal I_real = coef * FMath::Cos(FReal(0) * sph.getPhi() + i_pow_m[0]);
756 const FReal I_imag = coef * FMath::Sin(FReal(0) * sph.getPhi() + i_pow_m[0]);
757 magnitude += u[atLm(l,0)].getReal() * I_real - u[atLm(l,0)].getImag() * I_imag;
758 }
759 for(int m = 1 ; m <= l ; ++m ){
760 const FReal coef = FMath::pow(FReal(-1.0),l) * FMath::pow(r,l) * legendre[atLm(l,m)] / fact(l+m);
761 const FReal I_real = coef * FMath::Cos(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
762 const FReal I_imag = coef * FMath::Sin(FReal(m) * sph.getPhi() + i_pow_m[m & 0x3]);
763 magnitude += FReal(2.0) * ( u[atLm(l,m)].getReal() * I_real - u[atLm(l,m)].getImag() * I_imag );
764 }
765 }
766 // inc potential
767 potentials[idxPart] += magnitude;
768 }
769 }
770 }
