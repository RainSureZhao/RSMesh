48 void allocAndInit(){
49 FHarmonic<FReal> blasHarmonic(Parent::devP * 2);
50
51 // Matrix to fill and then transposed
52 FComplex<FReal>*const workMatrix = new FComplex<FReal>[FF_MATRIX_SIZE];
53
54 // M2L transfer, there is a maximum of 3 neighbors in each direction,
55 // so 6 in each dimension
56 FReal treeWidthAtLevel = Parent::boxWidth;
57 preM2LTransitions = new FComplex<FReal>**[Parent::treeHeight];
58 memset(preM2LTransitions.getPtr(), 0, sizeof(FComplex<FReal>**) * (Parent::treeHeight));
59
60 for(int idxLevel = 0 ; idxLevel < Parent::treeHeight ; ++idxLevel ){
61 preM2LTransitions[idxLevel] = new FComplex<FReal>*[(7 * 7 * 7)];
62 memset(preM2LTransitions[idxLevel], 0, sizeof(FComplex<FReal>*) * (7*7*7));
63
64 for(int idxX = -3 ; idxX <= 3 ; ++idxX ){
65 for(int idxY = -3 ; idxY <= 3 ; ++idxY ){
66 for(int idxZ = -3 ; idxZ <= 3 ; ++idxZ ){
67 if(FMath::Abs(idxX) > 1 || FMath::Abs(idxY) > 1 || FMath::Abs(idxZ) > 1){
68 // Compute harmonic
69 const FPoint<FReal> relativePos( FReal(-idxX) * treeWidthAtLevel , FReal(-idxY) * treeWidthAtLevel , FReal(-idxZ) * treeWidthAtLevel );
70 blasHarmonic.computeOuter(FSpherical<FReal>(relativePos));
71
72 // Reset Matrix
73 FMemUtils::setall<FComplex<FReal>>(workMatrix, FComplex<FReal>(), FF_MATRIX_SIZE);
74 FComplex<FReal>* FRestrict fillTransfer = workMatrix;
75
76 for(int M = 0 ; M <= Parent::devP ; ++M){
77 for (int m = 0 ; m <= M ; ++m){
78 for (int N = 0 ; N <= Parent::devP ; ++N){
79 for (int n = 0 ; n <= 2*N ; ++n, ++fillTransfer){
80 const int k = N-n-m;
81 if (k < 0){
82 const FReal pow_of_minus_1 = FReal((k&1) ? -1 : 1);
83 fillTransfer->setReal( pow_of_minus_1 * blasHarmonic.result()[blasHarmonic.getPreExpRedirJ(M+N)-k].getReal());
84 fillTransfer->setImag((-pow_of_minus_1) * blasHarmonic.result()[blasHarmonic.getPreExpRedirJ(M+N)-k].getImag());
85 }
86 else{
87 (*fillTransfer) = blasHarmonic.result()[blasHarmonic.getPreExpRedirJ(M+N)+k];
88 }
89
90 }
91 }
92 }
93 }
94
95 // Transpose and copy result
96 FComplex<FReal>*const matrix = new FComplex<FReal>[FF_MATRIX_SIZE];
97 for(int idxRow = 0 ; idxRow < FF_MATRIX_ROW_DIM ; ++idxRow){
98 for(int idxCol = 0 ; idxCol < FF_MATRIX_COLUMN_DIM ; ++idxCol){
99 matrix[idxCol * FF_MATRIX_ROW_DIM + idxRow] = workMatrix[idxCol + idxRow * FF_MATRIX_COLUMN_DIM];
100 }
101 }
102 preM2LTransitions[idxLevel][indexM2LTransition(idxX,idxY,idxZ)] = matrix;
103 }
104 }
105 }
106 }
107 treeWidthAtLevel /= 2;
108 }
109
110 // Clean
111 delete[] workMatrix;
112 }

192 void preExpNExp(FComplex<FReal>* const exp) const {
193 for(int j = Parent::devP; j>= 0 ; --j){
194 // Position in 'exp': (j*(j+1)*0.5) + k
195 // Position in 'nexp': j*(j+1) + k
196 const int j_j1 = j*(j+1);
197 const int j_j1_div_2 = int(j_j1 * 0.5);
198
199 // Positive (or null) orders:
200 for(int k = j ; k >= 0; --k){
201 exp[j_j1 + k] = exp[j_j1_div_2 + k];
202 }
203
204 // Negative orders:
205 FReal minus_1_pow_k = FReal( j&1 ? -1 : 1);
206 for(int k = -j ; k < 0 ; ++k ){
207 exp[j_j1 + k].setReal(minus_1_pow_k * exp[j_j1 + (-k)].getReal());
208 exp[j_j1 + k].setImag((-minus_1_pow_k) * exp[j_j1 + (-k)].getImag());
209 minus_1_pow_k = -minus_1_pow_k;
210 }
211 }

242 void multipoleToLocal(const int interactionIndex, const int inLevel){
