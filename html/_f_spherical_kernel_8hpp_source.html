27 void allocAndInit(){
28 // M2L transfer, there is a maximum of 3 neighbors in each direction,
29 // so 6 in each dimension
30 preM2LTransitions = new FComplex<FReal>*[Parent::treeHeight];
31 memset(preM2LTransitions.getPtr(), 0, sizeof(FComplex<FReal>*) * (Parent::treeHeight));
32 // We start from the higher level
33 FReal treeWidthAtLevel = Parent::boxWidth;
34 for(int idxLevel = 0 ; idxLevel < Parent::treeHeight ; ++idxLevel ){
35 // Allocate data for this level
36 preM2LTransitions[idxLevel] = new FComplex<FReal>[(7 * 7 * 7) * devM2lP];
37 // Precompute transfer vector
38 for(int idxX = -3 ; idxX <= 3 ; ++idxX ){
39 for(int idxY = -3 ; idxY <= 3 ; ++idxY ){
40 for(int idxZ = -3 ; idxZ <= 3 ; ++idxZ ){
41 if(FMath::Abs(idxX) > 1 || FMath::Abs(idxY) > 1 || FMath::Abs(idxZ) > 1){
42 const FPoint<FReal> relativePos( FReal(-idxX) * treeWidthAtLevel , FReal(-idxY) * treeWidthAtLevel , FReal(-idxZ) * treeWidthAtLevel );
43 Parent::harmonic.computeOuter(FSpherical<FReal>(relativePos));
44 FMemUtils::copyall<FComplex<FReal>>(&preM2LTransitions[idxLevel][indexM2LTransition(idxX,idxY,idxZ)], Parent::harmonic.result(), Parent::harmonic.getExpSize());
45 }
46 }
47 }
48 }
49 // We divide the bow per 2 when we go down
50 treeWidthAtLevel /= 2;
51 }

113 void multipoleToLocal(FComplex<FReal>*const FRestrict local_exp, const FComplex<FReal>* const FRestrict multipole_exp_src,
114 const FComplex<FReal>* const FRestrict M2L_Outer_transfer){
115 int index_j_k = 0;
116
117 // L_j^k
118 // HPMSTART(51, "M2L computation (loops)");
119 // j from 0 to P
120 for (int j = 0 ; j <= Parent::devP ; ++j){
121 // (-1)^k
122 FReal pow_of_minus_1_for_k = 1.0;
123 //k from 0 to j
124 for (int k = 0 ; k <= j ; ++k, ++index_j_k){
125 // (-1)^n
126 FReal pow_of_minus_1_for_n = 1.0;
127
128 // work with a local variable
129 FComplex<FReal> L_j_k = local_exp[index_j_k];
130 // n from 0 to P or do P-j
131 //for (int n = 0 ; n <= Parent::devP /*or*/ /*Parent::devP-j*/ ; ++n){
132 for (int n = 0 ; n <= Parent::devP-j ; ++n){ // faster than double height Parent::devP
133 // O_n^l : here points on the source multipole expansion term of degree n and order |l|
134 const int index_n = Parent::harmonic.getPreExpRedirJ(n);
135
136 // Outer_{j+n}^{-k-l} : here points on the M2L transfer function/expansion term of degree j+n and order |-k-l|
137 const int index_n_j = Parent::harmonic.getPreExpRedirJ(n+j);
138
139 FReal pow_of_minus_1_for_l = pow_of_minus_1_for_n; // (-1)^l
140
141 // We start with l=n (and not l=-n) so that we always set p_Outer_term to a correct value in the first loop.
142 int l = n;
143 for(/* l = n */ ; l > 0 ; --l){ // we have -k-l<0 and l>0
144 const FComplex<FReal> M_n_l = multipole_exp_src[index_n + l];
145 const FComplex<FReal> O_n_j__k_l = M2L_Outer_transfer[index_n_j + k + l];
146
147 L_j_k.incReal( pow_of_minus_1_for_l * pow_of_minus_1_for_k *
148 ((M_n_l.getReal() * O_n_j__k_l.getReal()) +
149 (M_n_l.getImag() * O_n_j__k_l.getImag())));
150 L_j_k.incImag( pow_of_minus_1_for_l * pow_of_minus_1_for_k *
151 ((M_n_l.getImag() * O_n_j__k_l.getReal()) -
152 (M_n_l.getReal() * O_n_j__k_l.getImag())));
153
154 pow_of_minus_1_for_l = -pow_of_minus_1_for_l;
155 }
156
157 for(/* l = 0 */; l >= -n && (-k-l) < 0 ; --l){ // we have -k-l<0 and l<=0
158 const FComplex<FReal> M_n_l = multipole_exp_src[index_n - l];
159 const FComplex<FReal> O_n_j__k_l = M2L_Outer_transfer[index_n_j + k + l];
160
161 L_j_k.incReal( pow_of_minus_1_for_k *

163 (M_n_l.getImag() * O_n_j__k_l.getImag())));

168 pow_of_minus_1_for_l = -pow_of_minus_1_for_l;
169 }

172 const FComplex<FReal> M_n_l = multipole_exp_src[index_n - l];
173 const FComplex<FReal> O_n_j__k_l = M2L_Outer_transfer[index_n_j - (k + l)];
174
175 L_j_k.incReal( pow_of_minus_1_for_l *

186 }//n
