33 RotationInfo(const int inDevP){
34 rotation_a = new FReal[int( ((inDevP)+1) * ((inDevP)+2) * 0.5 )];
35 for(int n = 0 ; n <= inDevP ; ++n){
36 for(int m = 0 ; m <= n ; ++m){
37 rotation_a[int(n*(n+1) * 0.5 + m)] = FMath::Sqrt( FReal((n+1+m)*(n+1-m)) / FReal(((2*n+1)*(2*n+3))) );
38 }
39 }
40 rotation_b = new FReal[(inDevP+1) * (inDevP+1)];
41 for(int n = 0 ; n <= inDevP ; ++n){
42 for(int m = -n ; m < 0 ; ++m){
43 rotation_b[n*(n+1) + m] = -FMath::Sqrt( FReal((n-m-1)*(n-m)) / FReal(((2*n-1)*(2*n+1))) );
44 }
45 for(int m = 0 ; m <= n ; ++m){
46 rotation_b[n*(n+1) + m] = FMath::Sqrt( FReal((n-m-1)*(n-m)) / FReal(((2*n-1)*(2*n+1))) );
47 }
48 }

103 void computeLegendre(const FReal inCosTheta, const FReal inSinTheta, FReal legendre[]){
104 const FReal invSinTheta = -inSinTheta;
105
106 legendre[0] = 1.0; // P_0,0(cosTheta) = 1
107 legendre[1] = inCosTheta; // P_1,0(cosTheta) = cosTheta
108 legendre[2] = invSinTheta;// P_1,1(cosTheta) = -sinTheta
109
110 int idxCurrentLM = 3; //current pointer on P_l,m
111 int idxCurrentL1M = 1; //pointer on P_{l-1},m => P_1,0
112 int idxCurrentL2M = 0; //pointer on P_{l-2},m => P_0,0
113 FReal fact = 3.0;
114
115 for(int l = 2; l <= devP ; ++l ){
116 // m from 0 to l - 2
117 for( int m = 0; m <= l - 2 ; ++m ){
118 legendre[idxCurrentLM] = (inCosTheta * FReal( 2 * l - 1 ) * legendre[idxCurrentL1M]
119 - FReal( l + m - 1 ) * legendre[idxCurrentL2M] )
120 / FReal( l - m );
121
122
123 // progress
124 ++idxCurrentLM;
125 ++idxCurrentL1M;
126 ++idxCurrentL2M;
127 }
128
129 // Compute P_l,{l-1}
130 legendre[idxCurrentLM++] = inCosTheta * FReal( 2 * l - 1 ) * legendre[idxCurrentL1M];
131
132 // Compute P_l,l
133 legendre[idxCurrentLM++] = fact * invSinTheta * legendre[idxCurrentL1M];
134
135 fact += FReal(2.0);
136 ++idxCurrentL1M;
137 }

158 static FReal A_div_A(int n, int m, int nu){
159 m = FMath::Abs(m);
160 nu = FMath::Abs(nu);
161 const int min = FMath::Min(m, nu);
162 const int max = FMath::Max(m, nu);
163 const int i_stop = max - min - 1; /* = n-min - (n-max+1) = n+max - (n+min+1) */
164
165 FReal num = FReal(n-max+1);
166 FReal denom = FReal(n+min+1);
167 FReal A_min_max = 1;
168 for (int i=0; i<=i_stop; ++i, ++num, ++denom){
169 A_min_max *= (num/denom);
170 }
171
172 if (nu == min)
173 return FMath::Sqrt(A_min_max);
174 else
175 return 1/FMath::Sqrt(A_min_max);
176 }

179 void rotation_coefficient_container_Fill(const FReal omega,
180 const FReal cos_gamma, const FReal sin_gamma,
181 const FReal chi, const RotationInfo& rotation_Info){
182
183 FComplex<FReal>** rcc_tmp_transposed = new FComplex<FReal>*[devP + 1];
184 for( int idxP = 0 ; idxP <= devP ; ++idxP){
185 const int rotationSize = ((idxP+1)*(2*idxP+1));
186 rcc_tmp_transposed[idxP] = new FComplex<FReal>[rotationSize];
187 }
188
189 FComplex<FReal> _pow_of_I_array[7];
190 _pow_of_I_array[0].setRealImag(0 , 1 ) /* I^{-3} */;
191 _pow_of_I_array[1].setRealImag(-1, 0 ) /* I^{-2} */;
192 _pow_of_I_array[2].setRealImag(0 , -1) /* I^{-1} */;
193 _pow_of_I_array[3].setRealImag(1 , 0 ) /* I^0 */;
194 _pow_of_I_array[4].setRealImag(0 , 1 ) /* I^1 */;
195 _pow_of_I_array[5].setRealImag(-1, 0 ) /* I^2 */;
196 _pow_of_I_array[6].setRealImag(0 , -1) /* I^3 */;
197
198 const FComplex<FReal>* pow_of_I_array = _pow_of_I_array + 3; /* points on I^0 */
199
200 FComplex<FReal>* const _precomputed_exp_I_chi_array = new FComplex<FReal>[2*devP + 1];
201 FComplex<FReal>* precomputed_exp_I_chi_array = _precomputed_exp_I_chi_array + devP;
202
203 FComplex<FReal>* const _precomputed_exp_I_omega_array = new FComplex<FReal>[2*devP + 1];
204 FComplex<FReal>* precomputed_exp_I_omega_array = _precomputed_exp_I_omega_array + devP;
205
206
207 // cos(x) = sin(x + Pi/2)
208 for(int m = -devP ; m <= devP ; ++m){
209 precomputed_exp_I_chi_array[m].setReal(FMath::Sin(FReal(m)*chi + FMath::FPiDiv2<FReal>()));
210 precomputed_exp_I_chi_array[m].setImag(FMath::Sin(FReal(m)*chi));
211 }
212 for(int nu = -devP ; nu <= devP ; ++nu){
213 precomputed_exp_I_omega_array[nu].setReal(FMath::Sin(FReal(nu)*omega + FMath::FPiDiv2<FReal>()));
214 precomputed_exp_I_omega_array[nu].setImag(FMath::Sin(FReal(nu)*omega));
215 }
216
217 FReal*const ass_Legendre_func_Array = new FReal[expSize];
218 FReal* p_ass_Legendre_func_Array = ass_Legendre_func_Array;
219 computeLegendre(cos_gamma, sin_gamma, ass_Legendre_func_Array);
220
221 for(int n = 0 ; n <= devP ; ++n){
222 // nu == 0:
223 FReal c_n_nu = 1;
224 rcc_tmp_transposed[n][getTranspRotationCoefP(n,0,0)].setReal(c_n_nu * (*p_ass_Legendre_func_Array));
225
226 ++p_ass_Legendre_func_Array;
227
228 // nu > 0:
229 FReal minus_1_pow_nu = -1;
230 for(int nu = 1 ; nu <= n; ++nu){
231 c_n_nu /= FMath::Sqrt(FReal((n-nu+1)*(n+nu)));
232 rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, 0)].setReal(minus_1_pow_nu * c_n_nu * (*p_ass_Legendre_func_Array));
233 rcc_tmp_transposed[n][getTranspRotationCoefP(n, -nu, 0)] = rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, 0)];
234 minus_1_pow_nu = -minus_1_pow_nu;
235 ++p_ass_Legendre_func_Array;
236 } // for nu
237
238 for(int m = 1 ; m <= n ; ++m){
239 for(int nu = -m; nu <= m ; ++nu){
240 const FReal H_nu_minus_1 = ( nu-1 <= -n ?
241 FReal(0.0) :
242 (cos_gamma +1) * rotation_Info.rotation_b[getRotationB(n, -nu)]
243 * rcc_tmp_transposed[n-1][getTranspRotationCoefP(n-1, nu-1, m-1)].getReal());
244 const FReal H_nu_plus_1 = ( nu+1 >= n ?
245 FReal(0.0) :
246 (cos_gamma -1) * rotation_Info.rotation_b[getRotationB(n, nu)]
247 * rcc_tmp_transposed[n-1][getTranspRotationCoefP(n-1, nu+1, m-1)].getReal());
248
249 const FReal H_nu = ( FMath::Abs(nu) >= n ?
250 FReal(0.0) :
251 sin_gamma * rotation_Info.rotation_a[getRotationA(n-1, nu)]
252 * rcc_tmp_transposed[n-1][getTranspRotationCoefP(n-1, nu, m-1)].getReal() );
253
254
255 rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, m)].setReal( (FReal(0.5) * (-H_nu_minus_1 - H_nu_plus_1) - H_nu)
256 / rotation_Info.rotation_b[getRotationB(n, -m)]);
257 } // for nu
258 } // for m
259
260 for(int m = 1 ; m <= n ; ++m){
261 for(int nu = -n ; nu <= -m-1; ++nu){
262 rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, m)] = rcc_tmp_transposed[n][getTranspRotationCoefP(n, -m, -nu)];
263 } // for nu
264
265 for(int nu = m+1 ; nu <= n; ++nu){
266 rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, m)] = rcc_tmp_transposed[n][getTranspRotationCoefP(n, m, nu)];
267 } // for nu
268 } // for m
269 } // for n
270
271 for(int n = 0 ; n <= devP ; ++n){
272 for(int nu = 0 ; nu <= n; ++nu){
273 for(int m = -n; m <= n; ++m){
274 FReal A_terms = A_div_A(n, m, nu); /* A_n^m / A_n^nu */
275 int abs_m_minus_abs_nu_mod4 = (FMath::Abs(m) - FMath::Abs(nu)) % 4; /* can be negative! */
276 const FComplex<FReal> p_H_tmp = ( m >= 0 ?
277 rcc_tmp_transposed[n][getTranspRotationCoefP(n, nu, m)] :
278 rcc_tmp_transposed[n][getTranspRotationCoefP(n, -nu, -m)]) ;
279
280 /*************** T_Outer_n^{nu, m}(omega, gamma, chi): ***************/
281 rcc_outer[n][getRotationCoefP(n, nu, m)] = p_H_tmp;
282 /* H_n^{nu, m}(gamma) => T_n^{nu, m}(omega, gamma, chi) */
283 rcc_outer[n][getRotationCoefP(n, nu, m)] *= precomputed_exp_I_chi_array[ + m];
284 rcc_outer[n][getRotationCoefP(n, nu, m)] *= precomputed_exp_I_omega_array[ - nu];
285 /* T_Outer_j^{nu, k}(omega, gamma, chi) = i^{|k|-|nu|} (A_j^nu / A_j^k) T_j^{nu, k}(omega, gamma, chi) (6) */
286 rcc_outer[n][getRotationCoefP(n, nu, m)] *= pow_of_I_array[abs_m_minus_abs_nu_mod4];
287 rcc_outer[n][getRotationCoefP(n, nu, m)] *= (FReal(1.0) / A_terms);
288
289
290 /*************** T_Inner_n^{nu, m}(chi, gamma, omega): ***************/
291 rcc_inner[n][getRotationCoefP(n, nu, m)] = p_H_tmp;
292 /* H_n^{nu, m}(gamma) => T_n^{nu, m}(chi, gamma, omega) */
293 rcc_inner[n][getRotationCoefP(n, nu, m)] *= precomputed_exp_I_omega_array[ + m];
294 rcc_inner[n][getRotationCoefP(n, nu, m)] *= precomputed_exp_I_chi_array[ - nu];
295 /* T_Inner_j^{nu, k}(omega, gamma, chi) = i^{|nu|-|k|} (A_j^k / A_j^nu) T_j^{nu, k}(omega, gamma, chi) (7) */
296 rcc_inner[n][getRotationCoefP(n, nu, m)] *= pow_of_I_array[- abs_m_minus_abs_nu_mod4];
297 rcc_inner[n][getRotationCoefP(n, nu, m)] *= A_terms;
298
299 }// for m
300 } // for nu
301 } // for n
302
303
304 delete[] (ass_Legendre_func_Array);
305 delete[] (_precomputed_exp_I_chi_array);
306 delete[] (_precomputed_exp_I_omega_array);
307 FMemUtils::DeleteAllArray( rcc_tmp_transposed, devP);
308 delete[] rcc_tmp_transposed;
309 }

344 void allocAndInit(){
345 // M2L transfer, there is a maximum of 3 neighbors in each direction,
346 // so 6 in each dimension
347 preM2LTransitions = new RotationM2LTransfer*[Parent::treeHeight];
348 memset(preM2LTransitions.getPtr(), 0, sizeof(FComplex<FReal>*) * (Parent::treeHeight));
349 // We start from the higher level
350 FReal treeWidthAtLevel = Parent::boxWidth;
351 for(int idxLevel = 0 ; idxLevel < Parent::treeHeight ; ++idxLevel ){
352 // Allocate data for this level
353 preM2LTransitions[idxLevel] = reinterpret_cast<RotationM2LTransfer*>(new char[(7 * 7 * 7) * sizeof(RotationM2LTransfer)]);
354 // Precompute transfer vector
355 for(int idxX = -3 ; idxX <= 3 ; ++idxX ){
356 for(int idxY = -3 ; idxY <= 3 ; ++idxY ){
357 for(int idxZ = -3 ; idxZ <= 3 ; ++idxZ ){
358 new (&preM2LTransitions[idxLevel][indexM2LTransition(idxX,idxY,idxZ)]) RotationM2LTransfer(Parent::devP,devM2lP,Parent::harmonic.getExpSize());
359
360 if(FMath::Abs(idxX) > 1 || FMath::Abs(idxY) > 1 || FMath::Abs(idxZ) > 1){
361 const FPoint<FReal> relativePos( FReal(-idxX) * treeWidthAtLevel , FReal(-idxY) * treeWidthAtLevel , FReal(-idxZ) * treeWidthAtLevel );
362 preM2LTransitions[idxLevel][indexM2LTransition(idxX,idxY,idxZ)].transfer_M2L_rotation_Fill(FSpherical<FReal>(relativePos), rotation_Info);
363 }
364 }
365 }
366 }

368 treeWidthAtLevel /= 2;

436 void rotation_Rotate_multipole_expansion_terms(const FComplex<FReal>*const FRestrict multipole_exp,
437 const FComplex<FReal>* const FRestrict * const FRestrict rcc_outer,
438 FComplex<FReal>*const FRestrict rot_multipole_exp){
439
440 FComplex<FReal>* p_rot_multipole_exp = rot_multipole_exp;
441
442 for(int nu = 0 ; nu <= (Parent::devP/2) ; ++nu){
443 for(int j = nu; j <= (Parent::devP-nu) ; ++j){
444 const FComplex<FReal>* p_rcc_outer = &rcc_outer[j][RotationM2LTransfer::getRotationCoefP(j, nu, j)];
445 const FComplex<FReal>* p_multipole_exp = &multipole_exp[Parent::harmonic.getPreExpRedirJ(j) + j];
446 FReal minus_1_pow_k = FReal(j&1 ? -1 : 1);
447
448 for(int k = -j ; k < 0 ; ++k){ /* k < 0 */
449 p_rot_multipole_exp->incReal( minus_1_pow_k *
450 ((p_multipole_exp->getReal() * p_rcc_outer->getReal()) +
451 (p_multipole_exp->getImag() * p_rcc_outer->getImag())) );
452 p_rot_multipole_exp->incImag( minus_1_pow_k *
453 ((p_multipole_exp->getReal() * p_rcc_outer->getImag()) -
454 (p_multipole_exp->getImag() * p_rcc_outer->getReal())) );
455
456 minus_1_pow_k = -minus_1_pow_k;
457 --p_rcc_outer;
458 --p_multipole_exp;
459 } /* for k */
460
461 for(int k = 0; k <= j ; ++k){ /* k >= 0 */
462 p_rot_multipole_exp->incReal(
463 ((p_multipole_exp->getReal() * p_rcc_outer->getReal()) -
464 (p_multipole_exp->getImag() * p_rcc_outer->getImag())) );
465 p_rot_multipole_exp->incImag(
466 ((p_multipole_exp->getReal() * p_rcc_outer->getImag()) +
467 (p_multipole_exp->getImag() * p_rcc_outer->getReal())) );
468
469 --p_rcc_outer;
470 ++p_multipole_exp;
471 } /* for k */
472
473 ++p_rot_multipole_exp;
474 } /* for j */
475 } /* for nu */

479 void M2L_z_axis(FComplex<FReal>* const FRestrict rot_local_exp,
480 const FComplex<FReal>* const FRestrict rot_multipole_exp,

483
484 for(int j = 0 ; j <= Parent::devP; ++j){
485 const FReal* p_outer_array_j = outer_array + j;

488 for(int k = 0 ; k <= min_j ; ++k){

494 } /* for n */

496 } /* for k */

501 void rotation_Rotate_local_expansion_terms(const FComplex<FReal>*const rot_local_exp,
502 const FComplex<FReal>*const FRestrict *const FRestrict rcc_inner,
503 FComplex<FReal>*const FRestrict local_exp){
