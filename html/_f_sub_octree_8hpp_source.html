64 MortonIndex getLeafIndex(const MortonIndex index, const int inTreeHeight) const {
65 // Remove right useless part - used by child
66 const MortonIndex fullIndex = index >> (3 * (inTreeHeight - (this->subOctreeHeight + this->subOctreePosition) ) );

79 void createPreviousCells(MortonIndex arrayIndex, MortonIndex inLeafCellIndex, const FTreeCoordinate& treePosition){
80 int indexLevel = this->subOctreeHeight - 1;
81 int bottomToTop = 0;
82 while(indexLevel >= 0 && !this->cells[indexLevel][arrayIndex]){
83 CellClass* const newNode = cellAllocator.newObject();//new CellClass();
84 newNode->setMortonIndex(inLeafCellIndex);
85
86 newNode->setCoordinate(treePosition.getX() >> bottomToTop,
87 treePosition.getY() >> bottomToTop,
88 treePosition.getZ() >> bottomToTop);
89
90 newNode->setLevel(this->subOctreePosition + indexLevel);
91
92 this->cells[indexLevel][arrayIndex] = newNode;
93
94 --indexLevel;
95 ++bottomToTop;
96
97 arrayIndex >>= 3;
98 inLeafCellIndex >>= 3;
99 }
100 }

107 void newLeafInserted(const int arrayIndex, const MortonIndex inLeafCellIndex, const FTreeCoordinate& host){
108 createPreviousCells(arrayIndex,inLeafCellIndex, host);
109 // Update if this is the bottom left

120 bool removeCellsFromLeaf( int arrayIndex ){
121 // last array index
122 int indexLevel = this->subOctreeHeight - 1;
123
124 // Manage border limits
125 if(arrayIndex == this->leftLeafIndex && arrayIndex == this->rightLeafIndex){
126 this->rightLeafIndex = -1;
127 // only one cells, return true
128 }
129 else if(arrayIndex == this->leftLeafIndex){
130 while( !this->cells[indexLevel][++this->leftLeafIndex] );
131 }
132 else if(arrayIndex == this->rightLeafIndex){
133 while( !this->cells[indexLevel][--this->rightLeafIndex] );
134 }
135
136 // remove the last cells
137 cellAllocator.deleteObject(this->cells[indexLevel][arrayIndex]);
138 this->cells[indexLevel][arrayIndex] =nullptr;
139 // progress upper
140 --indexLevel;
141 arrayIndex >>= 3;
142
143 // to test if 8 child are empty
144 CellClass* emptyArray[8];
145 memset(emptyArray , 0, sizeof(CellClass*) * 8);
146
147 // continue while we are not in the last level and our child are empty
148 while(indexLevel >= 0 && memcmp(&this->cells[indexLevel+1][arrayIndex<<3], emptyArray, 8 * sizeof(CellClass*)) == 0 ){
149 cellAllocator.deleteObject( this->cells[indexLevel][arrayIndex] );
150 this->cells[indexLevel][arrayIndex] = nullptr;
151
152 --indexLevel;
153 arrayIndex >>= 3;
154 }
155 // return true if there is no more child == 0 cell at level 0
156 return memcmp(this->cells[0], emptyArray, 8 * sizeof(CellClass*)) == 0;
157 }

172 FAbstractSubOctree(FAbstractSubOctree* const inParent, const int inIndexInParent,
173 const int inSubOctreeHeight, const int inSubOctreePosition, const bool inIsLeafSubtree) :
174 cells(nullptr), parent( inParent ), indexInParent(inIndexInParent), leftLeafIndex(1 << (3 * inSubOctreeHeight)), rightLeafIndex(-1),
175 subOctreeHeight( inSubOctreeHeight ), subOctreePosition( inSubOctreePosition ), isLeafSubtree(inIsLeafSubtree) {
176
177 this->cells = new CellClass**[this->subOctreeHeight];
178 FAssertLF(this->cells, "Allocation failled");
179
180 memset(this->cells, 0, sizeof(CellClass**) * subOctreeHeight);
181
182 // We start at a sub-level - 8^1
183 int cellsAtlevel = 8;
184 for( int indexLevel = 0 ; indexLevel < this->subOctreeHeight ; ++indexLevel ){
185 this->cells[indexLevel] = new CellClass*[cellsAtlevel];
186 FAssertLF(this->cells[indexLevel], "Allocation failled");
187
188 memset(this->cells[indexLevel], 0, sizeof(CellClass*) * cellsAtlevel);
189
190 cellsAtlevel <<= 3; // => * 8 >> 8^indexLevel
191 }
192 }

198 virtual ~FAbstractSubOctree(){
199 int mostRight = rightLeafIndex;
200 int mostLeft = leftLeafIndex;
201
202 for( int indexLevel = this->subOctreeHeight - 1 ; indexLevel >= 0 ; --indexLevel ){
203 for( int indexCells = mostLeft ; indexCells <= mostRight ; ++indexCells ){
204 if(this->cells[indexLevel][indexCells]){
205 cellAllocator.deleteObject( this->cells[indexLevel][indexCells] );
206 }
207 }
208
209 delete [] this->cells[indexLevel];
210 mostLeft >>= 3;
211 mostRight >>= 3;
212 }
213
214 delete [] this->cells;
215 }

514 template<typename... Args>
515 void insert(const MortonIndex index, const FTreeCoordinate& host, const int inTreeHeight, const FPoint<FReal>& inParticlePosition,
516 Args... args){
517 // We need the morton index at the bottom level of this sub octree
518 // so we remove the right side
519 const MortonIndex arrayIndex = Parent::getLeafIndex(index,inTreeHeight);
520 // Is there already a leaf?
521 if( !this->subleafs[arrayIndex] ){
522 // We need to create leaf sub octree
523 const int nextSubOctreePosition = this->subOctreePosition + this->subOctreeHeight;
524 const int nextSubOctreeHeight = FMath::Min(inTreeHeight - nextSubOctreePosition, this->subOctreeHeight);
525
526 // Next suboctree is a middle suboctree
527 if(inTreeHeight > nextSubOctreeHeight + nextSubOctreePosition){
528 this->subleafs[arrayIndex] = new FSubOctree(this,int(arrayIndex),nextSubOctreeHeight,nextSubOctreePosition);
529 }
530 // Or next suboctree contains the reail leaf!
531 else{
532 this->subleafs[arrayIndex] = new SubOctreeWithLeaf(this,int(arrayIndex),nextSubOctreeHeight,nextSubOctreePosition);
533 }
534
535 const FTreeCoordinate hostAtLevel(
536 host.getX() >> (inTreeHeight - nextSubOctreePosition ),
537 host.getY() >> (inTreeHeight - nextSubOctreePosition ),
538 host.getZ() >> (inTreeHeight - nextSubOctreePosition ));
539
540 // We need to inform parent class
541 Parent::newLeafInserted( int(arrayIndex), index >> (3 * (inTreeHeight-nextSubOctreePosition) ), hostAtLevel);
542 }
543 // Ask next suboctree to insert the particle
544 if(this->subleafs[arrayIndex]->isLeafPart()){
545 ((SubOctreeWithLeaf*)this->subleafs[arrayIndex])->insert( index, host, inTreeHeight, inParticlePosition, args... );
546 }
547 else{
548 ((FSubOctree*)this->subleafs[arrayIndex])->insert( index, host, inTreeHeight, inParticlePosition, args... );
549 }
550 }
