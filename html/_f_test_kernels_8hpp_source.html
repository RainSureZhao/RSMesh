137template< class OctreeClass, class CellClass, class ContainerClass, class LeafClass>
138void ValidateFMMAlgo(OctreeClass* const tree){
139 std::cout << "Check Result\n";
140 const int TreeHeight = tree->getHeight();
141 long long int NbPart = 0;
142 { // Check that each particle has been summed with all other
143 tree->forEachCellLeaf([&](CellClass* cell, LeafClass* leaf){
144 if(cell->getDataUp() != leaf->getSrc()->getNbParticles() ){
145 std::cout << "Problem P2M : " << cell->getDataUp() <<
146 " (should be " << leaf->getSrc()->getNbParticles() << ")\n";
147 }
148 NbPart += leaf->getSrc()->getNbParticles();
149 });
150 }
151 { // Ceck if there is number of NbPart summed at level 1
152 typename OctreeClass::Iterator octreeIterator(tree);
153 octreeIterator.moveDown();
154 long long int res = 0;
155 do{
156 res += octreeIterator.getCurrentCell()->getDataUp();
157 } while(octreeIterator.moveRight());
158 if(res != NbPart){
159 std::cout << "Problem M2M at level 1 : " << res << "\n";
160 }
161 }
162 { // Ceck if there is number of NbPart summed at level 1
163 typename OctreeClass::Iterator octreeIterator(tree);
164 octreeIterator.gotoBottomLeft();
165 for(int idxLevel = TreeHeight - 1 ; idxLevel > 1 ; --idxLevel ){
166 long long int res = 0;
167 do{
168 res += octreeIterator.getCurrentCell()->getDataUp();
169 } while(octreeIterator.moveRight());
170 if(res != NbPart){
171 std::cout << "Problem M2M at level " << idxLevel << " : " << res << "\n";
172 }
173 octreeIterator.moveUp();
174 octreeIterator.gotoLeft();
175 }
176 }
177 { // Check that each particle has been summed with all other
178 typename OctreeClass::Iterator octreeIterator(tree);
179 octreeIterator.gotoBottomLeft();
180 do{
181 const bool isUsingTsm = (octreeIterator.getCurrentListTargets() != octreeIterator.getCurrentListSrc());
182 const long long int* dataDown = octreeIterator.getCurrentListTargets()->getDataDown();
183 for(FSize idxPart = 0 ; idxPart < octreeIterator.getCurrentListTargets()->getNbParticles() ; ++idxPart){
184 if( (!isUsingTsm && dataDown[idxPart] != NbPart - 1) ||
185 (isUsingTsm && dataDown[idxPart] != NbPart) ){
186 std::cout << "Problem L2P + P2P : " << dataDown[idxPart] << ", " <<
187 " NbPart : " << NbPart << ", " <<
188 " ( Index " << octreeIterator.getCurrentGlobalIndex() << ")\n";
189 }
190 }
191 } while(octreeIterator.moveRight());
192 }
193
194 std::cout << "Done\n";
195}
