155 static bool Load(const char filename[], OctreeClass& tree){

157 FBufferReader buffer;
158
159 if(!file.good()){
160 return false;
161 }
162
163 size_t sizeof_freal = 0;
164 file.read((char*)&sizeof_freal, sizeof(size_t));
165 if( sizeof_freal != sizeof(FReal)){
166 std::cerr << "[Loader] Error Freal do not coincide with file type:\n";
167 std::cerr << "[Loader] In file : " << sizeof_freal << " Real : " << sizeof(FReal) << "\n";
168 return false;
169 }
170
171 int treeHeight = 0;
172 file.read((char*)&treeHeight, sizeof(int));
173 int treeSubHeight = 0;
174 file.read((char*)&treeSubHeight, sizeof(int));
175 FReal boxWidth = 0;
176 file.read((char*)&boxWidth, sizeof(FReal));
177 FPoint<FReal> center;
178 file.read((char*)&center, sizeof(FPoint<FReal>));
179
180 tree.~OctreeClass();
181 new (&tree) OctreeClass(treeHeight,treeSubHeight,boxWidth,center);
182
183 {
184 int nbLeaf = 0;
185 file.read((char*)&nbLeaf, sizeof(int));
186
187 TargetSourceDiff tsm = TsmUndef;
188 file.read((char*)&tsm,sizeof(tsm));
189
190 for(int idxLeaf = 0 ; idxLeaf < nbLeaf ; ++idxLeaf){
191 MortonIndex mindex = 0;
192 file.read((char*)&mindex, sizeof(mindex));
193
194 FSize sizeOfLeaf = 0;
195 file.read((char*)&sizeOfLeaf, sizeof(FSize));
196
197 buffer.seek(0);
198 buffer.reserve(sizeOfLeaf);
199 file.read((char*)buffer.data(), sizeOfLeaf);
200
201 LeafClass*const leaf = tree.createLeaf(mindex);
202 leaf->getSrc()->restore(buffer);
203
204 if( tsm == TsmUsed ){
205 leaf->getTargets()->restore(buffer);
206 }
207 }
208 }
209
210 // Start from leal level - 1
211 typename OctreeClass::Iterator octreeIterator(&tree);
212 octreeIterator.gotoBottomLeft();
213
214 typename OctreeClass::Iterator avoidGotoLeftIterator(octreeIterator);
215
216 // for each levels
217 for(int idxLevel = tree.getHeight() - 1 ; idxLevel > 1 ; --idxLevel ){
218 int nbCells = 0;
219 file.read((char*)&nbCells, sizeof(int));
220
221 do{
222 MortonIndex mindex;
223 file.read((char*)&mindex, sizeof(MortonIndex));
224 if(mindex != octreeIterator.getCurrentGlobalIndex()){
225 std::cerr << "[Loader] Error indexes are different\n";

230 file.read((char*)&sizeOfCell, sizeof(FSize));
231
232 buffer.seek(0);
233 buffer.reserve(sizeOfCell);
234 file.read((char*)buffer.data(), sizeOfCell);

236 octreeIterator.getCurrentCell()->restore(buffer);
237
238 --nbCells;
239 } while(octreeIterator.moveRight());

242 std::cerr << "[Loader] Wrong number of cells at level " << idxLevel << "\n";
243 return false;
244 }
245
246 avoidGotoLeftIterator.moveUp();
247 octreeIterator = avoidGotoLeftIterator;
248 }
249
250 file.close();
