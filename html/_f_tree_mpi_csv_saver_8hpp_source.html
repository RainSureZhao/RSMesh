56 void exportTree(OctreeClass*const tree){
57 char currentFilename[512];
58 sprintf(currentFilename, basefile, nbFrames++);
59
60 // Erase existing file
61 if(comm.processId() == 0){
62 FILE* fd = fopen(currentFilename, "w");
63 if(fd) fclose(fd);
64 }
65
66 MPI_Barrier(comm.getComm());
67
68 // All procs open the file
69 MPI_File file;
70 const int fileIsOpen = MPI_File_open( comm.getComm(), currentFilename, MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL, &file );
71
72 // Is it open?
73 if(fileIsOpen != MPI_SUCCESS){
74 std::cout << "Cannot create parallel file, exportTree abort." << std::endl;
75 return;
76 }
77
78 MPI_Offset disp = 0;
79 {
80 // Each procs count its particles
81 FSize mypart = 0;
82 typename OctreeClass::Iterator octreeIterator(tree);
83 octreeIterator.gotoBottomLeft();
84 do{
85 mypart += octreeIterator.getCurrentListTargets()->getNbParticles();
86 const bool isUsingTsm = (octreeIterator.getCurrentListTargets() != octreeIterator.getCurrentListSrc());
87 if( isUsingTsm ) mypart += octreeIterator.getCurrentListSrc()->getNbParticles();
88 } while(octreeIterator.moveRight());
89
90 // Gather particles number
91 FSize*const particlesPerProcs = new FSize[comm.processCount()];
92 FMpi::Assert( MPI_Allgather(&mypart, 1, FMpi::GetType(mypart), particlesPerProcs, 1, FMpi::GetType(mypart), comm.getComm()), __LINE__);
93 FSize previousPart = 0;
94 for(int idxProc = 0 ; idxProc < comm.processId() ; ++idxProc){
95 previousPart += particlesPerProcs[idxProc];
96 }
97 delete[] particlesPerProcs;
98 // How many particle before me * size of particle in file
99 disp = previousPart * SizeOfLine;
100 if(includeHeader && comm.processId() != 0){
101 disp += SizeOfHeader;
102 }
103 }
104
105 // Set view
106 const char* const native = "native";
107 MPI_File_set_view( file, disp, MPI_BYTE, MPI_BYTE, const_cast<char*>(native), MPI_INFO_NULL);
108
109 // Write header if needed
110 MPI_Offset offset = 0;
111 if(includeHeader && comm.processId() == 0){
112 MPI_File_write_at( file, 0, header, SizeOfHeader, MPI_BYTE, MPI_STATUS_IGNORE);
113 offset += SizeOfHeader;
114 }
115
116 // Write particle
117 char line[SizeOfLine+1];
118 typename OctreeClass::Iterator octreeIterator(tree);
119 octreeIterator.gotoBottomLeft();
120 do{
121 {
122 const ContainerClass* container = octreeIterator.getCurrentListTargets();
123 for(FSize idxPart = 0 ; idxPart < container->getNbParticles() ; ++idxPart){
124 FReal values[4];
125 container->fillToCsv(idxPart, values);
126 sprintf(line, lineFormat, values[0],values[1],
127 values[2],values[3]);
128
129 MPI_File_write_at( file, offset, line, SizeOfLine, MPI_BYTE, MPI_STATUS_IGNORE);
130 offset += SizeOfLine;
131 }
132 }
133
134
135 const bool isUsingTsm = (octreeIterator.getCurrentListTargets() != octreeIterator.getCurrentListSrc());
136 if( isUsingTsm ){
137 const ContainerClass* container = octreeIterator.getCurrentListSrc();
138 for(FSize idxPart = 0 ; idxPart < container->getNbParticles() ; ++idxPart){
139 FReal values[4];
140 container->fillToCsv(idxPart, values);
141 sprintf(line, lineFormat, values[0],values[1],
142 values[2],values[3]);
143
144 MPI_File_write_at( file, offset, line, SizeOfLine, MPI_BYTE, MPI_STATUS_IGNORE);
145 offset += SizeOfLine;
146 }
147 }
148 } while(octreeIterator.moveRight());
149
150 MPI_File_close( &file );
151 }
