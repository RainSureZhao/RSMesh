36 FComplex<FReal> transformed_local_exp[NLHS * NVALS * TransformedVectorSize];

38public:

42 memset(transformed_multipole_exp, 0,
43 sizeof(FComplex<FReal>) * NRHS * NVALS * TransformedVectorSize);
44 memset(transformed_local_exp, 0,
