99 void initTensorM2MandL2L(const int TreeLevel, const FReal ParentWidth)
100 {
101 FReal ChildCoords[3][ORDER];
102 FPoint<FReal> ChildCenter;
103
104 // Ratio of extended cell widths (definition: child ext / parent ext)
105 const FReal ExtendedCellRatio =
106 FReal(FReal(ParentWidth)/FReal(2.) + CellWidthExtension) / FReal(ParentWidth + CellWidthExtension);
107
108 // Child cell width
109 const FReal ChildWidth(FReal(2.)*ExtendedCellRatio);
110
111 // loop: child cells
112 for (unsigned int child=0; child<8; ++child) {
113
114 // set child info
115 FUnifTensor<FReal, ORDER>::setRelativeChildCenter(child, ChildCenter, ExtendedCellRatio);
116 FUnifTensor<FReal, ORDER>::setPolynomialsRoots(ChildCenter, ChildWidth, ChildCoords);
117
118 // allocate memory
119 ChildParentInterpolator[TreeLevel][child] = new FReal [3 * ORDER*ORDER];
120 assembleInterpolator(ORDER, ChildCoords[0], ChildParentInterpolator[TreeLevel][child]);
121 assembleInterpolator(ORDER, ChildCoords[1], ChildParentInterpolator[TreeLevel][child] + 1 * ORDER*ORDER);
122 assembleInterpolator(ORDER, ChildCoords[2], ChildParentInterpolator[TreeLevel][child] + 2 * ORDER*ORDER);
123 }
124
125
126 // init permutations
127 for (unsigned int i=0; i<ORDER; ++i) {
128 for (unsigned int j=0; j<ORDER; ++j) {
129 for (unsigned int k=0; k<ORDER; ++k) {
130 const unsigned int index = k*ORDER*ORDER + j*ORDER + i;
131 perm[0][index] = k*ORDER*ORDER + j*ORDER + i;
132 perm[1][index] = i*ORDER*ORDER + k*ORDER + j;
133 perm[2][index] = j*ORDER*ORDER + i*ORDER + k;
134 }
135 }

220 void assembleInterpolator(const unsigned int NumberOfLocalPoints,
221 const FPoint<FReal> *const LocalPoints,
222 FReal *const Interpolator) const
223 {
224 // values of Lagrange polynomials of source particle: L_o(x_i)
225 FReal L_of_x[ORDER][3];
226 // loop: local points (mapped in [-1,1])
227 for (unsigned int m=0; m<NumberOfLocalPoints; ++m) {
228 // evaluate Lagrange polynomials at local points
229 for (unsigned int o=0; o<ORDER; ++o) {
230 L_of_x[o][0] = BasisType::L(o, LocalPoints[m].getX());
231 L_of_x[o][1] = BasisType::L(o, LocalPoints[m].getY());
232 L_of_x[o][2] = BasisType::L(o, LocalPoints[m].getZ());
233 }
234
235 // assemble interpolator
236 for (unsigned int n=0; n<nnodes; ++n) {
237 Interpolator[n*NumberOfLocalPoints + m] = FReal(1.);
238 for (unsigned int d=0; d<3; ++d) {
239 const unsigned int j = node_ids[n][d];
240 // The Lagrange case is much simpler than the Chebyshev case
241 // as no summation is required
242 Interpolator[n*NumberOfLocalPoints + m] *= L_of_x[j][d];
243 }
244
245 }
246
247 }

338 // PB: improved version of applyM2M/L2L also applies to Lagrange interpolation
339 // PB: Multidim version handled in kernel !
340 void applyM2M(const unsigned int ChildIndex,
341 const FReal *const ChildExpansion,
342 FReal *const ParentExpansion,
343 const unsigned int TreeLevel = 2) const
344 {
345 FReal Exp[nnodes], PermExp[nnodes];
346 // ORDER*ORDER*ORDER * (2*ORDER-1)
347 FBlas::gemtm(ORDER, ORDER, ORDER*ORDER, FReal(1.),
348 ChildParentInterpolator[TreeLevel][ChildIndex], ORDER,
349 const_cast<FReal*>(ChildExpansion), ORDER, PermExp, ORDER);
350
351 for (unsigned int n=0; n<nnodes; ++n) Exp[n] = PermExp[perm[1][n]];
352 // ORDER*ORDER*ORDER * (2*ORDER-1)
353 FBlas::gemtm(ORDER, ORDER, ORDER*ORDER, FReal(1.),
354 ChildParentInterpolator[TreeLevel][ChildIndex] + 2 * ORDER*ORDER, ORDER,
355 Exp, ORDER, PermExp, ORDER);
356
357 for (unsigned int n=0; n<nnodes; ++n) Exp[perm[1][n]] = PermExp[perm[2][n]];
358 // ORDER*ORDER*ORDER * (2*ORDER-1)
359 FBlas::gemtm(ORDER, ORDER, ORDER*ORDER, FReal(1.),
360 ChildParentInterpolator[TreeLevel][ChildIndex] + 1 * ORDER*ORDER, ORDER,
361 Exp, ORDER, PermExp, ORDER);
362
363 for (unsigned int n=0; n<nnodes; ++n) ParentExpansion[perm[2][n]] += PermExp[n];
364 }

407inline void FUnifInterpolator<FReal, ORDER,MatrixKernelClass,NVALS>::applyP2M(const FPoint<FReal>& center,
408 const FReal width,
409 FReal *const multipoleExpansion,
410 const ContainerClass *const inParticles) const
411{
412
413 // allocate stuff
414 const map_glob_loc<FReal> map(center, width);
415 FPoint<FReal> localPosition;
416
417 // loop over source particles
418 const FReal*const positionsX = inParticles->getPositions()[0];
419 const FReal*const positionsY = inParticles->getPositions()[1];
420 const FReal*const positionsZ = inParticles->getPositions()[2];
421
422 for(FSize idxPart = 0 ; idxPart < inParticles->getNbParticles() ; ++idxPart){
423 // map global position to [-1,1]
424 map(FPoint<FReal>(positionsX[idxPart],positionsY[idxPart],positionsZ[idxPart]), localPosition); // 15 flops
425 // evaluate Lagrange polynomial at local position
426 FReal L_of_x[ORDER][3];
427 for (unsigned int o=0; o<ORDER; ++o) {
428 L_of_x[o][0] = BasisType::L(o, localPosition.getX()); // 3 * ORDER*(ORDER-1) flops PB: TODO confirm
429 L_of_x[o][1] = BasisType::L(o, localPosition.getY()); // 3 * ORDER*(ORDER-1) flops
430 L_of_x[o][2] = BasisType::L(o, localPosition.getZ()); // 3 * ORDER*(ORDER-1) flops
431 }
432
433 for(int idxRhs = 0 ; idxRhs < nRhs ; ++idxRhs){
434
435 // compute weight
436 FReal weight[nVals];
437 for(int idxVals = 0 ; idxVals < nVals ; ++idxVals){
438
439 // read physicalValue
440 const FReal*const physicalValues = inParticles->getPhysicalValues(idxVals,idxRhs);
441 weight[idxVals] = physicalValues[idxPart];
442
443 } // nVals
444
445 // assemble multipole expansions
446 for (unsigned int i=0; i<ORDER; ++i) {
447 for (unsigned int j=0; j<ORDER; ++j) {
448 for (unsigned int k=0; k<ORDER; ++k) {
449 const unsigned int idx = idxRhs*nVals*nnodes + k*ORDER*ORDER + j*ORDER + i;
450 const FReal S = L_of_x[i][0] * L_of_x[j][1] * L_of_x[k][2];
451
452 for(int idxVals = 0 ; idxVals < nVals ; ++idxVals)
453 multipoleExpansion[idxVals*nnodes+idx] += S * weight[idxVals]; // 3 * ORDER*ORDER*ORDER flops
454 }
455 }
456 }
457
458 } // nRhs
459
460 } // flops: N * (3 * ORDER*ORDER*ORDER + 3 * 3 * ORDER*(ORDER-1)) flops
461
462}

470inline void FUnifInterpolator<FReal, ORDER,MatrixKernelClass,NVALS>::applyL2P(const FPoint<FReal>& center,
471 const FReal width,
472 const FReal *const localExpansion,
473 ContainerClass *const inParticles) const
474{
475 // loop over particles
476 const map_glob_loc<FReal> map(center, width);
477 FPoint<FReal> localPosition;
478
479 const FReal*const positionsX = inParticles->getPositions()[0];
480 const FReal*const positionsY = inParticles->getPositions()[1];
481 const FReal*const positionsZ = inParticles->getPositions()[2];
482
483 const FSize nParticles = inParticles->getNbParticles();
484
485 for(FSize idxPart = 0 ; idxPart < nParticles ; ++ idxPart){
486
487 // map global position to [-1,1]
488 map(FPoint<FReal>(positionsX[idxPart],positionsY[idxPart],positionsZ[idxPart]), localPosition); // 15 flops
489
490 // evaluate Lagrange polynomial at local position
491 FReal L_of_x[ORDER][3];
492 for (unsigned int o=0; o<ORDER; ++o) {
493 L_of_x[o][0] = BasisType::L(o, localPosition.getX()); // 3 * ORDER*(ORDER-1) flops
494 L_of_x[o][1] = BasisType::L(o, localPosition.getY()); // 3 * ORDER*(ORDER-1) flops
495 L_of_x[o][2] = BasisType::L(o, localPosition.getZ()); // 3 * ORDER*(ORDER-1) flops
496 }
497
498 // loop over dim of local expansions
499 for(int idxLhs = 0 ; idxLhs < nLhs ; ++idxLhs){
500 // distribution over potential components:
501 // We sum the multidim contribution of PhysValue
502 // This was originally done at M2L step but moved here
503 // because their storage is required by the force computation.
504 // In fact : f_{ik}(x)=w_j(x) \nabla_{x_i} K_{ij}(x,y)w_j(y))
505 const unsigned int idxPot = idxLhs / nPV;
506
507
508
509 // interpolate and increment target value
510 FReal targetValue[nVals];
511 for(int idxVals = 0 ; idxVals < nVals ; ++idxVals)
512 targetValue[idxVals]=0.;
513 {
514 for (unsigned int l=0; l<ORDER; ++l) {
515 for (unsigned int m=0; m<ORDER; ++m) {
516 for (unsigned int n=0; n<ORDER; ++n) {
517 const unsigned int idx = idxLhs*nVals*nnodes + n*ORDER*ORDER + m*ORDER + l;
518 const FReal S = L_of_x[l][0] * L_of_x[m][1] * L_of_x[n][2];
519
520 for(int idxVals = 0 ; idxVals < nVals ; ++idxVals)
521 targetValue[idxVals] += S * localExpansion[idxVals*nnodes+idx];
522
523 } // ORDER * 4 flops
524 } // ORDER * ORDER * 4 flops
525 } // ORDER * ORDER * ORDER * 4 flops
526 }
527
528 for(int idxVals = 0 ; idxVals < nVals ; ++idxVals){
529
530 // get potential
531 FReal*const potentials = inParticles->getPotentials(idxVals,idxPot);
532 // add contribution to potential
533 potentials[idxPart] += (targetValue[idxVals]);
534
535 }// NVALS
536
537 } // idxLhs
538
539 } // N * (4 * ORDER * ORDER * ORDER + 9 * ORDER*(ORDER-1) ) flops
540}
