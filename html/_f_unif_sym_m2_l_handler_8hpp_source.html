28template < class FReal, int ORDER, typename MatrixKernelClass>
29static void precompute(const MatrixKernelClass *const MatrixKernel, const FReal CellWidth,
30 FComplex<FReal>* FC[343])
31{
32 // std::cout << "\nComputing 16 far-field interactions (l=" << ORDER << ", eps=" << Epsilon
33 // << ") for cells of width w = " << CellWidth << std::endl;
34
35 static const unsigned int nnodes = ORDER*ORDER*ORDER;
36
37 // interpolation points of source (Y) and target (X) cell
38 FPoint<FReal> X[nnodes], Y[nnodes];
39 // set roots of target cell (X)
40 FUnifTensor<FReal,ORDER>::setRoots(FPoint<FReal>(0.,0.,0.), CellWidth, X);
41 // temporary matrices
42 FReal *_C;
43 FComplex<FReal> *_FC;
44
45 // reduce storage from nnodes^2=order^6 to (2order-1)^3
46 const unsigned int rc = (2*ORDER-1)*(2*ORDER-1)*(2*ORDER-1);
47 _C = new FReal [rc];
48 _FC = new FComplex<FReal> [rc]; // TODO: do it in the non-sym version!!!
49
50 // init Discrete Fourier Transformator
51 const int dimfft = 1; // unidim FFT since fully circulant embedding
52 const int steps[dimfft] = {rc};
53// FDft Dft(rc);
54 FFft<FReal,dimfft> Dft(steps);
55
56 // reduce storage if real valued kernel
57 const unsigned int opt_rc = rc/2+1;
58
59 // get first column of K via permutation
60 unsigned int perm[rc];
61 for(unsigned int p=0; p<rc; ++p){
62 if(p<rc-1) perm[p]=p+1;
63 else perm[p]=p+1-rc;
64 }
65 unsigned int li,lj, mi,mj, ni,nj;
66 unsigned int idi, idj, ido;
67
68 // initialize timer
69 FTic time;
70 double overall_time(0.);
71 double elapsed_time(0.);
72
73 unsigned int counter = 0;
74 for (int i=2; i<=3; ++i) {
75 for (int j=0; j<=i; ++j) {
76 for (int k=0; k<=j; ++k) {
77
78 // set roots of source cell (Y)
79 const FPoint<FReal> cy(CellWidth*FReal(i), CellWidth*FReal(j), CellWidth*FReal(k));
80 FUnifTensor<FReal,ORDER>::setRoots(cy, CellWidth, Y);
81
82 // start timer
83 time.tic();
84
85 // evaluate m2l operator
86 ido=0;
87 for(unsigned int l=0; l<2*ORDER-1; ++l)
88 for(unsigned int m=0; m<2*ORDER-1; ++m)
89 for(unsigned int n=0; n<2*ORDER-1; ++n){
90
91 // l=0:(2*ORDER-1) => li-lj=-(ORDER-1):(ORDER-1)
92 // Convention:
93 // lj=ORDER-1 & li=0:ORDER-1 => li-lj=1-ORDER:0
94 // lj=1 & li=0:ORDER-1 => li-lj=1:ORDER-1
95 if(l<ORDER-1) lj=ORDER-1; else lj=0;
96 if(m<ORDER-1) mj=ORDER-1; else mj=0;
97 if(n<ORDER-1) nj=ORDER-1; else nj=0;
98 li=(l-(ORDER-1))+lj; mi=(m-(ORDER-1))+mj; ni=(n-(ORDER-1))+nj;
99 // deduce corresponding index of K[nnodes x nnodes]
100 idi=li*ORDER*ORDER + mi*ORDER + ni;
101 idj=lj*ORDER*ORDER + mj*ORDER + nj;
102
103 // store value at current position in C
104 // use permutation if DFT is used because
105 // the storage of the first column is required
106 // i.e. C[0] C[rc-1] C[rc-2] .. C[1] < WRONG!
107 // i.e. C[rc-1] C[0] C[1] .. C[rc-2] < RIGHT!
108// _C[counter*rc + ido]
109 _C[perm[ido]]
110 = MatrixKernel->evaluate(X[idi], Y[idj]);
111 ido++;
112 }
113
114 // Apply Discrete Fourier Transformation
115 Dft.applyDFT(_C,_FC);
116
117 // determine new index
118 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
119
120 // store
121 {
122 // allocate
123 assert(FC[idx]==NULL);
124 FC[idx] = new FComplex<FReal>[opt_rc];
125 FBlas::c_copy(opt_rc, reinterpret_cast<FReal*>(_FC),
126 reinterpret_cast<FReal*>(FC[idx]));
127 }
128
129 elapsed_time = time.tacAndElapsed();
130 overall_time += elapsed_time;
131
132 counter++;
133 }
134 }
135 }
136
137 std::cout << "The approximation of the " << counter
138 << " far-field interactions (sizeM2L= "
139 << counter*opt_rc*sizeof(FComplex<FReal>) << " B"
140 << ") took " << overall_time << "s\n" << std::endl;
141
142 // Free _C & _FC
143 delete [] _C;
144 delete [] _FC;
145}

186 {
187 // init all 343 item to zero, because effectively only 16 exist
188 for (unsigned int t=0; t<343; ++t)
189 K[t] = NULL;
190
191 // set permutation vector and indices
192 const FInterpSymmetries<ORDER> Symmetries;
193 for (int i=-3; i<=3; ++i)
194 for (int j=-3; j<=3; ++j)

196 const unsigned int idx = ((i+3) * 7 + (j+3)) * 7 + (k+3);
197 pindices[idx] = 0;

204 precompute<ORDER>(MatrixKernel, ReferenceCellWidth, K);
205 }

228template <int ORDER>
229class FUnifSymM2LHandler<ORDER, NON_HOMOGENEOUS>
230{
231 static const unsigned int nnodes = ORDER*ORDER*ORDER;
232
233 // Height of octree; needed only in the case of non-homogeneous kernel functions
234 const unsigned int TreeHeight;
235
236 // M2L operators for all levels in the octree
237 FComplex<FReal>*** K;
238
239public:
240
241 // permutation vectors and permutated indices
242 unsigned int pvectors[343][nnodes];
243 unsigned int pindices[343];
244
245
247 template <typename MatrixKernelClass>
248 FUnifSymM2LHandler(const MatrixKernelClass *const MatrixKernel,
249 const FReal RootCellWidth, const unsigned int inTreeHeight)
250 : TreeHeight(inTreeHeight)
251 {
252 // init all 343 item to zero, because effectively only 16 exist
253 K = new FComplex<FReal>** [TreeHeight];
254 K[0] = NULL; K[1] = NULL;
255 for (unsigned int l=2; l<TreeHeight; ++l) {
256 K[l] = new FComplex<FReal>* [343];
257 for (unsigned int t=0; t<343; ++t)
258 K[l][t] = NULL;
259 }
260
261
262 // set permutation vector and indices
263 const FInterpSymmetries<ORDER> Symmetries;
264 for (int i=-3; i<=3; ++i)
265 for (int j=-3; j<=3; ++j)
266 for (int k=-3; k<=3; ++k) {
267 const unsigned int idx = ((i+3) * 7 + (j+3)) * 7 + (k+3);
268 pindices[idx] = 0;
269 if (abs(i)>1 || abs(j)>1 || abs(k)>1)
270 pindices[idx] = Symmetries.getPermutationArrayAndIndex(i,j,k, pvectors[idx]);
271 }
272
273 // precompute 16 M2L operators at all levels having far-field interactions
274 FReal CellWidth = RootCellWidth / FReal(2.); // at level 1
275 CellWidth /= FReal(2.); // at level 2
276 for (unsigned int l=2; l<TreeHeight; ++l) {
277 precompute<ORDER>(MatrixKernel, CellWidth, K[l]);
278 CellWidth /= FReal(2.); // at level l+1
279 }
280 }
281
282
283
285 ~FUnifSymM2LHandler()
286 {
287 for (unsigned int l=0; l<TreeHeight; ++l) {
288 if (K[l]!=NULL) {
289 for (unsigned int t=0; t<343; ++t) if (K[l][t]!=NULL) delete [] K[l][t];
290 delete [] K[l];
291 }
292 }
293 delete [] K;
294 }
295
297 const FComplex<FReal> *const getK(const unsigned int l, const unsigned int t) const
298 { return K[l][t]; }
299
300};
