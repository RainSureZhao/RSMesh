45 void setNodeIdsDiff(unsigned int NodeIdsDiff[nnodes*nnodes])
46 {
47 unsigned int node_ids[nnodes][3];
48 ParentTensor::setNodeIds(node_ids);
49
50 for (unsigned int i=0; i<nnodes; ++i) {
51 for (unsigned int j=0; j<nnodes; ++j) {
52 // 0 <= id < 2*ORDER-1
53 unsigned int idl = node_ids[i][0]-node_ids[j][0]+ORDER-1;
54 unsigned int idm = node_ids[i][1]-node_ids[j][1]+ORDER-1;
55 unsigned int idn = node_ids[i][2]-node_ids[j][2]+ORDER-1;
56 NodeIdsDiff[i*nnodes+j]
57 = idn*(2*ORDER-1)*(2*ORDER-1) + idm*(2*ORDER-1) + idl;
58 }
59 }
60 }

67 static
68 void setNodeIdsPairs(unsigned int NodeIdsPairs[rc][2])
69 {
70
71 unsigned int li,lj,mi,mj,ni,nj;
72 unsigned int ido=0;
73
74 // If the diff i-j between indices is <0 (i.e. positive counter < ORDER-1)
75 // then source index is ORDER-1
76 // else it is the first
77 // we deduce the target index by simply considering i-j=counter-(ORDER-1)
78 for(unsigned int l=0; l<2*ORDER-1; ++l){
79
80 // l=0:(2*order-1) => li-lj=-(order-1):(order-1)
81 // Convention:
82 // lj=order-1 & li=0:order-1 => li-lj=1-order:0
83 // lj=1 & li=0:order-1 => li-lj=1:order-1
84 if(l<ORDER-1) lj=ORDER-1; else lj=0;
85 li=(l-(ORDER-1))+lj;
86
87 for(unsigned int m=0; m<2*ORDER-1; ++m){
88
89 if(m<ORDER-1) mj=ORDER-1; else mj=0;
90 mi=(m-(ORDER-1))+mj;
91
92 for(unsigned int n=0; n<2*ORDER-1; ++n){
93
94 if(n<ORDER-1) nj=ORDER-1; else nj=0;
95 ni=(n-(ORDER-1))+nj;
96
97 NodeIdsPairs[ido][0]=li*ORDER*ORDER + mi*ORDER + ni;
98 NodeIdsPairs[ido][1]=lj*ORDER*ORDER + mj*ORDER + nj;
99
100 ido++;
101 }
102 }
103 }
104
105 }
