98 void P2M(CellClass* const LeafCell,
99 const ContainerClass* const SourceParticles)
100 {
101 const FPoint<FReal> LeafCellCenter(AbstractBaseClass::getLeafCellCenter(LeafCell->getCoordinate()));
102 const FReal ExtendedLeafCellWidth(AbstractBaseClass::BoxWidthLeaf
103 + AbstractBaseClass::BoxWidthExtension);
104
105 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
106
107 // 1) apply Sy
108 AbstractBaseClass::Interpolator->applyP2M(LeafCellCenter, ExtendedLeafCellWidth,
109 LeafCell->getMultipole(idxV*nRhs), SourceParticles);
110
111 for(int idxRhs = 0 ; idxRhs < nRhs ; ++idxRhs){
112 // update multipole index
113 int idxMul = idxV*nRhs + idxRhs;
114
115 // 2) apply Discrete Fourier Transform
116 M2LHandler.applyZeroPaddingAndDFT(LeafCell->getMultipole(idxMul),
117 LeafCell->getTransformedMultipole(idxMul));
118
119 }
120 }// NVALS
121 }

124 void M2M(CellClass* const FRestrict ParentCell,
125 const CellClass*const FRestrict *const FRestrict ChildCells,
126 const int TreeLevel)
127 {
128 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
129 for(int idxRhs = 0 ; idxRhs < nRhs ; ++idxRhs){
130 // update multipole index
131 int idxMul = idxV*nRhs + idxRhs;
132
133 // 1) apply Sy
134 FBlas::scal(AbstractBaseClass::nnodes, FReal(0.), ParentCell->getMultipole(idxMul));
135 for (unsigned int ChildIndex=0; ChildIndex < 8; ++ChildIndex){
136 if (ChildCells[ChildIndex]){
137 AbstractBaseClass::Interpolator->applyM2M(ChildIndex,
138 ChildCells[ChildIndex]->getMultipole(idxMul),
139 ParentCell->getMultipole(idxMul),
140 TreeLevel/*Cell width extension specific*/);
141 }
142 }
143 // 2) Apply Discete Fourier Transform
144 M2LHandler.applyZeroPaddingAndDFT(ParentCell->getMultipole(idxMul),
145 ParentCell->getTransformedMultipole(idxMul));
146 }
147 }// NVALS

151 void M2L(CellClass* const FRestrict TargetCell, const CellClass* SourceCells[],
152 const int neighborPositions[], const int inSize, const int TreeLevel) override {
153 const FReal CellWidth(AbstractBaseClass::BoxWidth / FReal(FMath::pow(2, TreeLevel)));
154 const FReal ExtendedCellWidth(CellWidth + AbstractBaseClass::BoxWidthExtension);
155 const FReal scale(MatrixKernel->getScaleFactor(ExtendedCellWidth));
156
157 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
158 for (int idxLhs=0; idxLhs < nLhs; ++idxLhs){
159
160 // update local index
161 const int idxLoc = idxV*nLhs + idxLhs;
162
163 // load transformed local expansion
164 FComplex<FReal> *const TransformedLocalExpansion = TargetCell->getTransformedLocal(idxLoc);
165
166 // update idxRhs
167 const int idxRhs = idxLhs % nPV;
168
169 // update multipole index
170 const int idxMul = idxV*nRhs + idxRhs;
171
172 // get index in matrix kernel
173 const unsigned int d = MatrixKernel->getPosition(idxLhs);
174
175 for(int idxExistingNeigh = 0 ; idxExistingNeigh < inSize ; ++idxExistingNeigh){
176 const int idx = neighborPositions[idxExistingNeigh];
177
178 M2LHandler.applyFC(idx, TreeLevel, scale, d,
179 SourceCells[idxExistingNeigh]->getTransformedMultipole(idxMul),
180 TransformedLocalExpansion);
181
182 }
183 }// NLHS=NPOT*NPV
184 }// NVALS
185 }

188 void L2L(const CellClass* const FRestrict ParentCell,
189 CellClass* FRestrict *const FRestrict ChildCells,
190 const int TreeLevel)
191 {
192 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
193 for(int idxLhs = 0 ; idxLhs < nLhs ; ++idxLhs){
194 int idxLoc = idxV*nLhs + idxLhs;
195 // 1) Apply Inverse Discete Fourier Transform
196 M2LHandler.unapplyZeroPaddingAndDFT(ParentCell->getTransformedLocal(idxLoc),
197 const_cast<CellClass*>(ParentCell)->getLocal(idxLoc));
198 // 2) apply Sx
199 for (unsigned int ChildIndex=0; ChildIndex < 8; ++ChildIndex){
200 if (ChildCells[ChildIndex]){
201 AbstractBaseClass::Interpolator->applyL2L(ChildIndex,
202 ParentCell->getLocal(idxLoc),
203 ChildCells[ChildIndex]->getLocal(idxLoc),
204 TreeLevel/*Cell width extension specific*/);
205 }
206 }
207 }

211 void L2P(const CellClass* const LeafCell,
212 ContainerClass* const TargetParticles)
213 {
214 const FPoint<FReal> LeafCellCenter(AbstractBaseClass::getLeafCellCenter(LeafCell->getCoordinate()));
215 const FReal ExtendedLeafCellWidth(AbstractBaseClass::BoxWidthLeaf
216 + AbstractBaseClass::BoxWidthExtension);
217
218 for(int idxV = 0 ; idxV < NVALS ; ++idxV){
219 for(int idxLhs = 0 ; idxLhs < nLhs ; ++idxLhs){
220 int idxLoc = idxV*nLhs + idxLhs;
221 // 1) Apply Inverse Discete Fourier Transform
222 M2LHandler.unapplyZeroPaddingAndDFT(LeafCell->getTransformedLocal(idxLoc),
223 const_cast<CellClass*>(LeafCell)->getLocal(idxLoc));
224
225 }
226
227 // 2.a) apply Sx
228 AbstractBaseClass::Interpolator->applyL2P(LeafCellCenter, ExtendedLeafCellWidth,
229 LeafCell->getLocal(idxV*nLhs), TargetParticles);
230
231 // 2.b) apply Px (grad Sx)
232 AbstractBaseClass::Interpolator->applyL2PGradient(LeafCellCenter, ExtendedLeafCellWidth,
233 LeafCell->getLocal(idxV*nLhs), TargetParticles);
234
235 }// NVALS

238 void P2P(const FTreeCoordinate& inPosition,
239 ContainerClass* const FRestrict inTargets, const ContainerClass* const FRestrict inSources,
240 ContainerClass* const inNeighbors[], const int neighborPositions[],
241 const int inSize) override {
242 // Standard FMM separation criterion, i.e. max 27 neighbor clusters per leaf
243 if(LeafLevelSeparationCriterion==1) {
244 if(inTargets == inSources){
245 P2POuter(inPosition, inTargets, inNeighbors, neighborPositions, inSize);
246 DirectInteractionComputer<FReal, MatrixKernelClass::NCMP, NVALS>::P2PInner(inTargets,MatrixKernel);
247 }
248 else{
249 const ContainerClass* const srcPtr[1] = {inSources};
250 DirectInteractionComputer<FReal, MatrixKernelClass::NCMP, NVALS>::P2PRemote(inTargets,srcPtr,1,MatrixKernel);
251 DirectInteractionComputer<FReal, MatrixKernelClass::NCMP, NVALS>::P2PRemote(inTargets,inNeighbors,inSize,MatrixKernel);
252 }
253 }
254 // Nearfield interactions are only computed within the target leaf
255 else if(LeafLevelSeparationCriterion==0){
256 DirectInteractionComputer<FReal,MatrixKernelClass::NCMP, NVALS>::P2PRemote(inTargets,inNeighbors,inSize,MatrixKernel);
257 }
258 // If criterion equals -1 then no P2P need to be performed.
259 }
