26template < class FReal, int ORDER, class MatrixKernelClass>
27static void Compute(const MatrixKernelClass *const MatrixKernel,
28 const FReal CellWidth,
29 const FReal CellWidthExtension,
30 FComplex<FReal>** &FC,
31 const int SeparationCriterion = 1)
32{
33 // dimensions of operators
34 const unsigned int order = ORDER;
35 const unsigned int nnodes = TensorTraits<ORDER>::nnodes;
36 const unsigned int ninteractions = 316+26*(SeparationCriterion<1 ? 1 : 0) + 1*(SeparationCriterion<0 ? 1 : 0);
37 const unsigned int ncmp = MatrixKernelClass::NCMP;
38
39 // utils
40 typedef FUnifTensor<FReal,ORDER> TensorType;
41
42 // allocate memory and store compressed M2L operators
43 for (unsigned int d=0; d<ncmp; ++d)
44 if (FC[d]) throw std::runtime_error("M2L operators are already set");
45
46 // interpolation points of source (Y) and target (X) cell
47 FPoint<FReal> X[nnodes], Y[nnodes];
48 // set roots of target cell (X)
49 const FReal ExtendedCellWidth(CellWidth+CellWidthExtension);
50 FUnifTensor<FReal,order>::setRoots(FPoint<FReal>(0.,0.,0.), ExtendedCellWidth, X);
51
52 // allocate memory and compute 316 m2l operators
53 FReal** _C;
54 FComplex<FReal>** _FC;
55
56 // reduce storage from nnodes^2=order^6 to (2order-1)^3
57 const unsigned int rc = (2*order-1)*(2*order-1)*(2*order-1);
58 _C = new FReal* [ncmp];
59 _FC = new FComplex<FReal>* [ncmp];
60 for (unsigned int d=0; d<ncmp; ++d)
61 _C[d] = new FReal[rc];
62 for (unsigned int d=0; d<ncmp; ++d)
63 _FC[d] = new FComplex<FReal>[rc * ninteractions];
64
65 // initialize root node ids pairs
66 unsigned int node_ids_pairs[rc][2];
67 TensorType::setNodeIdsPairs(node_ids_pairs);
68
69 // init Discrete Fourier Transformator
70 const int dimfft = 1; // unidim FFT since fully circulant embedding
71 FFftw<FReal,FComplex<FReal>,dimfft> Dft(rc);
72
73 // get first column of K via permutation
74 unsigned int perm[rc];
75 TensorType::setStoragePermutation(perm);
76
77 // Allocate intermediate interaction block
78 FReal* block = new FReal[ncmp];
79
80 unsigned int counter = 0;
81 for (int i=-3; i<=3; ++i) {
82 for (int j=-3; j<=3; ++j) {
83 for (int k=-3; k<=3; ++k) {
84 if (abs(i)>SeparationCriterion || abs(j)>SeparationCriterion || abs(k)>SeparationCriterion) {
85 // set roots of source cell (Y)
86 const FPoint<FReal> cy(CellWidth*FReal(i), CellWidth*FReal(j), CellWidth*FReal(k));
87 FUnifTensor<FReal,order>::setRoots(cy, ExtendedCellWidth, Y);
88 // evaluate m2l operator
89 unsigned int ido=0;
90 for(unsigned int l=0; l<2*order-1; ++l)
91 for(unsigned int m=0; m<2*order-1; ++m)
92 for(unsigned int n=0; n<2*order-1; ++n){
93 // Compute current M2L interaction (block matrix)
94 MatrixKernel->evaluateBlock(X[node_ids_pairs[ido][0]],
95 Y[node_ids_pairs[ido][1]],
96 block);
97 // store value at current position in C
98 // use permutation if DFT is used because
99 // the storage of the first column is required
100 // i.e. C[0] C[rc-1] C[rc-2] .. C[1] < WRONG!
101 // i.e. C[rc-1] C[0] C[1] .. C[rc-2] < RIGHT!
102 // _C[counter*rc + ido]
103 for (unsigned int d=0; d<ncmp; ++d)
104 _C[d][perm[ido]] = block[d];
105 ido++;
106 }
107
108 // Apply Discrete Fourier Transformation
109 for (unsigned int d=0; d<ncmp; ++d)
110 Dft.applyDFT(_C[d],_FC[d]+counter*rc);
111
112 // increment interaction counter
113 counter++;
114 }
115 }
116 }
117 }
118 if (counter != ninteractions)
119 throw std::runtime_error("Number of interactions must correspond to 316");
120
121 // Free block
122 delete [] block;
123
124 // Free _C
125 for (unsigned int d=0; d<ncmp; ++d)
126 delete [] _C[d];
127
128 // store FC
129 counter = 0;
130 // reduce storage if real valued kernel
131 const unsigned int opt_rc = rc/2+1;
132 // allocate M2L
133 for (unsigned int d=0; d<ncmp; ++d)
134 FC[d] = new FComplex<FReal>[343 * opt_rc];
135
136 for (int i=-3; i<=3; ++i)
137 for (int j=-3; j<=3; ++j)
138 for (int k=-3; k<=3; ++k) {
139 const unsigned int idx = (i+3)*7*7 + (j+3)*7 + (k+3);
140 if (abs(i)>SeparationCriterion || abs(j)>SeparationCriterion || abs(k)>SeparationCriterion) {
141 for (unsigned int d=0; d<ncmp; ++d)
142 FBlas::c_copy(opt_rc, reinterpret_cast<FReal*>(_FC[d] + counter*rc),
143 reinterpret_cast<FReal*>(FC[d] + idx*opt_rc));
144 counter++;
145 } else {
146 for (unsigned int d=0; d<ncmp; ++d)
147 FBlas::c_setzero(opt_rc, reinterpret_cast<FReal*>(FC[d] + idx*opt_rc));
148 }
149 }
150
151 if (counter != ninteractions)
152 throw std::runtime_error("Number of interactions must correspond to 316");
153 for (unsigned int d=0; d<ncmp; ++d)
154 delete [] _FC[d];
155}

250 void ComputeAndSet(const MatrixKernelClass *const MatrixKernel)
251 {
252 // measure time
253 FTic time; time.tic();
254
255 // check if aready set
256 for (unsigned int d=0; d<ncmp; ++d)
257 if (FC[d]) throw std::runtime_error("M2L operator already set");
258
259 // Compute matrix of interactions
260 // reference cell width is arbitrarly set to 2.
261 // but it NEEDS to match the numerator of the scale factor in matrix kernel!

264 Compute<FReal,order>(MatrixKernel,ReferenceCellWidth, 0., FC, LeafLevelSeparationCriterion);

271 << time.tacAndElapsed() << "sec." << std::endl;

437 void ComputeAndSet(const MatrixKernelClass *const MatrixKernel)
438 {
439 // measure time
440 FTic time; time.tic();
441
442 // Compute matrix of interactions at each level !! (since non homog)
443 FReal CellWidth = RootCellWidth / FReal(2.); // at level 1
444 CellWidth /= FReal(2.); // at level 2
445 for (unsigned int l=2; l<TreeHeight; ++l) {
446 // Determine separation criteria wrt level
447 const int SeparationCriterion = (l != TreeHeight-1 ? 1 : LeafLevelSeparationCriterion);
448 // check if already set
449 for (unsigned int d=0; d<ncmp; ++d)
450 if (FC[l][d]) throw std::runtime_error("M2L operator already set");
451 Compute<FReal,order>(MatrixKernel,CellWidth,CellWidthExtension,FC[l],SeparationCriterion);
452 CellWidth /= FReal(2.); // at level l+1
453 }
454
455 // Compute memory usage
456 unsigned long sizeM2L = (TreeHeight-2)*343*ncmp*opt_rc*sizeof(FComplex<FReal>);
457
458 // write info
459 std::cout << "Compute and set M2L operators ("<< long(sizeM2L) <<" B) in "
460 << time.tacAndElapsed() << "sec." << std::endl;
461 }
